89/// Salsa query. This will look for lang items in a specific crate.
90#[salsa_macros::tracked(returns(ref))]
91pub fn crate_lang_items(db: &dyn DefDatabase, krate: Crate) -> Option<Box<LangItems>> {
92    let _p = tracing::info_span!("crate_lang_items_query").entered();
93
94    let mut lang_items = LangItems::default();
95
96    let crate_def_map = crate_def_map(db, krate);
97
98    for (_, module_data) in crate_def_map.modules() {
99        for impl_def in module_data.scope.impls() {
100            lang_items.collect_lang_item(db, impl_def, LangItemTarget::ImplDef);
101            for &(_, assoc) in impl_def.impl_items(db).items.iter() {
102                match assoc {
103                    AssocItemId::FunctionId(f) => {
104                        lang_items.collect_lang_item(db, f, LangItemTarget::Function)
105                    }
106                    AssocItemId::TypeAliasId(t) => {
107                        lang_items.collect_lang_item(db, t, LangItemTarget::TypeAlias)
108                    }
109                    AssocItemId::ConstId(_) => (),
110                }
111            }
112        }
113
114        for def in module_data.scope.declarations() {
115            match def {
116                ModuleDefId::TraitId(trait_) => {
117                    lang_items.collect_lang_item(db, trait_, LangItemTarget::Trait);
118                    TraitItems::query(db, trait_).items.iter().for_each(|&(_, assoc_id)| {
119                        match assoc_id {
120                            AssocItemId::FunctionId(f) => {
121                                lang_items.collect_lang_item(db, f, LangItemTarget::Function);
122                            }
123                            AssocItemId::TypeAliasId(alias) => {
124                                lang_items.collect_lang_item(db, alias, LangItemTarget::TypeAlias)
125                            }
126                            AssocItemId::ConstId(_) => {}
127                        }
128                    });
129                }
130                ModuleDefId::AdtId(AdtId::EnumId(e)) => {
131                    lang_items.collect_lang_item(db, e, LangItemTarget::EnumId);
132                    e.enum_variants(db).variants.iter().for_each(|&(id, _, _)| {
133                        lang_items.collect_lang_item(db, id, LangItemTarget::EnumVariant);
134                    });
135                }
136                ModuleDefId::AdtId(AdtId::StructId(s)) => {
137                    lang_items.collect_lang_item(db, s, LangItemTarget::Struct);
138                }
139                ModuleDefId::AdtId(AdtId::UnionId(u)) => {
140                    lang_items.collect_lang_item(db, u, LangItemTarget::Union);
141                }
142                ModuleDefId::FunctionId(f) => {
143                    lang_items.collect_lang_item(db, f, LangItemTarget::Function);
144                }
145                ModuleDefId::StaticId(s) => {
146                    lang_items.collect_lang_item(db, s, LangItemTarget::Static);
147                }
148                ModuleDefId::TypeAliasId(t) => {
149                    lang_items.collect_lang_item(db, t, LangItemTarget::TypeAlias);
150                }
151                _ => {}
152            }
153        }
154    }
155
156    if lang_items.items.is_empty() { None } else { Some(Box::new(lang_items)) }
157}

159/// Salsa query. Look for a lang item, starting from the specified crate and recursively
160/// traversing its dependencies.
161#[salsa_macros::tracked]
162pub fn lang_item(
163    db: &dyn DefDatabase,
164    start_crate: Crate,
165    item: LangItem,
166) -> Option<LangItemTarget> {
167    let _p = tracing::info_span!("lang_item_query").entered();
168    if let Some(target) =
169        crate_lang_items(db, start_crate).as_ref().and_then(|it| it.items.get(&item).copied())
170    {
171        return Some(target);
172    }
173    start_crate.data(db).dependencies.iter().find_map(|dep| lang_item(db, dep.crate_id, item))
174}

176#[derive(Default, Debug, Clone, PartialEq, Eq)]
177pub struct LangItems {
178    items: FxHashMap<LangItem, LangItemTarget>,
179}
180
181impl LangItems {
182    pub fn target(&self, item: LangItem) -> Option<LangItemTarget> {
183        self.items.get(&item).copied()
184    }
185
186    fn collect_lang_item<T>(
187        &mut self,
188        db: &dyn DefDatabase,
189        item: T,
190        constructor: fn(T) -> LangItemTarget,
191    ) where
192        T: Into<AttrDefId> + Copy,
193    {
194        let _p = tracing::info_span!("collect_lang_item").entered();
195        if let Some(lang_item) = lang_attr(db, item.into()) {
196            self.items.entry(lang_item).or_insert_with(|| constructor(item));
197        }
198    }
199}

201pub(crate) fn lang_attr(db: &dyn DefDatabase, item: AttrDefId) -> Option<LangItem> {
202    db.attrs(item).lang_item()
203}

205pub(crate) fn notable_traits_in_deps(db: &dyn DefDatabase, krate: Crate) -> Arc<[Arc<[TraitId]>]> {
206    let _p = tracing::info_span!("notable_traits_in_deps", ?krate).entered();
207    Arc::from_iter(
208        db.transitive_deps(krate).into_iter().filter_map(|krate| db.crate_notable_traits(krate)),
209    )
210}

212pub(crate) fn crate_notable_traits(db: &dyn DefDatabase, krate: Crate) -> Option<Arc<[TraitId]>> {
213    let _p = tracing::info_span!("crate_notable_traits", ?krate).entered();
214
215    let mut traits = Vec::new();
216
217    let crate_def_map = crate_def_map(db, krate);
218
219    for (_, module_data) in crate_def_map.modules() {
220        for def in module_data.scope.declarations() {
221            if let ModuleDefId::TraitId(trait_) = def
222                && db.attrs(trait_.into()).has_doc_notable_trait()
223            {
224                traits.push(trait_);
225            }
226        }
227    }
228
229    if traits.is_empty() { None } else { Some(traits.into_iter().collect()) }
230}

238macro_rules! language_item_table {
239    (
240        $( $(#[$attr:meta])* $variant:ident, $module:ident :: $name:ident, $method:ident, $target:expr, $generics:expr; )*
241    ) => {
242
243        /// A representation of all the valid language items in Rust.
244        #[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
245        pub enum LangItem {
246            $(
247                #[doc = concat!("The `", stringify!($name), "` lang item.")]
248                $(#[$attr])*
249                $variant,
250            )*
251        }
252
253        impl LangItem {
254            pub fn name(self) -> &'static str {
255                match self {
256                    $( LangItem::$variant => stringify!($name), )*
257                }
258            }
259
260            /// Opposite of [`LangItem::name`]
261            pub fn from_symbol(sym: &Symbol) -> Option<Self> {
262                match sym {
263                    $(sym if *sym == $module::$name => Some(LangItem::$variant), )*
264                    _ => None,
265                }
266            }
267        }
268    }
269}

271impl LangItem {
272    pub fn resolve_function(self, db: &dyn DefDatabase, start_crate: Crate) -> Option<FunctionId> {
273        lang_item(db, start_crate, self).and_then(|t| t.as_function())
274    }
275
276    pub fn resolve_trait(self, db: &dyn DefDatabase, start_crate: Crate) -> Option<TraitId> {
277        lang_item(db, start_crate, self).and_then(|t| t.as_trait())
278    }
279
280    pub fn resolve_enum(self, db: &dyn DefDatabase, start_crate: Crate) -> Option<EnumId> {
281        lang_item(db, start_crate, self).and_then(|t| t.as_enum())
282    }
283
284    pub fn resolve_type_alias(
285        self,
286        db: &dyn DefDatabase,
287        start_crate: Crate,
288    ) -> Option<TypeAliasId> {
289        lang_item(db, start_crate, self).and_then(|t| t.as_type_alias())
290    }
291
292    /// Opposite of [`LangItem::name`]
293    pub fn from_name(name: &hir_expand::name::Name) -> Option<Self> {
294        Self::from_symbol(name.symbol())
295    }
296
297    pub fn path(&self, db: &dyn DefDatabase, start_crate: Crate) -> Option<Path> {
298        let t = lang_item(db, start_crate, *self)?;
299        Some(Path::LangItem(t, None))
300    }
301
302    pub fn ty_rel_path(&self, db: &dyn DefDatabase, start_crate: Crate, seg: Name) -> Option<Path> {
303        let t = lang_item(db, start_crate, *self)?;
304        Some(Path::LangItem(t, Some(seg)))
305    }
306}
