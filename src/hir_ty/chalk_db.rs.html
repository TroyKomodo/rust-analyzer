85    fn discriminant_type(&self, ty: chalk_ir::Ty<Interner>) -> chalk_ir::Ty<Interner> {
86        if let chalk_ir::TyKind::Adt(id, _) = ty.kind(Interner)
87            && let hir_def::AdtId::EnumId(e) = id.0
88        {
89            let enum_data = self.db.enum_signature(e);
90            let ty = enum_data.repr.unwrap_or_default().discr_type();
91            return chalk_ir::TyKind::Scalar(match ty {
92                hir_def::layout::IntegerType::Pointer(is_signed) => match is_signed {
93                    true => chalk_ir::Scalar::Int(chalk_ir::IntTy::Isize),
94                    false => chalk_ir::Scalar::Uint(chalk_ir::UintTy::Usize),
95                },
96                hir_def::layout::IntegerType::Fixed(size, is_signed) => match is_signed {
97                    true => chalk_ir::Scalar::Int(match size {
98                        hir_def::layout::Integer::I8 => chalk_ir::IntTy::I8,
99                        hir_def::layout::Integer::I16 => chalk_ir::IntTy::I16,
100                        hir_def::layout::Integer::I32 => chalk_ir::IntTy::I32,
101                        hir_def::layout::Integer::I64 => chalk_ir::IntTy::I64,
102                        hir_def::layout::Integer::I128 => chalk_ir::IntTy::I128,
103                    }),
104                    false => chalk_ir::Scalar::Uint(match size {
105                        hir_def::layout::Integer::I8 => chalk_ir::UintTy::U8,
106                        hir_def::layout::Integer::I16 => chalk_ir::UintTy::U16,
107                        hir_def::layout::Integer::I32 => chalk_ir::UintTy::U32,
108                        hir_def::layout::Integer::I64 => chalk_ir::UintTy::U64,
109                        hir_def::layout::Integer::I128 => chalk_ir::UintTy::U128,
110                    }),
111                },
112            })
113            .intern(Interner);
114        }
115        chalk_ir::TyKind::Scalar(chalk_ir::Scalar::Uint(chalk_ir::UintTy::U8)).intern(Interner)
116    }

128    fn impls_for_trait(
129        &self,
130        trait_id: TraitId,
131        parameters: &[chalk_ir::GenericArg<Interner>],
132        binders: &CanonicalVarKinds<Interner>,
133    ) -> Vec<ImplId> {
134        debug!("impls_for_trait {:?}", trait_id);
135        let trait_: hir_def::TraitId = from_chalk_trait_id(trait_id);
136
137        let ty: Ty = parameters[0].assert_ty_ref(Interner).clone();
138
139        fn binder_kind(
140            ty: &Ty,
141            binders: &CanonicalVarKinds<Interner>,
142        ) -> Option<chalk_ir::TyVariableKind> {
143            if let TyKind::BoundVar(bv) = ty.kind(Interner) {
144                let binders = binders.as_slice(Interner);
145                if bv.debruijn == DebruijnIndex::INNERMOST
146                    && let chalk_ir::VariableKind::Ty(tk) = binders[bv.index].kind
147                {
148                    return Some(tk);
149                }
150            }
151            None
152        }
153
154        let self_ty_fp = TyFingerprint::for_trait_impl(&ty);
155        let fps: &[TyFingerprint] = match binder_kind(&ty, binders) {
156            Some(chalk_ir::TyVariableKind::Integer) => &ALL_INT_FPS,
157            Some(chalk_ir::TyVariableKind::Float) => &ALL_FLOAT_FPS,
158            _ => self_ty_fp.as_slice(),
159        };
160
161        let id_to_chalk = |id: hir_def::ImplId| id.to_chalk(self.db);
162
163        let mut result = vec![];
164        if fps.is_empty() {
165            debug!("Unrestricted search for {:?} impls...", trait_);
166            _ = self.for_trait_impls(trait_, self_ty_fp, |impls| {
167                result.extend(impls.for_trait(trait_).map(id_to_chalk));
168                ControlFlow::Continue(())
169            });
170        } else {
171            _ =
172                self.for_trait_impls(trait_, self_ty_fp, |impls| {
173                    result.extend(fps.iter().flat_map(move |fp| {
174                        impls.for_trait_and_self_ty(trait_, *fp).map(id_to_chalk)
175                    }));
176                    ControlFlow::Continue(())
177                });
178        };
179
180        debug!("impls_for_trait returned {} impls", result.len());
181        result
182    }

184    fn impl_provided_for(&self, auto_trait_id: TraitId, kind: &chalk_ir::TyKind<Interner>) -> bool {
185        debug!("impl_provided_for {:?}, {:?}", auto_trait_id, kind);
186
187        let trait_id = from_chalk_trait_id(auto_trait_id);
188        let self_ty = kind.clone().intern(Interner);
189        // We cannot filter impls by `TyFingerprint` for the following types:
190        let self_ty_fp = match kind {
191            // because we need to find any impl whose Self type is a ref with the same mutability
192            // (we don't care about the inner type).
193            TyKind::Ref(..) => None,
194            // because we need to find any impl whose Self type is a tuple with the same arity.
195            TyKind::Tuple(..) => None,
196            _ => TyFingerprint::for_trait_impl(&self_ty),
197        };
198
199        let check_kind = |impl_id| {
200            let impl_self_ty = self.db.impl_self_ty(impl_id);
201            // NOTE(skip_binders): it's safe to skip binders here as we don't check substitutions.
202            let impl_self_kind = impl_self_ty.skip_binders().kind(Interner);
203
204            match (kind, impl_self_kind) {
205                (TyKind::Adt(id_a, _), TyKind::Adt(id_b, _)) => id_a == id_b,
206                (TyKind::AssociatedType(id_a, _), TyKind::AssociatedType(id_b, _)) => id_a == id_b,
207                (TyKind::Scalar(scalar_a), TyKind::Scalar(scalar_b)) => scalar_a == scalar_b,
208                (TyKind::Error, TyKind::Error)
209                | (TyKind::Str, TyKind::Str)
210                | (TyKind::Slice(_), TyKind::Slice(_))
211                | (TyKind::Never, TyKind::Never)
212                | (TyKind::Array(_, _), TyKind::Array(_, _)) => true,
213                (TyKind::Tuple(arity_a, _), TyKind::Tuple(arity_b, _)) => arity_a == arity_b,
214                (TyKind::OpaqueType(id_a, _), TyKind::OpaqueType(id_b, _)) => id_a == id_b,
215                (TyKind::FnDef(id_a, _), TyKind::FnDef(id_b, _)) => id_a == id_b,
216                (TyKind::Ref(id_a, _, _), TyKind::Ref(id_b, _, _))
217                | (TyKind::Raw(id_a, _), TyKind::Raw(id_b, _)) => id_a == id_b,
218                (TyKind::Closure(id_a, _), TyKind::Closure(id_b, _)) => id_a == id_b,
219                (TyKind::Coroutine(id_a, _), TyKind::Coroutine(id_b, _))
220                | (TyKind::CoroutineWitness(id_a, _), TyKind::CoroutineWitness(id_b, _)) => {
221                    id_a == id_b
222                }
223                (TyKind::Foreign(id_a), TyKind::Foreign(id_b)) => id_a == id_b,
224                (_, _) => false,
225            }
226        };
227
228        if let Some(fp) = self_ty_fp {
229            self.for_trait_impls(trait_id, self_ty_fp, |impls| {
230                match impls.for_trait_and_self_ty(trait_id, fp).any(check_kind) {
231                    true => ControlFlow::Break(()),
232                    false => ControlFlow::Continue(()),
233                }
234            })
235        } else {
236            self.for_trait_impls(trait_id, self_ty_fp, |impls| {
237                match impls.for_trait(trait_id).any(check_kind) {
238                    true => ControlFlow::Break(()),
239                    false => ControlFlow::Continue(()),
240                }
241            })
242        }
243        .is_break()
244    }

286    fn opaque_ty_data(&self, id: chalk_ir::OpaqueTyId<Interner>) -> Arc<OpaqueTyDatum> {
287        let full_id = self.db.lookup_intern_impl_trait_id(id.into());
288        let bound = match full_id {
289            crate::ImplTraitId::ReturnTypeImplTrait(func, idx) => {
290                let datas = self
291                    .db
292                    .return_type_impl_traits(func)
293                    .expect("impl trait id without impl traits");
294                let (datas, binders) = (*datas).as_ref().into_value_and_skipped_binders();
295                let data = &datas.impl_traits[idx];
296                let bound = OpaqueTyDatumBound {
297                    bounds: make_single_type_binders(data.bounds.skip_binders().to_vec()),
298                    where_clauses: chalk_ir::Binders::empty(Interner, vec![]),
299                };
300                chalk_ir::Binders::new(binders, bound)
301            }
302            crate::ImplTraitId::TypeAliasImplTrait(alias, idx) => {
303                let datas = self
304                    .db
305                    .type_alias_impl_traits(alias)
306                    .expect("impl trait id without impl traits");
307                let (datas, binders) = (*datas).as_ref().into_value_and_skipped_binders();
308                let data = &datas.impl_traits[idx];
309                let bound = OpaqueTyDatumBound {
310                    bounds: make_single_type_binders(data.bounds.skip_binders().to_vec()),
311                    where_clauses: chalk_ir::Binders::empty(Interner, vec![]),
312                };
313                chalk_ir::Binders::new(binders, bound)
314            }
315            crate::ImplTraitId::AsyncBlockTypeImplTrait(..) => {
316                if let Some((future_trait, future_output)) =
317                    LangItem::Future.resolve_trait(self.db, self.krate).and_then(|trait_| {
318                        let alias = trait_
319                            .trait_items(self.db)
320                            .associated_type_by_name(&Name::new_symbol_root(sym::Output))?;
321                        Some((trait_, alias))
322                    })
323                {
324                    // Making up Symbol’s value as variable is void: AsyncBlock<T>:
325                    //
326                    // |--------------------OpaqueTyDatum-------------------|
327                    //        |-------------OpaqueTyDatumBound--------------|
328                    // for<T> <Self> [Future<Self>, Future::Output<Self> = T]
329                    //     ^1  ^0            ^0                    ^0      ^1
330                    let impl_bound = WhereClause::Implemented(TraitRef {
331                        trait_id: to_chalk_trait_id(future_trait),
332                        // Self type as the first parameter.
333                        substitution: Substitution::from1(
334                            Interner,
335                            TyKind::BoundVar(BoundVar {
336                                debruijn: DebruijnIndex::INNERMOST,
337                                index: 0,
338                            })
339                            .intern(Interner),
340                        ),
341                    });
342                    let mut binder = vec![];
343                    binder.push(crate::wrap_empty_binders(impl_bound));
344                    let sized_trait = LangItem::Sized.resolve_trait(self.db, self.krate);
345                    if let Some(sized_trait_) = sized_trait {
346                        let sized_bound = WhereClause::Implemented(TraitRef {
347                            trait_id: to_chalk_trait_id(sized_trait_),
348                            // Self type as the first parameter.
349                            substitution: Substitution::from1(
350                                Interner,
351                                TyKind::BoundVar(BoundVar {
352                                    debruijn: DebruijnIndex::INNERMOST,
353                                    index: 0,
354                                })
355                                .intern(Interner),
356                            ),
357                        });
358                        binder.push(crate::wrap_empty_binders(sized_bound));
359                    }
360                    let proj_bound = WhereClause::AliasEq(AliasEq {
361                        alias: AliasTy::Projection(ProjectionTy {
362                            associated_ty_id: to_assoc_type_id(future_output),
363                            // Self type as the first parameter.
364                            substitution: Substitution::from1(
365                                Interner,
366                                TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, 0))
367                                    .intern(Interner),
368                            ),
369                        }),
370                        // The parameter of the opaque type.
371                        ty: TyKind::BoundVar(BoundVar { debruijn: DebruijnIndex::ONE, index: 0 })
372                            .intern(Interner),
373                    });
374                    binder.push(crate::wrap_empty_binders(proj_bound));
375                    let bound = OpaqueTyDatumBound {
376                        bounds: make_single_type_binders(binder),
377                        where_clauses: chalk_ir::Binders::empty(Interner, vec![]),
378                    };
379                    // The opaque type has 1 parameter.
380                    make_single_type_binders(bound)
381                } else {
382                    // If failed to find Symbol’s value as variable is void: Future::Output, return empty bounds as fallback.
383                    let bound = OpaqueTyDatumBound {
384                        bounds: chalk_ir::Binders::empty(Interner, vec![]),
385                        where_clauses: chalk_ir::Binders::empty(Interner, vec![]),
386                    };
387                    // The opaque type has 1 parameter.
388                    make_single_type_binders(bound)
389                }
390            }
391        };
392
393        Arc::new(OpaqueTyDatum { opaque_ty_id: id, bound })
394    }

477    fn coroutine_datum(
478        &self,
479        id: chalk_ir::CoroutineId<Interner>,
480    ) -> Arc<chalk_solve::rust_ir::CoroutineDatum<Interner>> {
481        let InternedCoroutine(parent, expr) = self.db.lookup_intern_coroutine(id.into());
482
483        // We fill substitution with unknown type, because we only need to know whether the generic
484        // params are types or consts to build `Binders` and those being filled up are for
485        // `resume_type`, `yield_type`, and `return_type` of the coroutine in question.
486        let subst = TyBuilder::subst_for_coroutine(self.db, parent).fill_with_unknown().build();
487
488        let len = subst.len(Interner);
489        let input_output = rust_ir::CoroutineInputOutputDatum {
490            resume_type: TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, len - 3))
491                .intern(Interner),
492            yield_type: TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, len - 2))
493                .intern(Interner),
494            return_type: TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, len - 1))
495                .intern(Interner),
496            // FIXME: calculate upvars
497            upvars: vec![],
498        };
499
500        let it = subst
501            .iter(Interner)
502            .map(|it| it.constant(Interner).map(|c| c.data(Interner).ty.clone()));
503        let input_output = crate::make_type_and_const_binders(it, input_output);
504
505        let movability = match self.db.body(parent)[expr] {
506            hir_def::hir::Expr::Closure {
507                closure_kind: hir_def::hir::ClosureKind::Coroutine(movability),
508                ..
509            } => movability,
510            _ => unreachable!("non coroutine expression interned as coroutine"),
511        };
512        let movability = match movability {
513            Movability::Static => rust_ir::Movability::Static,
514            Movability::Movable => rust_ir::Movability::Movable,
515        };
516
517        Arc::new(rust_ir::CoroutineDatum { movability, input_output })
518    }
519    fn coroutine_witness_datum(
520        &self,
521        id: chalk_ir::CoroutineId<Interner>,
522    ) -> Arc<chalk_solve::rust_ir::CoroutineWitnessDatum<Interner>> {
523        // FIXME: calculate inner types
524        let inner_types =
525            rust_ir::CoroutineWitnessExistential { types: wrap_empty_binders(vec![]) };
526
527        let InternedCoroutine(parent, _) = self.db.lookup_intern_coroutine(id.into());
528        // See the comment in `coroutine_datum()` for unknown types.
529        let subst = TyBuilder::subst_for_coroutine(self.db, parent).fill_with_unknown().build();
530        let it = subst
531            .iter(Interner)
532            .map(|it| it.constant(Interner).map(|c| c.data(Interner).ty.clone()));
533        let inner_types = crate::make_type_and_const_binders(it, inner_types);
534
535        Arc::new(rust_ir::CoroutineWitnessDatum { inner_types })
536    }

539        &self.db
540    }

543impl ChalkContext<'_> {
544    fn edition(&self) -> Edition {
545        self.krate.data(self.db).edition
546    }
547
548    fn for_trait_impls(
549        &self,
550        trait_id: hir_def::TraitId,
551        self_ty_fp: Option<TyFingerprint>,
552        mut f: impl FnMut(&TraitImpls) -> ControlFlow<()>,
553    ) -> ControlFlow<()> {
554        // Note: Since we're using `impls_for_trait` and `impl_provided_for`,
555        // only impls where the trait can be resolved should ever reach Chalk.
556        // `impl_datum` relies on that and will panic if the trait can't be resolved.
557        let in_deps = self.db.trait_impls_in_deps(self.krate);
558        let in_self = self.db.trait_impls_in_crate(self.krate);
559        let trait_module = trait_id.module(self.db);
560        let type_module = match self_ty_fp {
561            Some(TyFingerprint::Adt(adt_id)) => Some(adt_id.module(self.db)),
562            Some(TyFingerprint::ForeignType(type_id)) => {
563                Some(from_foreign_def_id(type_id).module(self.db))
564            }
565            Some(TyFingerprint::Dyn(trait_id)) => Some(trait_id.module(self.db)),
566            _ => None,
567        };
568
569        let mut def_blocks =
570            [trait_module.containing_block(), type_module.and_then(|it| it.containing_block())];
571
572        let block_impls = iter::successors(self.block, |&block_id| {
573            cov_mark::hit!(block_local_impls);
574            block_id.loc(self.db).module.containing_block()
575        })
576        .inspect(|&block_id| {
577            // make sure we don't search the same block twice
578            def_blocks.iter_mut().for_each(|block| {
579                if *block == Some(block_id) {
580                    *block = None;
581                }
582            });
583        })
584        .filter_map(|block_id| self.db.trait_impls_in_block(block_id));
585        f(&in_self)?;
586        for it in in_deps.iter().map(ops::Deref::deref) {
587            f(it)?;
588        }
589        for it in block_impls {
590            f(&it)?;
591        }
592        for it in def_blocks.into_iter().flatten().filter_map(|it| self.db.trait_impls_in_block(it))
593        {
594            f(&it)?;
595        }
596        ControlFlow::Continue(())
597    }
598}

622pub(crate) fn associated_ty_data_query(
623    db: &dyn HirDatabase,
624    type_alias: TypeAliasId,
625) -> Arc<AssociatedTyDatum> {
626    debug!("associated_ty_data {:?}", type_alias);
627    let trait_ = match type_alias.lookup(db).container {
628        ItemContainerId::TraitId(t) => t,
629        _ => panic!("associated type not in trait"),
630    };
631
632    // Lower bounds -- we could/should maybe move this to a separate query in `lower`
633    let type_alias_data = db.type_alias_signature(type_alias);
634    let generic_params = generics(db, type_alias.into());
635    let resolver = hir_def::resolver::HasResolver::resolver(type_alias, db);
636    let mut ctx = crate::TyLoweringContext::new(
637        db,
638        &resolver,
639        &type_alias_data.store,
640        type_alias.into(),
641        LifetimeElisionKind::AnonymousReportError,
642    )
643    .with_type_param_mode(crate::lower::ParamLoweringMode::Variable);
644
645    let trait_subst = TyBuilder::subst_for_def(db, trait_, None)
646        .fill_with_bound_vars(crate::DebruijnIndex::INNERMOST, 0)
647        .build();
648    let pro_ty = TyBuilder::assoc_type_projection(db, type_alias, Some(trait_subst))
649        .fill_with_bound_vars(
650            crate::DebruijnIndex::INNERMOST,
651            generic_params.parent_generics().map_or(0, |it| it.len()),
652        )
653        .build();
654    let self_ty = TyKind::Alias(AliasTy::Projection(pro_ty)).intern(Interner);
655
656    let mut bounds = Vec::new();
657    for bound in &type_alias_data.bounds {
658        ctx.lower_type_bound(bound, self_ty.clone(), false).for_each(|pred| {
659            if let Some(pred) = generic_predicate_to_inline_bound(db, &pred, &self_ty) {
660                bounds.push(pred);
661            }
662        });
663    }
664
665    if !ctx.unsized_types.contains(&self_ty) {
666        let sized_trait =
667            LangItem::Sized.resolve_trait(db, resolver.krate()).map(to_chalk_trait_id);
668        let sized_bound = sized_trait.into_iter().map(|sized_trait| {
669            let trait_bound =
670                rust_ir::TraitBound { trait_id: sized_trait, args_no_self: Default::default() };
671            let inline_bound = rust_ir::InlineBound::TraitBound(trait_bound);
672            chalk_ir::Binders::empty(Interner, inline_bound)
673        });
674        bounds.extend(sized_bound);
675        bounds.shrink_to_fit();
676    }
677
678    // FIXME: Re-enable where clauses on associated types when an upstream chalk bug is fixed.
679    //        (rust-analyzer#9052)
680    // let where_clauses = convert_where_clauses(db, type_alias.into(), &bound_vars);
681    let bound_data = rust_ir::AssociatedTyDatumBound { bounds, where_clauses: vec![] };
682    let datum = AssociatedTyDatum {
683        trait_id: to_chalk_trait_id(trait_),
684        id: to_assoc_type_id(type_alias),
685        name: type_alias,
686        binders: make_binders(db, &generic_params, bound_data),
687    };
688    Arc::new(datum)
689}

692    db: &dyn HirDatabase,
693    krate: Crate,
694    trait_id: TraitId,
695) -> Arc<TraitDatum> {
696    debug!("trait_datum {:?}", trait_id);
697    let trait_ = from_chalk_trait_id(trait_id);
698    let trait_data = db.trait_signature(trait_);
699    debug!("trait {:?} = {:?}", trait_id, trait_data.name);

704        upstream: trait_.lookup(db).container.krate() != krate,

707        // FIXME: set these flags correctly
708        marker: false,
709        fundamental: trait_data.flags.contains(TraitFlags::FUNDAMENTAL),
710    };

716    let trait_datum = TraitDatum {
717        id: trait_id,
718        binders: make_binders(db, &generic_params, trait_datum_bound),
719        flags,
720        associated_ty_ids,
721        well_known,
722    };
723    Arc::new(trait_datum)

777pub(crate) fn adt_datum_query(
778    db: &dyn HirDatabase,
779    krate: Crate,
780    chalk_ir::AdtId(adt_id): AdtId,
781) -> Arc<AdtDatum> {
782    debug!("adt_datum {:?}", adt_id);
783    let generic_params = generics(db, adt_id.into());
784    let bound_vars_subst = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
785    let where_clauses = convert_where_clauses(db, adt_id.into(), &bound_vars_subst);
786
787    let (fundamental, phantom_data) = match adt_id {
788        hir_def::AdtId::StructId(s) => {
789            let flags = db.struct_signature(s).flags;
790            (flags.contains(StructFlags::FUNDAMENTAL), flags.contains(StructFlags::IS_PHANTOM_DATA))
791        }
792        // FIXME set fundamental flags correctly
793        hir_def::AdtId::UnionId(_) => (false, false),
794        hir_def::AdtId::EnumId(_) => (false, false),
795    };
796    let flags = rust_ir::AdtFlags {
797        upstream: adt_id.module(db).krate() != krate,
798        fundamental,
799        phantom_data,
800    };
801
802    // this slows down rust-analyzer by quite a bit unfortunately, so enabling this is currently not worth it
803    let _variant_id_to_fields = |id: VariantId| {
804        let variant_data = &id.fields(db);
805        let fields = if variant_data.fields().is_empty() {
806            vec![]
807        } else {
808            let field_types = db.field_types(id);
809            variant_data
810                .fields()
811                .iter()
812                .map(|(idx, _)| field_types[idx].clone().substitute(Interner, &bound_vars_subst))
813                .filter(|it| !it.contains_unknown())
814                .collect()
815        };
816        rust_ir::AdtVariantDatum { fields }
817    };
818    let variant_id_to_fields = |_: VariantId| rust_ir::AdtVariantDatum { fields: vec![] };
819
820    let (kind, variants) = match adt_id {
821        hir_def::AdtId::StructId(id) => {
822            (rust_ir::AdtKind::Struct, vec![variant_id_to_fields(id.into())])
823        }
824        hir_def::AdtId::EnumId(id) => {
825            let variants = id
826                .enum_variants(db)
827                .variants
828                .iter()
829                .map(|&(variant_id, _, _)| variant_id_to_fields(variant_id.into()))
830                .collect();
831            (rust_ir::AdtKind::Enum, variants)
832        }
833        hir_def::AdtId::UnionId(id) => {
834            (rust_ir::AdtKind::Union, vec![variant_id_to_fields(id.into())])
835        }
836    };
837
838    let struct_datum_bound = rust_ir::AdtDatumBound { variants, where_clauses };
839    let struct_datum = AdtDatum {
840        kind,
841        id: chalk_ir::AdtId(adt_id),
842        binders: make_binders(db, &generic_params, struct_datum_bound),
843        flags,
844    };
845    Arc::new(struct_datum)
846}

859fn impl_def_datum(db: &dyn HirDatabase, krate: Crate, impl_id: hir_def::ImplId) -> Arc<ImplDatum> {
860    let trait_ref = db
861        .impl_trait(impl_id)
862        // ImplIds for impls where the trait ref can't be resolved should never reach Chalk
863        .expect("invalid impl passed to Chalk")
864        .into_value_and_skipped_binders()
865        .0;
866    let impl_data = db.impl_signature(impl_id);
867
868    let generic_params = generics(db, impl_id.into());
869    let bound_vars = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
870    let trait_ = trait_ref.hir_trait_id();
871    let impl_type = if impl_id.lookup(db).container.krate() == krate {
872        rust_ir::ImplType::Local
873    } else {
874        rust_ir::ImplType::External
875    };
876    let where_clauses = convert_where_clauses(db, impl_id.into(), &bound_vars);
877    let negative = impl_data.flags.contains(ImplFlags::NEGATIVE);
878    let polarity = if negative { rust_ir::Polarity::Negative } else { rust_ir::Polarity::Positive };
879
880    let impl_datum_bound = rust_ir::ImplDatumBound { trait_ref, where_clauses };
881    let trait_data = trait_.trait_items(db);
882    let associated_ty_value_ids = impl_id
883        .impl_items(db)
884        .items
885        .iter()
886        .filter_map(|(_, item)| match item {
887            AssocItemId::TypeAliasId(type_alias) => Some(*type_alias),
888            _ => None,
889        })
890        .filter(|&type_alias| {
891            // don't include associated types that don't exist in the trait
892            let name = &db.type_alias_signature(type_alias).name;
893            trait_data.associated_type_by_name(name).is_some()
894        })
895        .map(|type_alias| TypeAliasAsValue(type_alias).to_chalk(db))
896        .collect();
897    debug!("impl_datum: {:?}", impl_datum_bound);
898    let impl_datum = ImplDatum {
899        binders: make_binders(db, &generic_params, impl_datum_bound),
900        impl_type,
901        polarity,
902        associated_ty_value_ids,
903    };
904    Arc::new(impl_datum)
905}

1025pub(super) fn generic_predicate_to_inline_bound(
1026    db: &dyn HirDatabase,
1027    pred: &QuantifiedWhereClause,
1028    self_ty: &Ty,
1029) -> Option<chalk_ir::Binders<rust_ir::InlineBound<Interner>>> {
1030    // An InlineBound is like a GenericPredicate, except the self type is left out.
1031    // We don't have a special type for this, but Chalk does.
1032    let self_ty_shifted_in = self_ty.clone().shifted_in_from(Interner, DebruijnIndex::ONE);
1033    let (pred, binders) = pred.as_ref().into_value_and_skipped_binders();
1034    match pred {
1035        WhereClause::Implemented(trait_ref) => {
1036            if trait_ref.self_type_parameter(Interner) != self_ty_shifted_in {
1037                // we can only convert predicates back to type bounds if they
1038                // have the expected self type
1039                return None;
1040            }
1041            let args_no_self = trait_ref.substitution.as_slice(Interner)[1..]
1042                .iter()
1043                .cloned()
1044                .casted(Interner)
1045                .collect();
1046            let trait_bound = rust_ir::TraitBound { trait_id: trait_ref.trait_id, args_no_self };
1047            Some(chalk_ir::Binders::new(binders, rust_ir::InlineBound::TraitBound(trait_bound)))
1048        }
1049        WhereClause::AliasEq(AliasEq { alias: AliasTy::Projection(projection_ty), ty }) => {
1050            let generics = generics(db, from_assoc_type_id(projection_ty.associated_ty_id).into());
1051            let parent_len = generics.parent_generics().map_or(0, |g| g.len_self());
1052            let (trait_args, assoc_args) =
1053                projection_ty.substitution.as_slice(Interner).split_at(parent_len);
1054            let (self_ty, args_no_self) =
1055                trait_args.split_first().expect("projection without trait self type");
1056            if self_ty.assert_ty_ref(Interner) != &self_ty_shifted_in {
1057                return None;
1058            }
1059
1060            let args_no_self = args_no_self.iter().cloned().casted(Interner).collect();
1061            let parameters = assoc_args.to_vec();
1062
1063            let alias_eq_bound = rust_ir::AliasEqBound {
1064                value: ty.clone(),
1065                trait_bound: rust_ir::TraitBound {
1066                    trait_id: to_chalk_trait_id(projection_ty.trait_(db)),
1067                    args_no_self,
1068                },
1069                associated_ty_id: projection_ty.associated_ty_id,
1070                parameters,
1071            };
1072            Some(chalk_ir::Binders::new(
1073                binders,
1074                rust_ir::InlineBound::AliasEqBound(alias_eq_bound),
1075            ))
1076        }
1077        _ => None,
1078    }
1079}
