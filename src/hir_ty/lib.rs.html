195/// A constant can have reference to other things. Memory map job is holding
196/// the necessary bits of memory of the const eval session to keep the constant
197/// meaningful.
198#[derive(Debug, Default, Clone, PartialEq, Eq)]
199pub enum MemoryMap {
200    #[default]
201    Empty,
202    Simple(Box<[u8]>),
203    Complex(Box<ComplexMemoryMap>),
204}
205
206#[derive(Debug, Default, Clone, PartialEq, Eq)]
207pub struct ComplexMemoryMap {
208    memory: IndexMap<usize, Box<[u8]>, FxBuildHasher>,
209    vtable: VTableMap,
210}
211
212impl ComplexMemoryMap {
213    fn insert(&mut self, addr: usize, val: Box<[u8]>) {
214        match self.memory.entry(addr) {
215            Entry::Occupied(mut e) => {
216                if e.get().len() < val.len() {
217                    e.insert(val);
218                }
219            }
220            Entry::Vacant(e) => {
221                e.insert(val);
222            }
223        }
224    }
225}
226
227impl MemoryMap {
228    pub fn vtable_ty(&self, id: usize) -> Result<&Ty, MirEvalError> {
229        match self {
230            MemoryMap::Empty | MemoryMap::Simple(_) => Err(MirEvalError::InvalidVTableId(id)),
231            MemoryMap::Complex(cm) => cm.vtable.ty(id),
232        }
233    }
234
235    fn simple(v: Box<[u8]>) -> Self {
236        MemoryMap::Simple(v)
237    }
238
239    /// This functions convert each address by a function `f` which gets the byte intervals and assign an address
240    /// to them. It is useful when you want to load a constant with a memory map in a new memory. You can pass an
241    /// allocator function as `f` and it will return a mapping of old addresses to new addresses.
242    fn transform_addresses(
243        &self,
244        mut f: impl FnMut(&[u8], usize) -> Result<usize, MirEvalError>,
245    ) -> Result<FxHashMap<usize, usize>, MirEvalError> {
246        let mut transform = |(addr, val): (&usize, &[u8])| {
247            let addr = *addr;
248            let align = if addr == 0 { 64 } else { (addr - (addr & (addr - 1))).min(64) };
249            f(val, align).map(|it| (addr, it))
250        };
251        match self {
252            MemoryMap::Empty => Ok(Default::default()),
253            MemoryMap::Simple(m) => transform((&0, m)).map(|(addr, val)| {
254                let mut map = FxHashMap::with_capacity_and_hasher(1, rustc_hash::FxBuildHasher);
255                map.insert(addr, val);
256                map
257            }),
258            MemoryMap::Complex(cm) => {
259                cm.memory.iter().map(|(addr, val)| transform((addr, val))).collect()
260            }
261        }
262    }
263
264    fn get(&self, addr: usize, size: usize) -> Option<&[u8]> {
265        if size == 0 {
266            Some(&[])
267        } else {
268            match self {
269                MemoryMap::Empty => Some(&[]),
270                MemoryMap::Simple(m) if addr == 0 => m.get(0..size),
271                MemoryMap::Simple(_) => None,
272                MemoryMap::Complex(cm) => cm.memory.get(&addr)?.get(0..size),
273            }
274        }
275    }
276}

314pub(crate) fn make_type_and_const_binders<T: HasInterner<Interner = Interner>>(
315    which_is_const: impl Iterator<Item = Option<Ty>>,
316    value: T,
317) -> Binders<T> {
318    Binders::new(
319        VariableKinds::from_iter(
320            Interner,
321            which_is_const.map(|x| {
322                if let Some(ty) = x {
323                    chalk_ir::VariableKind::Const(ty)
324                } else {
325                    chalk_ir::VariableKind::Ty(chalk_ir::TyVariableKind::General)
326                }
327            }),

329        value,

331}

353pub(crate) fn variable_kinds_from_iter(
354    db: &dyn HirDatabase,
355    iter: impl Iterator<Item = hir_def::GenericParamId>,
356) -> VariableKinds {
357    VariableKinds::from_iter(
358        Interner,
359        iter.map(|x| match x {
360            hir_def::GenericParamId::ConstParamId(id) => {
361                chalk_ir::VariableKind::Const(db.const_param_ty(id))
362            }
363            hir_def::GenericParamId::TypeParamId(_) => {
364                chalk_ir::VariableKind::Ty(chalk_ir::TyVariableKind::General)
365            }
366            hir_def::GenericParamId::LifetimeParamId(_) => chalk_ir::VariableKind::Lifetime,
367        }),
368    )

530impl CallableSig {
531    pub fn from_params_and_return(
532        params: impl Iterator<Item = Ty>,
533        ret: Ty,
534        is_varargs: bool,
535        safety: Safety,
536        abi: FnAbi,
537    ) -> CallableSig {
538        let mut params_and_return = Vec::with_capacity(params.size_hint().0 + 1);
539        params_and_return.extend(params);
540        params_and_return.push(ret);
541        CallableSig { params_and_return: params_and_return.into(), is_varargs, safety, abi }
542    }
543
544    pub fn from_def(db: &dyn HirDatabase, def: FnDefId, substs: &Substitution) -> CallableSig {
545        let callable_def = ToChalk::from_chalk(db, def);
546        let sig = db.callable_item_signature(callable_def);
547        sig.substitute(Interner, substs)
548    }
549    pub fn from_fn_ptr(fn_ptr: &FnPointer) -> CallableSig {
550        CallableSig {
551            // FIXME: what to do about lifetime params? -> return PolyFnSig
552            params_and_return: Arc::from_iter(
553                fn_ptr
554                    .substitution
555                    .clone()
556                    .shifted_out_to(Interner, DebruijnIndex::ONE)
557                    .expect("unexpected lifetime vars in fn ptr")
558                    .0
559                    .as_slice(Interner)
560                    .iter()
561                    .map(|arg| arg.assert_ty_ref(Interner).clone()),
562            ),
563            is_varargs: fn_ptr.sig.variadic,
564            safety: fn_ptr.sig.safety,
565            abi: fn_ptr.sig.abi,
566        }
567    }
568
569    pub fn to_fn_ptr(&self) -> FnPointer {
570        FnPointer {
571            num_binders: 0,
572            sig: FnSig { abi: self.abi, safety: self.safety, variadic: self.is_varargs },
573            substitution: FnSubst(Substitution::from_iter(
574                Interner,
575                self.params_and_return.iter().cloned(),
576            )),
577        }
578    }

580    pub fn abi(&self) -> FnAbi {
581        self.abi
582    }

584    pub fn params(&self) -> &[Ty] {

588    pub fn ret(&self) -> &Ty {

590    }

593impl TypeFoldable<Interner> for CallableSig {
594    fn try_fold_with<E>(
595        self,
596        folder: &mut dyn chalk_ir::fold::FallibleTypeFolder<Interner, Error = E>,
597        outer_binder: DebruijnIndex,
598    ) -> Result<Self, E> {
599        let vec = self.params_and_return.to_vec();
600        let folded = vec.try_fold_with(folder, outer_binder)?;
601        Ok(CallableSig {
602            params_and_return: folded.into(),
603            is_varargs: self.is_varargs,
604            safety: self.safety,
605            abi: self.abi,
606        })
607    }
608}

639pub(crate) fn fold_free_vars<T: HasInterner<Interner = Interner> + TypeFoldable<Interner>>(
640    t: T,
641    for_ty: impl FnMut(BoundVar, DebruijnIndex) -> Ty,
642    for_const: impl FnMut(Ty, BoundVar, DebruijnIndex) -> Const,
643) -> T {
644    use chalk_ir::fold::TypeFolder;
645
646    #[derive(chalk_derive::FallibleTypeFolder)]
647    #[has_interner(Interner)]
648    struct FreeVarFolder<
649        F1: FnMut(BoundVar, DebruijnIndex) -> Ty,
650        F2: FnMut(Ty, BoundVar, DebruijnIndex) -> Const,
651    >(F1, F2);
652    impl<F1: FnMut(BoundVar, DebruijnIndex) -> Ty, F2: FnMut(Ty, BoundVar, DebruijnIndex) -> Const>
653        TypeFolder<Interner> for FreeVarFolder<F1, F2>
654    {
655        fn as_dyn(&mut self) -> &mut dyn TypeFolder<Interner> {
656            self
657        }
658
659        fn interner(&self) -> Interner {
660            Interner
661        }
662
663        fn fold_free_var_ty(&mut self, bound_var: BoundVar, outer_binder: DebruijnIndex) -> Ty {
664            self.0(bound_var, outer_binder)
665        }
666
667        fn fold_free_var_const(
668            &mut self,
669            ty: Ty,
670            bound_var: BoundVar,
671            outer_binder: DebruijnIndex,
672        ) -> Const {
673            self.1(ty, bound_var, outer_binder)
674        }
675    }
676    t.fold_with(&mut FreeVarFolder(for_ty, for_const), DebruijnIndex::INNERMOST)
677}

694pub(crate) fn fold_tys_and_consts<T: HasInterner<Interner = Interner> + TypeFoldable<Interner>>(
695    t: T,
696    f: impl FnMut(Either<Ty, Const>, DebruijnIndex) -> Either<Ty, Const>,
697    binders: DebruijnIndex,
698) -> T {
699    use chalk_ir::fold::{TypeFolder, TypeSuperFoldable};
700    #[derive(chalk_derive::FallibleTypeFolder)]
701    #[has_interner(Interner)]
702    struct TyFolder<F: FnMut(Either<Ty, Const>, DebruijnIndex) -> Either<Ty, Const>>(F);
703    impl<F: FnMut(Either<Ty, Const>, DebruijnIndex) -> Either<Ty, Const>> TypeFolder<Interner>
704        for TyFolder<F>
705    {
706        fn as_dyn(&mut self) -> &mut dyn TypeFolder<Interner> {
707            self
708        }
709
710        fn interner(&self) -> Interner {
711            Interner
712        }
713
714        fn fold_ty(&mut self, ty: Ty, outer_binder: DebruijnIndex) -> Ty {
715            let ty = ty.super_fold_with(self.as_dyn(), outer_binder);
716            self.0(Either::Left(ty), outer_binder).left().unwrap()
717        }
718
719        fn fold_const(&mut self, c: Const, outer_binder: DebruijnIndex) -> Const {
720            self.0(Either::Right(c), outer_binder).right().unwrap()
721        }
722    }
723    t.fold_with(&mut TyFolder(f), binders)
724}

726pub(crate) fn fold_generic_args<T: HasInterner<Interner = Interner> + TypeFoldable<Interner>>(
727    t: T,
728    f: impl FnMut(GenericArgData, DebruijnIndex) -> GenericArgData,
729    binders: DebruijnIndex,

732    #[derive(chalk_derive::FallibleTypeFolder)]
733    #[has_interner(Interner)]
734    struct TyFolder<F: FnMut(GenericArgData, DebruijnIndex) -> GenericArgData>(F);
735    impl<F: FnMut(GenericArgData, DebruijnIndex) -> GenericArgData> TypeFolder<Interner>
736        for TyFolder<F>
737    {
738        fn as_dyn(&mut self) -> &mut dyn TypeFolder<Interner> {
739            self
740        }
741
742        fn interner(&self) -> Interner {
743            Interner
744        }
745
746        fn fold_ty(&mut self, ty: Ty, outer_binder: DebruijnIndex) -> Ty {
747            let ty = ty.super_fold_with(self.as_dyn(), outer_binder);
748            self.0(GenericArgData::Ty(ty), outer_binder)
749                .intern(Interner)

751                .unwrap()
752                .clone()
753        }
754
755        fn fold_const(&mut self, c: Const, outer_binder: DebruijnIndex) -> Const {

758                .constant(Interner)

765            self.0(GenericArgData::Lifetime(lt), outer_binder)
766                .intern(Interner)
767                .lifetime(Interner)
768                .unwrap()
769                .clone()
770        }
771    }
772    t.fold_with(&mut TyFolder(f), binders)
773}

775/// 'Canonicalizes' the `t` by replacing any errors with new variables. Also
776/// ensures there are no unbound variables or inference variables anywhere in
777/// the `t`.
778pub fn replace_errors_with_variables<T>(t: &T) -> Canonical<T>
779where
780    T: HasInterner<Interner = Interner> + TypeFoldable<Interner> + Clone,
781{
782    use chalk_ir::{
783        Fallible,
784        fold::{FallibleTypeFolder, TypeSuperFoldable},
785    };
786    struct ErrorReplacer {
787        vars: usize,
788    }
789    impl FallibleTypeFolder<Interner> for ErrorReplacer {
790        type Error = NoSolution;
791
792        fn as_dyn(&mut self) -> &mut dyn FallibleTypeFolder<Interner, Error = Self::Error> {
793            self
794        }
795
796        fn interner(&self) -> Interner {
797            Interner
798        }
799
800        fn try_fold_ty(&mut self, ty: Ty, outer_binder: DebruijnIndex) -> Fallible<Ty> {
801            if let TyKind::Error = ty.kind(Interner) {
802                let index = self.vars;
803                self.vars += 1;
804                Ok(TyKind::BoundVar(BoundVar::new(outer_binder, index)).intern(Interner))
805            } else {
806                ty.try_super_fold_with(self.as_dyn(), outer_binder)
807            }
808        }
809
810        fn try_fold_inference_ty(
811            &mut self,
812            _var: InferenceVar,
813            _kind: TyVariableKind,
814            _outer_binder: DebruijnIndex,
815        ) -> Fallible<Ty> {
816            if cfg!(debug_assertions) {
817                // we don't want to just panic here, because then the error message
818                // won't contain the whole thing, which would not be very helpful
819                Err(NoSolution)
820            } else {
821                Ok(TyKind::Error.intern(Interner))
822            }
823        }
824
825        fn try_fold_free_var_ty(
826            &mut self,
827            _bound_var: BoundVar,
828            _outer_binder: DebruijnIndex,
829        ) -> Fallible<Ty> {
830            if cfg!(debug_assertions) {
831                // we don't want to just panic here, because then the error message
832                // won't contain the whole thing, which would not be very helpful
833                Err(NoSolution)
834            } else {
835                Ok(TyKind::Error.intern(Interner))
836            }
837        }
838
839        fn try_fold_inference_const(
840            &mut self,
841            ty: Ty,
842            _var: InferenceVar,
843            _outer_binder: DebruijnIndex,
844        ) -> Fallible<Const> {
845            if cfg!(debug_assertions) { Err(NoSolution) } else { Ok(unknown_const(ty)) }
846        }
847
848        fn try_fold_free_var_const(
849            &mut self,
850            ty: Ty,
851            _bound_var: BoundVar,
852            _outer_binder: DebruijnIndex,
853        ) -> Fallible<Const> {
854            if cfg!(debug_assertions) { Err(NoSolution) } else { Ok(unknown_const(ty)) }
855        }
856
857        fn try_fold_inference_lifetime(
858            &mut self,
859            _var: InferenceVar,
860            _outer_binder: DebruijnIndex,
861        ) -> Fallible<Lifetime> {
862            if cfg!(debug_assertions) { Err(NoSolution) } else { Ok(error_lifetime()) }
863        }
864
865        fn try_fold_free_var_lifetime(
866            &mut self,
867            _bound_var: BoundVar,
868            _outer_binder: DebruijnIndex,
869        ) -> Fallible<Lifetime> {
870            if cfg!(debug_assertions) { Err(NoSolution) } else { Ok(error_lifetime()) }
871        }
872    }
873    let mut error_replacer = ErrorReplacer { vars: 0 };
874    let value = match t.clone().try_fold_with(&mut error_replacer, DebruijnIndex::INNERMOST) {
875        Ok(t) => t,
876        Err(_) => panic!("Encountered unbound or inference vars in {t:?}"),
877    };
878    let kinds = (0..error_replacer.vars).map(|_| {
879        chalk_ir::CanonicalVarKind::new(
880            chalk_ir::VariableKind::Ty(TyVariableKind::General),
881            chalk_ir::UniverseIndex::ROOT,
882        )
883    });
884    Canonical { value, binders: chalk_ir::CanonicalVarKinds::from_iter(Interner, kinds) }
885}

887pub fn callable_sig_from_fn_trait(
888    self_ty: &Ty,
889    trait_env: Arc<TraitEnvironment>,
890    db: &dyn HirDatabase,
891) -> Option<(FnTrait, CallableSig)> {
892    let krate = trait_env.krate;
893    let fn_once_trait = FnTrait::FnOnce.get_id(db, krate)?;
894    let output_assoc_type = fn_once_trait
895        .trait_items(db)
896        .associated_type_by_name(&Name::new_symbol_root(sym::Output))?;
897
898    let mut table = InferenceTable::new(db, trait_env.clone());
899    let b = TyBuilder::trait_ref(db, fn_once_trait);
900    if b.remaining() != 2 {
901        return None;
902    }
903
904    // Register two obligations:
905    // - Self: FnOnce<?args_ty>
906    // - <Self as FnOnce<?args_ty>>::Output == ?ret_ty
907    let args_ty = table.new_type_var();
908    let mut trait_ref = b.push(self_ty.clone()).push(args_ty.clone()).build();
909    let projection = TyBuilder::assoc_type_projection(
910        db,
911        output_assoc_type,
912        Some(trait_ref.substitution.clone()),
913    )
914    .build();
915
916    let block = trait_env.block;
917    let trait_env = trait_env.env.clone();
918    let obligation =
919        InEnvironment { goal: trait_ref.clone().cast(Interner), environment: trait_env.clone() };
920    let canonical = table.canonicalize(obligation.clone());
921    if db.trait_solve(krate, block, canonical.cast(Interner)).is_some() {
922        table.register_obligation(obligation.goal);
923        let return_ty = table.normalize_projection_ty(projection);
924        for fn_x in [FnTrait::Fn, FnTrait::FnMut, FnTrait::FnOnce] {
925            let fn_x_trait = fn_x.get_id(db, krate)?;
926            trait_ref.trait_id = to_chalk_trait_id(fn_x_trait);
927            let obligation: chalk_ir::InEnvironment<chalk_ir::Goal<Interner>> = InEnvironment {
928                goal: trait_ref.clone().cast(Interner),
929                environment: trait_env.clone(),
930            };
931            let canonical = table.canonicalize(obligation.clone());
932            if db.trait_solve(krate, block, canonical.cast(Interner)).is_some() {
933                let ret_ty = table.resolve_completely(return_ty);
934                let args_ty = table.resolve_completely(args_ty);
935                let params = args_ty
936                    .as_tuple()?
937                    .iter(Interner)
938                    .map(|it| it.assert_ty_ref(Interner))
939                    .cloned();
940
941                return Some((
942                    fn_x,
943                    CallableSig::from_params_and_return(
944                        params,
945                        ret_ty,
946                        false,
947                        Safety::Safe,
948                        FnAbi::RustCall,
949                    ),
950                ));
951            }
952        }
953        unreachable!("It should at least implement FnOnce at this point");
954    } else {
955        None
956    }
957}

959struct PlaceholderCollector<'db> {
960    db: &'db dyn HirDatabase,
961    placeholders: FxHashSet<TypeOrConstParamId>,
962}
963
964impl PlaceholderCollector<'_> {
965    fn collect(&mut self, idx: PlaceholderIndex) {
966        let id = from_placeholder_idx(self.db, idx);
967        self.placeholders.insert(id);
968    }
969}
970
971impl TypeVisitor<Interner> for PlaceholderCollector<'_> {
972    type BreakTy = ();
973
974    fn as_dyn(&mut self) -> &mut dyn TypeVisitor<Interner, BreakTy = Self::BreakTy> {
975        self
976    }
977
978    fn interner(&self) -> Interner {
979        Interner
980    }
981
982    fn visit_ty(
983        &mut self,
984        ty: &Ty,
985        outer_binder: DebruijnIndex,
986    ) -> std::ops::ControlFlow<Self::BreakTy> {
987        let has_placeholder_bits = TypeFlags::HAS_TY_PLACEHOLDER | TypeFlags::HAS_CT_PLACEHOLDER;
988        let TyData { kind, flags } = ty.data(Interner);
989
990        if let TyKind::Placeholder(idx) = kind {
991            self.collect(*idx);
992        } else if flags.intersects(has_placeholder_bits) {
993            return ty.super_visit_with(self, outer_binder);
994        } else {
995            // Fast path: don't visit inner types (e.g. generic arguments) when `flags` indicate
996            // that there are no placeholders.
997        }
998
999        std::ops::ControlFlow::Continue(())
1000    }
1001
1002    fn visit_const(
1003        &mut self,
1004        constant: &chalk_ir::Const<Interner>,
1005        _outer_binder: DebruijnIndex,
1006    ) -> std::ops::ControlFlow<Self::BreakTy> {
1007        if let chalk_ir::ConstValue::Placeholder(idx) = constant.data(Interner).value {
1008            self.collect(idx);
1009        }
1010        std::ops::ControlFlow::Continue(())
1011    }
1012}

1014/// Returns unique placeholders for types and consts contained in `value`.
1015pub fn collect_placeholders<T>(value: &T, db: &dyn HirDatabase) -> Vec<TypeOrConstParamId>
1016where
1017    T: ?Sized + TypeVisitable<Interner>,
1018{
1019    let mut collector = PlaceholderCollector { db, placeholders: FxHashSet::default() };
1020    _ = value.visit_with(&mut collector, DebruijnIndex::INNERMOST);
1021    collector.placeholders.into_iter().collect()
1022}
