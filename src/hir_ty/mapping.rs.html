1//! This module contains the implementations of the `ToChalk` trait, which
2//! handles conversion between our data types and their corresponding types in
3//! Chalk (in both directions); plus some helper functions for more specialized
4//! conversions.

20pub trait ToChalk {
21    type Chalk;
22    fn to_chalk(self, db: &dyn HirDatabase) -> Self::Chalk;
23    fn from_chalk(db: &dyn HirDatabase, chalk: Self::Chalk) -> Self;
24}
25
26pub(crate) fn from_chalk<T, ChalkT>(db: &dyn HirDatabase, chalk: ChalkT) -> T
27where
28    T: ToChalk<Chalk = ChalkT>,
29{
30    T::from_chalk(db, chalk)
31}
32
33impl ToChalk for hir_def::ImplId {
34    type Chalk = chalk_db::ImplId;
35
36    fn to_chalk(self, _db: &dyn HirDatabase) -> chalk_db::ImplId {
37        chalk_ir::ImplId(self.as_id())
38    }
39
40    fn from_chalk(_db: &dyn HirDatabase, impl_id: chalk_db::ImplId) -> hir_def::ImplId {
41        FromId::from_id(impl_id.0.as_id())
42    }
43}
44
45impl ToChalk for CallableDefId {
46    type Chalk = FnDefId;
47
48    fn to_chalk(self, _db: &dyn HirDatabase) -> FnDefId {
49        chalk_ir::FnDefId(salsa::plumbing::AsId::as_id(&self))
50    }
51
52    fn from_chalk(db: &dyn HirDatabase, fn_def_id: FnDefId) -> CallableDefId {
53        salsa::plumbing::FromIdWithDb::from_id(fn_def_id.0, db.zalsa())
54    }
55}

126pub fn from_placeholder_idx(db: &dyn HirDatabase, idx: PlaceholderIndex) -> TypeOrConstParamId {
127    assert_eq!(idx.ui, chalk_ir::UniverseIndex::ROOT);
128    // SAFETY: We cannot really encapsulate this unfortunately, so just hope this is sound.
129    let interned_id =
130        InternedTypeOrConstParamId::from_id(unsafe { Id::from_index(idx.idx.try_into().unwrap()) });
131    interned_id.loc(db)
132}
133
134pub fn to_placeholder_idx(db: &dyn HirDatabase, id: TypeOrConstParamId) -> PlaceholderIndex {
135    let interned_id = InternedTypeOrConstParamId::new(db, id);
136    PlaceholderIndex {
137        ui: chalk_ir::UniverseIndex::ROOT,
138        idx: interned_id.as_id().index() as usize,
139    }
140}
