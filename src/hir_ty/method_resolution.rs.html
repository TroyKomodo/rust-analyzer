38/// This is used as a key for indexing impls.
39#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
40pub enum TyFingerprint {
41    // These are lang item impls:
42    Str,
43    Slice,
44    Array,
45    Never,
46    RawPtr(Mutability),
47    Scalar(Scalar),
48    // These can have user-defined impls:
49    Adt(hir_def::AdtId),
50    Dyn(TraitId),
51    ForeignType(ForeignDefId),
52    // These only exist for trait impls
53    Unit,
54    Unnameable,
55    Function(u32),
56}

58impl TyFingerprint {
59    /// Creates a TyFingerprint for looking up an inherent impl. Only certain
60    /// types can have inherent impls: if we have some `struct S`, we can have
61    /// an `impl S`, but not `impl &S`. Hence, this will return `None` for
62    /// reference types and such.
63    pub fn for_inherent_impl(ty: &Ty) -> Option<TyFingerprint> {
64        let fp = match ty.kind(Interner) {
65            TyKind::Str => TyFingerprint::Str,
66            TyKind::Never => TyFingerprint::Never,
67            TyKind::Slice(..) => TyFingerprint::Slice,
68            TyKind::Array(..) => TyFingerprint::Array,
69            TyKind::Scalar(scalar) => TyFingerprint::Scalar(*scalar),
70            TyKind::Adt(AdtId(adt), _) => TyFingerprint::Adt(*adt),
71            TyKind::Raw(mutability, ..) => TyFingerprint::RawPtr(*mutability),
72            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(*alias_id),
73            TyKind::Dyn(_) => ty.dyn_trait().map(TyFingerprint::Dyn)?,
74            _ => return None,
75        };
76        Some(fp)
77    }
78
79    /// Creates a TyFingerprint for looking up a trait impl.
80    pub fn for_trait_impl(ty: &Ty) -> Option<TyFingerprint> {
81        let fp = match ty.kind(Interner) {
82            TyKind::Str => TyFingerprint::Str,
83            TyKind::Never => TyFingerprint::Never,
84            TyKind::Slice(..) => TyFingerprint::Slice,
85            TyKind::Array(..) => TyFingerprint::Array,
86            TyKind::Scalar(scalar) => TyFingerprint::Scalar(*scalar),
87            TyKind::Adt(AdtId(adt), _) => TyFingerprint::Adt(*adt),
88            TyKind::Raw(mutability, ..) => TyFingerprint::RawPtr(*mutability),
89            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(*alias_id),
90            TyKind::Dyn(_) => ty.dyn_trait().map(TyFingerprint::Dyn)?,
91            TyKind::Ref(_, _, ty) => return TyFingerprint::for_trait_impl(ty),
92            TyKind::Tuple(_, subst) => {
93                let first_ty = subst.interned().first().map(|arg| arg.assert_ty_ref(Interner));
94                match first_ty {
95                    Some(ty) => return TyFingerprint::for_trait_impl(ty),
96                    None => TyFingerprint::Unit,
97                }
98            }
99            TyKind::AssociatedType(_, _)
100            | TyKind::OpaqueType(_, _)
101            | TyKind::FnDef(_, _)
102            | TyKind::Closure(_, _)
103            | TyKind::Coroutine(..)
104            | TyKind::CoroutineWitness(..) => TyFingerprint::Unnameable,
105            TyKind::Function(fn_ptr) => {
106                TyFingerprint::Function(fn_ptr.substitution.0.len(Interner) as u32)
107            }
108            TyKind::Alias(_)
109            | TyKind::Placeholder(_)
110            | TyKind::BoundVar(_)
111            | TyKind::InferenceVar(_, _)
112            | TyKind::Error => return None,
113        };
114        Some(fp)
115    }
116}

150impl TraitImpls {
151    pub(crate) fn trait_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
152        let _p = tracing::info_span!("trait_impls_in_crate_query", ?krate).entered();
153        let mut impls = FxHashMap::default();
154
155        Self::collect_def_map(db, &mut impls, crate_def_map(db, krate));
156
157        Arc::new(Self::finish(impls))
158    }
159
160    pub(crate) fn trait_impls_in_block_query(
161        db: &dyn HirDatabase,
162        block: BlockId,
163    ) -> Option<Arc<Self>> {
164        let _p = tracing::info_span!("trait_impls_in_block_query").entered();
165        let mut impls = FxHashMap::default();
166
167        Self::collect_def_map(db, &mut impls, block_def_map(db, block));
168
169        if impls.is_empty() { None } else { Some(Arc::new(Self::finish(impls))) }
170    }
171
172    pub(crate) fn trait_impls_in_deps_query(
173        db: &dyn HirDatabase,
174        krate: Crate,
175    ) -> Arc<[Arc<Self>]> {
176        let _p = tracing::info_span!("trait_impls_in_deps_query", ?krate).entered();
177        Arc::from_iter(
178            db.transitive_deps(krate).into_iter().map(|krate| db.trait_impls_in_crate(krate)),
179        )
180    }
181
182    fn finish(map: TraitFpMapCollector) -> TraitImpls {
183        TraitImpls {
184            map: map
185                .into_iter()
186                .map(|(k, v)| (k, v.into_iter().map(|(k, v)| (k, v.into_boxed_slice())).collect()))
187                .collect(),
188        }
189    }
190
191    fn collect_def_map(db: &dyn HirDatabase, map: &mut TraitFpMapCollector, def_map: &DefMap) {
192        for (_module_id, module_data) in def_map.modules() {
193            for impl_id in module_data.scope.impls() {
194                // Reservation impls should be ignored during trait resolution, so we never need
195                // them during type analysis. See rust-lang/rust#64631 for details.
196                //
197                // FIXME: Reservation impls should be considered during coherence checks. If we are
198                // (ever) to implement coherence checks, this filtering should be done by the trait
199                // solver.
200                if db.attrs(impl_id.into()).by_key(sym::rustc_reservation_impl).exists() {
201                    continue;
202                }
203                let target_trait = match db.impl_trait(impl_id) {
204                    Some(tr) => tr.skip_binders().hir_trait_id(),
205                    None => continue,
206                };
207                let self_ty = db.impl_self_ty(impl_id);
208                let self_ty_fp = TyFingerprint::for_trait_impl(self_ty.skip_binders());
209                map.entry(target_trait).or_default().entry(self_ty_fp).or_default().push(impl_id);
210            }
211
212            // To better support custom derives, collect impls in all unnamed const items.
213            // const _: () = { ... };
214            for konst in module_data.scope.unnamed_consts() {
215                let body = db.body(konst.into());
216                for (_, block_def_map) in body.blocks(db) {
217                    Self::collect_def_map(db, map, block_def_map);
218                }
219            }
220        }
221    }

223    /// Queries all trait impls for the given type.
224    pub fn for_self_ty_without_blanket_impls(
225        &self,
226        fp: TyFingerprint,
227    ) -> impl Iterator<Item = ImplId> + '_ {
228        self.map
229            .values()
230            .flat_map(move |impls| impls.get(&Some(fp)).into_iter())
231            .flat_map(|it| it.iter().copied())

234    /// Queries all impls of the given trait.
235    pub fn for_trait(&self, trait_: TraitId) -> impl Iterator<Item = ImplId> + '_ {
236        self.map
237            .get(&trait_)
238            .into_iter()
239            .flat_map(|map| map.values().flat_map(|v| v.iter().copied()))
240    }
241
242    /// Queries all impls of `trait_` that may apply to `self_ty`.
243    pub fn for_trait_and_self_ty(
244        &self,
245        trait_: TraitId,
246        self_ty: TyFingerprint,
247    ) -> impl Iterator<Item = ImplId> + '_ {
248        self.map
249            .get(&trait_)
250            .into_iter()
251            .flat_map(move |map| map.get(&Some(self_ty)).into_iter().chain(map.get(&None)))
252            .flat_map(|v| v.iter().copied())
253    }
254
255    /// Queries whether `self_ty` has potentially applicable implementations of `trait_`.
256    pub fn has_impls_for_trait_and_self_ty(&self, trait_: TraitId, self_ty: TyFingerprint) -> bool {
257        self.for_trait_and_self_ty(trait_, self_ty).next().is_some()
258    }

260    pub fn all_impls(&self) -> impl Iterator<Item = ImplId> + '_ {
261        self.map.values().flat_map(|map| map.values().flat_map(|v| v.iter().copied()))
262    }

265/// Inherent impls defined in some crate.
266///
267/// Inherent impls can only be defined in the crate that also defines the self type of the impl
268/// (note that some primitives are considered to be defined by both libcore and liballoc).
269///
270/// This makes inherent impl lookup easier than trait impl lookup since we only have to consider a
271/// single crate.
272#[derive(Debug, Eq, PartialEq)]
273pub struct InherentImpls {
274    map: FxHashMap<TyFingerprint, Vec<ImplId>>,
275    invalid_impls: Vec<ImplId>,
276}
277
278impl InherentImpls {
279    pub(crate) fn inherent_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
280        let _p = tracing::info_span!("inherent_impls_in_crate_query", ?krate).entered();
281        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
282
283        let crate_def_map = crate_def_map(db, krate);
284        impls.collect_def_map(db, crate_def_map);
285        impls.shrink_to_fit();
286
287        Arc::new(impls)
288    }
289
290    pub(crate) fn inherent_impls_in_block_query(
291        db: &dyn HirDatabase,
292        block: BlockId,
293    ) -> Option<Arc<Self>> {
294        let _p = tracing::info_span!("inherent_impls_in_block_query").entered();
295        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
296
297        let block_def_map = block_def_map(db, block);
298        impls.collect_def_map(db, block_def_map);
299        impls.shrink_to_fit();
300
301        if impls.map.is_empty() && impls.invalid_impls.is_empty() {
302            None
303        } else {
304            Some(Arc::new(impls))
305        }
306    }
307
308    fn shrink_to_fit(&mut self) {
309        self.map.values_mut().for_each(Vec::shrink_to_fit);
310        self.map.shrink_to_fit();
311    }
312
313    fn collect_def_map(&mut self, db: &dyn HirDatabase, def_map: &DefMap) {
314        for (_module_id, module_data) in def_map.modules() {
315            for impl_id in module_data.scope.impls() {
316                let data = db.impl_signature(impl_id);
317                if data.target_trait.is_some() {
318                    continue;
319                }
320
321                let self_ty = db.impl_self_ty(impl_id);
322                let self_ty = self_ty.skip_binders();
323
324                match is_inherent_impl_coherent(db, def_map, impl_id, self_ty) {
325                    true => {
326                        // `fp` should only be `None` in error cases (either erroneous code or incomplete name resolution)
327                        if let Some(fp) = TyFingerprint::for_inherent_impl(self_ty) {
328                            self.map.entry(fp).or_default().push(impl_id);
329                        }
330                    }
331                    false => self.invalid_impls.push(impl_id),
332                }
333            }
334
335            // To better support custom derives, collect impls in all unnamed const items.
336            // const _: () = { ... };
337            for konst in module_data.scope.unnamed_consts() {
338                let body = db.body(konst.into());
339                for (_, block_def_map) in body.blocks(db) {
340                    self.collect_def_map(db, block_def_map);
341                }
342            }
343        }
344    }
345
346    pub fn for_self_ty(&self, self_ty: &Ty) -> &[ImplId] {
347        match TyFingerprint::for_inherent_impl(self_ty) {
348            Some(fp) => self.map.get(&fp).map(|vec| vec.as_ref()).unwrap_or(&[]),
349            None => &[],
350        }
351    }
352
353    pub fn all_impls(&self) -> impl Iterator<Item = ImplId> + '_ {
354        self.map.values().flat_map(|v| v.iter().copied())
355    }
356
357    pub fn invalid_impls(&self) -> &[ImplId] {
358        &self.invalid_impls
359    }
360}

362pub(crate) fn incoherent_inherent_impl_crates(
363    db: &dyn HirDatabase,
364    krate: Crate,
365    fp: TyFingerprint,
366) -> SmallVec<[Crate; 2]> {
367    let _p = tracing::info_span!("incoherent_inherent_impl_crates").entered();
368    let mut res = SmallVec::new();
369
370    // should pass crate for finger print and do reverse deps
371
372    for krate in db.transitive_deps(krate) {
373        let impls = db.inherent_impls_in_crate(krate);
374        if impls.map.get(&fp).is_some_and(|v| !v.is_empty()) {
375            res.push(krate);
376        }
377    }
378
379    res
380}

382pub fn def_crates(db: &dyn HirDatabase, ty: &Ty, cur_crate: Crate) -> Option<SmallVec<[Crate; 2]>> {
383    match ty.kind(Interner) {
384        &TyKind::Adt(AdtId(def_id), _) => {
385            let rustc_has_incoherent_inherent_impls = match def_id {
386                hir_def::AdtId::StructId(id) => db
387                    .struct_signature(id)
388                    .flags
389                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
390                hir_def::AdtId::UnionId(id) => db
391                    .union_signature(id)
392                    .flags
393                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
394                hir_def::AdtId::EnumId(id) => db
395                    .enum_signature(id)
396                    .flags
397                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
398            };
399            Some(if rustc_has_incoherent_inherent_impls {
400                db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Adt(def_id))
401            } else {
402                smallvec![def_id.module(db).krate()]
403            })
404        }
405        &TyKind::Foreign(id) => {
406            let alias = from_foreign_def_id(id);
407            Some(
408                if db
409                    .type_alias_signature(alias)
410                    .flags
411                    .contains(TypeAliasFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPL)
412                {
413                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::ForeignType(id))
414                } else {
415                    smallvec![alias.module(db).krate()]
416                },
417            )
418        }
419        TyKind::Dyn(_) => {
420            let trait_id = ty.dyn_trait()?;
421            Some(
422                if db
423                    .trait_signature(trait_id)
424                    .flags
425                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
426                {
427                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Dyn(trait_id))
428                } else {
429                    smallvec![trait_id.module(db).krate()]
430                },
431            )
432        }
433        // for primitives, there may be impls in various places (core and alloc
434        // mostly). We just check the whole crate graph for crates with impls
435        // (cached behind a query).
436        TyKind::Scalar(_)
437        | TyKind::Str
438        | TyKind::Slice(_)
439        | TyKind::Array(..)
440        | TyKind::Raw(..) => Some(db.incoherent_inherent_impl_crates(
441            cur_crate,
442            TyFingerprint::for_inherent_impl(ty).expect("fingerprint for primitive"),
443        )),
444        _ => None,
445    }
446}

448/// Look up the method with the given name.
449pub(crate) fn lookup_method(
450    db: &dyn HirDatabase,
451    ty: &Canonical<Ty>,
452    env: Arc<TraitEnvironment>,
453    traits_in_scope: &FxHashSet<TraitId>,
454    visible_from_module: VisibleFromModule,
455    name: &Name,
456) -> Option<(ReceiverAdjustments, FunctionId, bool)> {
457    let mut not_visible = None;
458    let res = iterate_method_candidates(
459        ty,
460        db,
461        env,
462        traits_in_scope,
463        visible_from_module,
464        Some(name),
465        LookupMode::MethodCall,
466        |adjustments, f, visible| match f {
467            AssocItemId::FunctionId(f) if visible => Some((adjustments, f, true)),
468            AssocItemId::FunctionId(f) if not_visible.is_none() => {
469                not_visible = Some((adjustments, f, false));
470                None
471            }
472            _ => None,
473        },
474    );
475    res.or(not_visible)
476}

478/// Whether we're looking up a dotted method call (like `v.len()`) or a path
479/// (like `Vec::new`).
480#[derive(Copy, Clone, Debug, PartialEq, Eq)]
481pub enum LookupMode {
482    /// Looking up a method call like `v.len()`: We only consider candidates
483    /// that have a `self` parameter, and do autoderef.
484    MethodCall,
485    /// Looking up a path like `Vec::new` or `Vec::default`: We consider all
486    /// candidates including associated constants, but don't do autoderef.
487    Path,
488}

524#[derive(Debug, Clone, Default)]
525pub struct ReceiverAdjustments {
526    autoref: Option<AutorefOrPtrAdjustment>,
527    autoderefs: usize,
528    unsize_array: bool,
529}
530
531impl ReceiverAdjustments {
532    pub(crate) fn apply(&self, table: &mut InferenceTable<'_>, ty: Ty) -> (Ty, Vec<Adjustment>) {
533        let mut ty = table.resolve_ty_shallow(&ty);
534        let mut adjust = Vec::new();
535        for _ in 0..self.autoderefs {
536            match autoderef::autoderef_step(table, ty.clone(), true, false) {
537                None => {
538                    never!("autoderef not possible for {:?}", ty);
539                    ty = TyKind::Error.intern(Interner);
540                    break;
541                }
542                Some((kind, new_ty)) => {
543                    ty = new_ty.clone();
544                    let mutbl = match self.autoref {
545                        Some(AutorefOrPtrAdjustment::Autoref(m)) => Some(m),
546                        Some(AutorefOrPtrAdjustment::ToConstPtr) => Some(Mutability::Not),
547                        // FIXME should we know the mutability here, when autoref is `None`?
548                        None => None,
549                    };
550                    adjust.push(Adjustment {
551                        kind: Adjust::Deref(match kind {
552                            AutoderefKind::Overloaded => Some(OverloadedDeref(mutbl)),
553                            AutoderefKind::Builtin => None,
554                        }),
555                        target: new_ty,
556                    });
557                }
558            }
559        }
560        if let Some(autoref) = &self.autoref {
561            let lt = table.new_lifetime_var();
562            match autoref {
563                AutorefOrPtrAdjustment::Autoref(m) => {
564                    let a = Adjustment::borrow(*m, ty, lt);
565                    ty = a.target.clone();
566                    adjust.push(a);
567                }
568                AutorefOrPtrAdjustment::ToConstPtr => {
569                    if let TyKind::Raw(Mutability::Mut, pointee) = ty.kind(Interner) {
570                        let a = Adjustment {
571                            kind: Adjust::Pointer(PointerCast::MutToConstPointer),
572                            target: TyKind::Raw(Mutability::Not, pointee.clone()).intern(Interner),
573                        };
574                        ty = a.target.clone();
575                        adjust.push(a);
576                    } else {
577                        never!("`ToConstPtr` target is not a raw mutable pointer");
578                    }
579                }
580            };
581        }
582        if self.unsize_array {
583            ty = 'it: {
584                if let TyKind::Ref(m, l, inner) = ty.kind(Interner)
585                    && let TyKind::Array(inner, _) = inner.kind(Interner)
586                {
587                    break 'it TyKind::Ref(
588                        *m,
589                        l.clone(),
590                        TyKind::Slice(inner.clone()).intern(Interner),
591                    )
592                    .intern(Interner);
593                }
594                // FIXME: report diagnostic if array unsizing happens without indirection.
595                ty
596            };
597            adjust.push(Adjustment {
598                kind: Adjust::Pointer(PointerCast::Unsize),
599                target: ty.clone(),
600            });
601        }
602        (ty, adjust)
603    }
604
605    fn with_autoref(&self, a: AutorefOrPtrAdjustment) -> ReceiverAdjustments {
606        Self { autoref: Some(a), ..*self }
607    }

670/// Checks if the self parameter of `Trait` method is the `dyn Trait` and we should
671/// call the method using the vtable.
672pub fn is_dyn_method(
673    db: &dyn HirDatabase,
674    _env: Arc<TraitEnvironment>,
675    func: FunctionId,
676    fn_subst: Substitution,
677) -> Option<usize> {
678    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
679        return None;
680    };
681    let trait_params = db.generic_params(trait_id.into()).len();
682    let fn_params = fn_subst.len(Interner) - trait_params;
683    let trait_ref = TraitRef {
684        trait_id: to_chalk_trait_id(trait_id),
685        substitution: Substitution::from_iter(Interner, fn_subst.iter(Interner).take(trait_params)),
686    };
687    let self_ty = trait_ref.self_type_parameter(Interner);
688    if let TyKind::Dyn(d) = self_ty.kind(Interner) {
689        let is_my_trait_in_bounds = d
690            .bounds
691            .skip_binders()
692            .as_slice(Interner)
693            .iter()
694            .map(|it| it.skip_binders())
695            .flat_map(|it| match it {
696                WhereClause::Implemented(tr) => {
697                    all_super_traits(db, from_chalk_trait_id(tr.trait_id))
698                }
699                _ => smallvec![],
700            })
701            // rustc doesn't accept `impl Foo<2> for dyn Foo<5>`, so if the trait id is equal, no matter
702            // what the generics are, we are sure that the method is come from the vtable.
703            .any(|x| x == trait_id);
704        if is_my_trait_in_bounds {
705            return Some(fn_params);
706        }
707    }
708    None
709}

711/// Looks up the impl method that actually runs for the trait method `func`.
712///
713/// Returns `func` if it's not a method defined in a trait or the lookup failed.
714pub(crate) fn lookup_impl_method_query(
715    db: &dyn HirDatabase,
716    env: Arc<TraitEnvironment>,
717    func: FunctionId,
718    fn_subst: Substitution,
719) -> (FunctionId, Substitution) {
720    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
721        return (func, fn_subst);
722    };
723    let trait_params = db.generic_params(trait_id.into()).len();
724    let trait_ref = TraitRef {
725        trait_id: to_chalk_trait_id(trait_id),
726        substitution: Substitution::from_iter(Interner, fn_subst.iter(Interner).take(trait_params)),
727    };
728
729    let name = &db.function_signature(func).name;
730    let Some((impl_fn, impl_subst)) =
731        lookup_impl_assoc_item_for_trait_ref(trait_ref, db, env, name).and_then(|assoc| {
732            if let (AssocItemId::FunctionId(id), subst) = assoc { Some((id, subst)) } else { None }
733        })
734    else {
735        return (func, fn_subst);
736    };
737
738    (
739        impl_fn,
740        Substitution::from_iter(
741            Interner,
742            impl_subst.iter(Interner).chain(fn_subst.iter(Interner).skip(trait_params)),
743        ),
744    )
745}

747fn lookup_impl_assoc_item_for_trait_ref(
748    trait_ref: TraitRef,
749    db: &dyn HirDatabase,
750    env: Arc<TraitEnvironment>,
751    name: &Name,
752) -> Option<(AssocItemId, Substitution)> {
753    let hir_trait_id = trait_ref.hir_trait_id();
754    let self_ty = trait_ref.self_type_parameter(Interner);
755    let self_ty_fp = TyFingerprint::for_trait_impl(&self_ty)?;
756    let impls = db.trait_impls_in_deps(env.krate);
757
758    let trait_module = hir_trait_id.module(db);
759    let type_module = match self_ty_fp {
760        TyFingerprint::Adt(adt_id) => Some(adt_id.module(db)),
761        TyFingerprint::ForeignType(type_id) => Some(from_foreign_def_id(type_id).module(db)),
762        TyFingerprint::Dyn(trait_id) => Some(trait_id.module(db)),
763        _ => None,
764    };
765
766    let def_blocks: ArrayVec<_, 2> =
767        [trait_module.containing_block(), type_module.and_then(|it| it.containing_block())]
768            .into_iter()
769            .flatten()
770            .filter_map(|block_id| db.trait_impls_in_block(block_id))
771            .collect();
772
773    let impls = impls
774        .iter()
775        .chain(&def_blocks)
776        .flat_map(|impls| impls.for_trait_and_self_ty(hir_trait_id, self_ty_fp));
777
778    let table = InferenceTable::new(db, env);
779
780    let (impl_data, impl_subst) = find_matching_impl(impls, table, trait_ref)?;
781    let item = impl_data.items.iter().find_map(|(n, it)| match *it {
782        AssocItemId::FunctionId(f) => (n == name).then_some(AssocItemId::FunctionId(f)),
783        AssocItemId::ConstId(c) => (n == name).then_some(AssocItemId::ConstId(c)),
784        AssocItemId::TypeAliasId(_) => None,
785    })?;
786    Some((item, impl_subst))
787}

789fn find_matching_impl(
790    mut impls: impl Iterator<Item = ImplId>,
791    mut table: InferenceTable<'_>,
792    actual_trait_ref: TraitRef,
793) -> Option<(&ImplItems, Substitution)> {
794    let db = table.db;
795    impls.find_map(|impl_| {
796        table.run_in_snapshot(|table| {
797            let impl_substs =
798                TyBuilder::subst_for_def(db, impl_, None).fill_with_inference_vars(table).build();
799            let trait_ref = db
800                .impl_trait(impl_)
801                .expect("non-trait method in find_matching_impl")
802                .substitute(Interner, &impl_substs);
803
804            if !table.unify(&trait_ref, &actual_trait_ref) {
805                return None;
806            }
807
808            let wcs = crate::chalk_db::convert_where_clauses(db, impl_.into(), &impl_substs)
809                .into_iter()
810                .map(|b| b.cast(Interner));
811            let goal = crate::Goal::all(Interner, wcs);
812            table.try_obligation(goal.clone())?;
813            table.register_obligation(goal);
814            Some((impl_.impl_items(db), table.resolve_completely(impl_substs)))
815        })
816    })
817}

819fn is_inherent_impl_coherent(
820    db: &dyn HirDatabase,
821    def_map: &DefMap,
822    impl_id: ImplId,
823    self_ty: &Ty,
824) -> bool {
825    let self_ty = self_ty.kind(Interner);
826    let impl_allowed = match self_ty {
827        TyKind::Tuple(_, _)
828        | TyKind::FnDef(_, _)
829        | TyKind::Array(_, _)
830        | TyKind::Never
831        | TyKind::Raw(_, _)
832        | TyKind::Ref(_, _, _)
833        | TyKind::Slice(_)
834        | TyKind::Str
835        | TyKind::Scalar(_) => def_map.is_rustc_coherence_is_core(),
836
837        &TyKind::Adt(AdtId(adt), _) => adt.module(db).krate() == def_map.krate(),
838        TyKind::Dyn(it) => it.principal_id().is_some_and(|trait_id| {
839            from_chalk_trait_id(trait_id).module(db).krate() == def_map.krate()
840        }),
841
842        _ => true,
843    };
844    impl_allowed || {
845        let rustc_has_incoherent_inherent_impls = match self_ty {
846            TyKind::Tuple(_, _)
847            | TyKind::FnDef(_, _)
848            | TyKind::Array(_, _)
849            | TyKind::Never
850            | TyKind::Raw(_, _)
851            | TyKind::Ref(_, _, _)
852            | TyKind::Slice(_)
853            | TyKind::Str
854            | TyKind::Scalar(_) => true,
855
856            &TyKind::Adt(AdtId(adt), _) => match adt {
857                hir_def::AdtId::StructId(id) => db
858                    .struct_signature(id)
859                    .flags
860                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
861                hir_def::AdtId::UnionId(id) => db
862                    .union_signature(id)
863                    .flags
864                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
865                hir_def::AdtId::EnumId(it) => db
866                    .enum_signature(it)
867                    .flags
868                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
869            },
870            TyKind::Dyn(it) => it.principal_id().is_some_and(|trait_id| {
871                db.trait_signature(from_chalk_trait_id(trait_id))
872                    .flags
873                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
874            }),
875
876            _ => false,
877        };
878        let items = impl_id.impl_items(db);
879        rustc_has_incoherent_inherent_impls
880            && !items.items.is_empty()
881            && items.items.iter().all(|&(_, assoc)| match assoc {
882                AssocItemId::FunctionId(it) => {
883                    db.function_signature(it).flags.contains(FnFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
884                }
885                AssocItemId::ConstId(it) => {
886                    db.const_signature(it).flags.contains(ConstFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
887                }
888                AssocItemId::TypeAliasId(it) => db
889                    .type_alias_signature(it)
890                    .flags
891                    .contains(TypeAliasFlags::RUSTC_ALLOW_INCOHERENT_IMPL),
892            })
893    }
894}

896/// Checks whether the impl satisfies the orphan rules.
897///
898/// Given `impl<P1..=Pn> Trait<T1..=Tn> for T0`, an `impl`` is valid only if at least one of the following is true:
899/// - Trait is a local trait
900/// - All of
901///   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
902///   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
903pub fn check_orphan_rules(db: &dyn HirDatabase, impl_: ImplId) -> bool {
904    let substs = TyBuilder::placeholder_subst(db, impl_);
905    let Some(impl_trait) = db.impl_trait(impl_) else {
906        // not a trait impl
907        return true;
908    };
909
910    let local_crate = impl_.lookup(db).container.krate();
911    let is_local = |tgt_crate| tgt_crate == local_crate;
912
913    let trait_ref = impl_trait.substitute(Interner, &substs);
914    let trait_id = from_chalk_trait_id(trait_ref.trait_id);
915    if is_local(trait_id.module(db).krate()) {
916        // trait to be implemented is local
917        return true;
918    }
919
920    let unwrap_fundamental = |mut ty: Ty| {
921        // Unwrap all layers of fundamental types with a loop.
922        loop {
923            match ty.kind(Interner) {
924                TyKind::Ref(_, _, referenced) => ty = referenced.clone(),
925                &TyKind::Adt(AdtId(hir_def::AdtId::StructId(s)), ref subs) => {
926                    let struct_signature = db.struct_signature(s);
927                    if struct_signature.flags.contains(StructFlags::FUNDAMENTAL) {
928                        let next = subs.type_parameters(Interner).next();
929                        match next {
930                            Some(it) => ty = it,
931                            None => break ty,
932                        }
933                    } else {
934                        break ty;
935                    }
936                }
937                _ => break ty,
938            }
939        }
940    };
941    //   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
942
943    // FIXME: param coverage
944    //   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
945    let is_not_orphan = trait_ref.substitution.type_parameters(Interner).any(|ty| {
946        match unwrap_fundamental(ty).kind(Interner) {
947            &TyKind::Adt(AdtId(id), _) => is_local(id.module(db).krate()),
948            TyKind::Error => true,
949            TyKind::Dyn(it) => it
950                .principal_id()
951                .is_some_and(|trait_id| is_local(from_chalk_trait_id(trait_id).module(db).krate())),
952            _ => false,
953        }
954    });
955    #[allow(clippy::let_and_return)]
956    is_not_orphan
957}

981pub fn iterate_method_candidates_dyn(
982    ty: &Canonical<Ty>,
983    db: &dyn HirDatabase,
984    env: Arc<TraitEnvironment>,
985    traits_in_scope: &FxHashSet<TraitId>,
986    visible_from_module: VisibleFromModule,
987    name: Option<&Name>,
988    mode: LookupMode,
989    callback: &mut dyn MethodCandidateCallback,
990) -> ControlFlow<()> {
991    let _p = tracing::info_span!(
992        "iterate_method_candidates_dyn",
993        ?mode,
994        ?name,
995        traits_in_scope_len = traits_in_scope.len()
996    )
997    .entered();
998
999    match mode {
1000        LookupMode::MethodCall => {
1001            // For method calls, rust first does any number of autoderef, and
1002            // then one autoref (i.e. when the method takes &self or &mut self).
1003            // Note that when we've got a receiver like &S, even if the method
1004            // we find in the end takes &self, we still do the autoderef step
1005            // (just as rustc does an autoderef and then autoref again).
1006
1007            // We have to be careful about the order we're looking at candidates
1008            // in here. Consider the case where we're resolving `it.clone()`
1009            // where `it: &Vec<_>`. This resolves to the clone method with self
1010            // type `Vec<_>`, *not* `&_`. I.e. we need to consider methods where
1011            // the receiver type exactly matches before cases where we have to
1012            // do autoref. But in the autoderef steps, the `&_` self type comes
1013            // up *before* the `Vec<_>` self type.
1014            //
1015            // On the other hand, we don't want to just pick any by-value method
1016            // before any by-autoref method; it's just that we need to consider
1017            // the methods by autoderef order of *receiver types*, not *self
1018            // types*.
1019
1020            let mut table = InferenceTable::new(db, env);
1021            let ty = table.instantiate_canonical(ty.clone());
1022            let deref_chain = autoderef_method_receiver(&mut table, ty);
1023
1024            deref_chain.into_iter().try_for_each(|(receiver_ty, adj)| {
1025                iterate_method_candidates_with_autoref(
1026                    &mut table,
1027                    receiver_ty,
1028                    adj,
1029                    traits_in_scope,
1030                    visible_from_module,
1031                    name,
1032                    callback,
1033                )
1034            })
1035        }
1036        LookupMode::Path => {
1037            // No autoderef for path lookups
1038            iterate_method_candidates_for_self_ty(
1039                ty,
1040                db,
1041                env,
1042                traits_in_scope,
1043                visible_from_module,
1044                name,
1045                callback,
1046            )
1047        }
1048    }
1049}

1051#[tracing::instrument(skip_all, fields(name = ?name))]
1052fn iterate_method_candidates_with_autoref(
1053    table: &mut InferenceTable<'_>,
1054    receiver_ty: Canonical<Ty>,
1055    first_adjustment: ReceiverAdjustments,
1056    traits_in_scope: &FxHashSet<TraitId>,
1057    visible_from_module: VisibleFromModule,
1058    name: Option<&Name>,
1059    callback: &mut dyn MethodCandidateCallback,
1060) -> ControlFlow<()> {
1061    if receiver_ty.value.is_general_var(Interner, &receiver_ty.binders) {
1062        // don't try to resolve methods on unknown types
1063        return ControlFlow::Continue(());
1064    }
1065
1066    let mut iterate_method_candidates_by_receiver = move |receiver_ty, first_adjustment| {
1067        iterate_method_candidates_by_receiver(
1068            table,
1069            receiver_ty,
1070            first_adjustment,
1071            traits_in_scope,
1072            visible_from_module,
1073            name,
1074            callback,
1075        )
1076    };
1077
1078    let mut maybe_reborrowed = first_adjustment.clone();
1079    if let Some((_, _, m)) = receiver_ty.value.as_reference() {
1080        // Prefer reborrow of references to move
1081        maybe_reborrowed.autoref = Some(AutorefOrPtrAdjustment::Autoref(m));
1082        maybe_reborrowed.autoderefs += 1;
1083    }
1084
1085    iterate_method_candidates_by_receiver(receiver_ty.clone(), maybe_reborrowed)?;
1086
1087    let refed = Canonical {
1088        value: TyKind::Ref(Mutability::Not, error_lifetime(), receiver_ty.value.clone())
1089            .intern(Interner),
1090        binders: receiver_ty.binders.clone(),
1091    };
1092
1093    iterate_method_candidates_by_receiver(
1094        refed,
1095        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Not)),
1096    )?;
1097
1098    let ref_muted = Canonical {
1099        value: TyKind::Ref(Mutability::Mut, error_lifetime(), receiver_ty.value.clone())
1100            .intern(Interner),
1101        binders: receiver_ty.binders.clone(),
1102    };
1103
1104    iterate_method_candidates_by_receiver(
1105        ref_muted,
1106        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Mut)),
1107    )?;
1108
1109    if let Some((ty, Mutability::Mut)) = receiver_ty.value.as_raw_ptr() {
1110        let const_ptr_ty = Canonical {
1111            value: TyKind::Raw(Mutability::Not, ty.clone()).intern(Interner),
1112            binders: receiver_ty.binders,
1113        };
1114        iterate_method_candidates_by_receiver(
1115            const_ptr_ty,
1116            first_adjustment.with_autoref(AutorefOrPtrAdjustment::ToConstPtr),
1117        )?;
1118    }
1119
1120    ControlFlow::Continue(())
1121}

1162#[tracing::instrument(skip_all, fields(name = ?name))]
1163fn iterate_method_candidates_by_receiver(
1164    table: &mut InferenceTable<'_>,
1165    receiver_ty: Canonical<Ty>,
1166    receiver_adjustments: ReceiverAdjustments,
1167    traits_in_scope: &FxHashSet<TraitId>,
1168    visible_from_module: VisibleFromModule,
1169    name: Option<&Name>,
1170    callback: &mut dyn MethodCandidateCallback,
1171) -> ControlFlow<()> {
1172    let receiver_ty = table.instantiate_canonical(receiver_ty);
1173    // We're looking for methods with *receiver* type receiver_ty. These could
1174    // be found in any of the derefs of receiver_ty, so we have to go through
1175    // that, including raw derefs.
1176    table.run_in_snapshot(|table| {
1177        let mut autoderef =
1178            autoderef::Autoderef::new_no_tracking(table, receiver_ty.clone(), true, true);
1179        while let Some((self_ty, _)) = autoderef.next() {
1180            iterate_inherent_methods(
1181                &self_ty,
1182                autoderef.table,
1183                name,
1184                Some(&receiver_ty),
1185                Some(receiver_adjustments.clone()),
1186                visible_from_module,
1187                &mut |adjustments, item, is_visible| {
1188                    callback.on_inherent_method(adjustments, item, is_visible)
1189                },
1190            )?
1191        }
1192        ControlFlow::Continue(())
1193    })?;
1194    table.run_in_snapshot(|table| {
1195        let mut autoderef =
1196            autoderef::Autoderef::new_no_tracking(table, receiver_ty.clone(), true, true);
1197        while let Some((self_ty, _)) = autoderef.next() {
1198            if matches!(self_ty.kind(Interner), TyKind::InferenceVar(_, TyVariableKind::General)) {
1199                // don't try to resolve methods on unknown types
1200                return ControlFlow::Continue(());
1201            }
1202
1203            iterate_trait_method_candidates(
1204                &self_ty,
1205                autoderef.table,
1206                traits_in_scope,
1207                name,
1208                Some(&receiver_ty),
1209                Some(receiver_adjustments.clone()),
1210                &mut |adjustments, item, is_visible| {
1211                    callback.on_trait_method(adjustments, item, is_visible)
1212                },
1213            )?
1214        }
1215        ControlFlow::Continue(())
1216    })
1217}

1255#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1256fn iterate_trait_method_candidates(
1257    self_ty: &Ty,
1258    table: &mut InferenceTable<'_>,
1259    traits_in_scope: &FxHashSet<TraitId>,
1260    name: Option<&Name>,
1261    receiver_ty: Option<&Ty>,
1262    receiver_adjustments: Option<ReceiverAdjustments>,
1263    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1264) -> ControlFlow<()> {
1265    let db = table.db;
1266
1267    let canonical_self_ty = table.canonicalize(self_ty.clone());
1268    let TraitEnvironment { krate, block, .. } = *table.trait_env;
1269
1270    'traits: for &t in traits_in_scope {
1271        let data = db.trait_signature(t);
1272
1273        // Traits annotated with `#[rustc_skip_during_method_dispatch]` are skipped during
1274        // method resolution, if the receiver is an array, and we're compiling for editions before
1275        // 2021.
1276        // This is to make `[a].into_iter()` not break code with the new `IntoIterator` impl for
1277        // arrays.
1278        if data.flags.contains(TraitFlags::SKIP_ARRAY_DURING_METHOD_DISPATCH)
1279            && matches!(self_ty.kind(Interner), TyKind::Array(..))
1280        {
1281            // FIXME: this should really be using the edition of the method name's span, in case it
1282            // comes from a macro
1283            if !krate.data(db).edition.at_least_2021() {
1284                continue;
1285            }
1286        }
1287        if data.flags.contains(TraitFlags::SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH)
1288            && matches!(
1289                self_ty.kind(Interner), TyKind::Adt(AdtId(def), subst)
1290                if is_box(table.db, *def)
1291                    && matches!(subst.at(Interner, 0).assert_ty_ref(Interner).kind(Interner), TyKind::Slice(..))
1292            )
1293        {
1294            // FIXME: this should really be using the edition of the method name's span, in case it
1295            // comes from a macro
1296            if !krate.data(db).edition.at_least_2024() {
1297                continue;
1298            }
1299        }
1300
1301        // we'll be lazy about checking whether the type implements the
1302        // trait, but if we find out it doesn't, we'll skip the rest of the
1303        // iteration
1304        let mut known_implemented = false;
1305        for &(_, item) in t.trait_items(db).items.iter() {
1306            // Don't pass a `visible_from_module` down to `is_valid_candidate`,
1307            // since only inherent methods should be included into visibility checking.
1308            let visible =
1309                match is_valid_trait_method_candidate(table, t, name, receiver_ty, item, self_ty) {
1310                    IsValidCandidate::Yes => true,
1311                    IsValidCandidate::NotVisible => false,
1312                    IsValidCandidate::No => continue,
1313                };
1314            if !known_implemented {
1315                let goal = generic_implements_goal(db, &table.trait_env, t, &canonical_self_ty);
1316                if db.trait_solve(krate, block, goal.cast(Interner)).is_none() {
1317                    continue 'traits;
1318                }
1319            }
1320            known_implemented = true;
1321            callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1322        }
1323    }
1324    ControlFlow::Continue(())
1325}

1327#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1328fn iterate_inherent_methods(
1329    self_ty: &Ty,
1330    table: &mut InferenceTable<'_>,
1331    name: Option<&Name>,
1332    receiver_ty: Option<&Ty>,
1333    receiver_adjustments: Option<ReceiverAdjustments>,
1334    visible_from_module: VisibleFromModule,
1335    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1336) -> ControlFlow<()> {
1337    let db = table.db;
1338    let env = table.trait_env.clone();
1339
1340    // For trait object types and placeholder types with trait bounds, the methods of the trait and
1341    // its super traits are considered inherent methods. This matters because these methods have
1342    // higher priority than the other traits' methods, which would be considered in
1343    // `iterate_trait_method_candidates()` only after this function.
1344    match self_ty.kind(Interner) {
1345        TyKind::Placeholder(_) => {
1346            let env = table.trait_env.clone();
1347            let traits = env
1348                .traits_in_scope_from_clauses(self_ty.clone())
1349                .flat_map(|t| all_super_traits(db, t));
1350            iterate_inherent_trait_methods(
1351                self_ty,
1352                table,
1353                name,
1354                receiver_ty,
1355                receiver_adjustments.clone(),
1356                callback,
1357                traits,
1358            )?;
1359        }
1360        TyKind::Dyn(_) => {
1361            if let Some(principal_trait) = self_ty.dyn_trait() {
1362                let traits = all_super_traits(db, principal_trait);
1363                iterate_inherent_trait_methods(
1364                    self_ty,
1365                    table,
1366                    name,
1367                    receiver_ty,
1368                    receiver_adjustments.clone(),
1369                    callback,
1370                    traits.into_iter(),
1371                )?;
1372            }
1373        }
1374        _ => {}
1375    }
1376
1377    let def_crates = match def_crates(db, self_ty, env.krate) {
1378        Some(k) => k,
1379        None => return ControlFlow::Continue(()),
1380    };
1381
1382    let (module, mut block) = match visible_from_module {
1383        VisibleFromModule::Filter(module) => (Some(module), module.containing_block()),
1384        VisibleFromModule::IncludeBlock(block) => (None, Some(block)),
1385        VisibleFromModule::None => (None, None),
1386    };
1387
1388    while let Some(block_id) = block {
1389        if let Some(impls) = db.inherent_impls_in_block(block_id) {
1390            impls_for_self_ty(
1391                &impls,
1392                self_ty,
1393                table,
1394                name,
1395                receiver_ty,
1396                receiver_adjustments.clone(),
1397                module,
1398                callback,
1399            )?;
1400        }
1401
1402        block = block_def_map(db, block_id).parent().and_then(|module| module.containing_block());
1403    }
1404
1405    for krate in def_crates {
1406        let impls = db.inherent_impls_in_crate(krate);
1407        impls_for_self_ty(
1408            &impls,
1409            self_ty,
1410            table,
1411            name,
1412            receiver_ty,
1413            receiver_adjustments.clone(),
1414            module,
1415            callback,
1416        )?;
1417    }
1418    return ControlFlow::Continue(());
1419
1420    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1421    fn iterate_inherent_trait_methods(
1422        self_ty: &Ty,
1423        table: &mut InferenceTable<'_>,
1424        name: Option<&Name>,
1425        receiver_ty: Option<&Ty>,
1426        receiver_adjustments: Option<ReceiverAdjustments>,
1427        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1428        traits: impl Iterator<Item = TraitId>,
1429    ) -> ControlFlow<()> {
1430        let db = table.db;
1431        for t in traits {
1432            let data = t.trait_items(db);
1433            for &(_, item) in data.items.iter() {
1434                // We don't pass `visible_from_module` as all trait items should be visible.
1435                let visible = match is_valid_trait_method_candidate(
1436                    table,
1437                    t,
1438                    name,
1439                    receiver_ty,
1440                    item,
1441                    self_ty,
1442                ) {
1443                    IsValidCandidate::Yes => true,
1444                    IsValidCandidate::NotVisible => false,
1445                    IsValidCandidate::No => continue,
1446                };
1447                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1448            }
1449        }
1450        ControlFlow::Continue(())
1451    }
1452
1453    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1454    fn impls_for_self_ty(
1455        impls: &InherentImpls,
1456        self_ty: &Ty,
1457        table: &mut InferenceTable<'_>,
1458        name: Option<&Name>,
1459        receiver_ty: Option<&Ty>,
1460        receiver_adjustments: Option<ReceiverAdjustments>,
1461        visible_from_module: Option<ModuleId>,
1462        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1463    ) -> ControlFlow<()> {
1464        for &impl_id in impls.for_self_ty(self_ty) {
1465            for &(ref item_name, item) in impl_id.impl_items(table.db).items.iter() {
1466                let visible = match is_valid_impl_method_candidate(
1467                    table,
1468                    self_ty,
1469                    receiver_ty,
1470                    visible_from_module,
1471                    name,
1472                    impl_id,
1473                    item,
1474                    item_name,
1475                ) {
1476                    IsValidCandidate::Yes => true,
1477                    IsValidCandidate::NotVisible => false,
1478                    IsValidCandidate::No => continue,
1479                };
1480                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1481            }
1482        }
1483        ControlFlow::Continue(())
1484    }
1485}

1525#[tracing::instrument(skip_all, fields(name))]
1526fn is_valid_impl_method_candidate(
1527    table: &mut InferenceTable<'_>,
1528    self_ty: &Ty,
1529    receiver_ty: Option<&Ty>,
1530    visible_from_module: Option<ModuleId>,
1531    name: Option<&Name>,
1532    impl_id: ImplId,
1533    item: AssocItemId,
1534    item_name: &Name,
1535) -> IsValidCandidate {
1536    match item {
1537        AssocItemId::FunctionId(f) => is_valid_impl_fn_candidate(
1538            table,
1539            impl_id,
1540            f,
1541            name,
1542            receiver_ty,
1543            self_ty,
1544            visible_from_module,
1545            item_name,
1546        ),
1547        AssocItemId::ConstId(c) => {
1548            let db = table.db;
1549            check_that!(receiver_ty.is_none());
1550            check_that!(name.is_none_or(|n| n == item_name));
1551
1552            if let Some(from_module) = visible_from_module
1553                && !db.assoc_visibility(c.into()).is_visible_from(db, from_module)
1554            {
1555                cov_mark::hit!(const_candidate_not_visible);
1556                return IsValidCandidate::NotVisible;
1557            }
1558            let self_ty_matches = table.run_in_snapshot(|table| {
1559                let expected_self_ty =
1560                    TyBuilder::impl_self_ty(db, impl_id).fill_with_inference_vars(table).build();
1561                table.unify(&expected_self_ty, self_ty)
1562            });
1563            if !self_ty_matches {
1564                cov_mark::hit!(const_candidate_self_type_mismatch);
1565                return IsValidCandidate::No;
1566            }
1567            IsValidCandidate::Yes
1568        }
1569        _ => IsValidCandidate::No,
1570    }
1571}

1573/// Checks whether a given `AssocItemId` is applicable for `receiver_ty`.
1574#[tracing::instrument(skip_all, fields(name))]
1575fn is_valid_trait_method_candidate(
1576    table: &mut InferenceTable<'_>,
1577    trait_id: TraitId,
1578    name: Option<&Name>,
1579    receiver_ty: Option<&Ty>,
1580    item: AssocItemId,
1581    self_ty: &Ty,
1582) -> IsValidCandidate {
1583    let db = table.db;
1584    match item {
1585        AssocItemId::FunctionId(fn_id) => {
1586            let data = db.function_signature(fn_id);
1587
1588            check_that!(name.is_none_or(|n| n == &data.name));
1589
1590            table.run_in_snapshot(|table| {
1591                let impl_subst = TyBuilder::subst_for_def(db, trait_id, None)
1592                    .fill_with_inference_vars(table)
1593                    .build();
1594                let expect_self_ty = impl_subst.at(Interner, 0).assert_ty_ref(Interner).clone();
1595
1596                check_that!(table.unify(&expect_self_ty, self_ty));
1597
1598                if let Some(receiver_ty) = receiver_ty {
1599                    check_that!(data.has_self_param());
1600
1601                    let fn_subst = TyBuilder::subst_for_def(db, fn_id, Some(impl_subst))
1602                        .fill_with_inference_vars(table)
1603                        .build();
1604
1605                    let sig = db.callable_item_signature(fn_id.into());
1606                    let expected_receiver =
1607                        sig.map(|s| s.params()[0].clone()).substitute(Interner, &fn_subst);
1608
1609                    check_that!(table.unify(receiver_ty, &expected_receiver));
1610                }
1611
1612                IsValidCandidate::Yes
1613            })
1614        }
1615        AssocItemId::ConstId(c) => {
1616            check_that!(receiver_ty.is_none());
1617            check_that!(name.is_none_or(|n| db.const_signature(c).name.as_ref() == Some(n)));
1618
1619            IsValidCandidate::Yes
1620        }
1621        _ => IsValidCandidate::No,
1622    }
1623}

1625#[tracing::instrument(skip_all, fields(name))]
1626fn is_valid_impl_fn_candidate(
1627    table: &mut InferenceTable<'_>,
1628    impl_id: ImplId,
1629    fn_id: FunctionId,
1630    name: Option<&Name>,
1631    receiver_ty: Option<&Ty>,
1632    self_ty: &Ty,
1633    visible_from_module: Option<ModuleId>,
1634    item_name: &Name,
1635) -> IsValidCandidate {
1636    check_that!(name.is_none_or(|n| n == item_name));
1637
1638    let db = table.db;
1639    let data = db.function_signature(fn_id);
1640
1641    if let Some(from_module) = visible_from_module
1642        && !db.assoc_visibility(fn_id.into()).is_visible_from(db, from_module)
1643    {
1644        cov_mark::hit!(autoderef_candidate_not_visible);
1645        return IsValidCandidate::NotVisible;
1646    }
1647    table.run_in_snapshot(|table| {
1648        let _p = tracing::info_span!("subst_for_def").entered();
1649        let impl_subst =
1650            TyBuilder::subst_for_def(db, impl_id, None).fill_with_inference_vars(table).build();
1651        let expect_self_ty = db.impl_self_ty(impl_id).substitute(Interner, &impl_subst);
1652
1653        check_that!(table.unify(&expect_self_ty, self_ty));
1654
1655        if let Some(receiver_ty) = receiver_ty {
1656            let _p = tracing::info_span!("check_receiver_ty").entered();
1657            check_that!(data.has_self_param());
1658
1659            let fn_subst = TyBuilder::subst_for_def(db, fn_id, Some(impl_subst.clone()))
1660                .fill_with_inference_vars(table)
1661                .build();
1662
1663            let sig = db.callable_item_signature(fn_id.into());
1664            let expected_receiver =
1665                sig.map(|s| s.params()[0].clone()).substitute(Interner, &fn_subst);
1666
1667            check_that!(table.unify(receiver_ty, &expected_receiver));
1668        }
1669
1670        // We need to consider the bounds on the impl to distinguish functions of the same name
1671        // for a type.
1672        let predicates = db.generic_predicates(impl_id.into());
1673        let goals = predicates.iter().map(|p| {
1674            let (p, b) = p
1675                .clone()
1676                .substitute(Interner, &impl_subst)
1677                // Skipping the inner binders is ok, as we don't handle quantified where
1678                // clauses yet.
1679                .into_value_and_skipped_binders();
1680            stdx::always!(b.len(Interner) == 0);
1681
1682            p.cast::<Goal>(Interner)
1683        });
1684
1685        for goal in goals.clone() {
1686            let in_env = InEnvironment::new(&table.trait_env.env, goal);
1687            let canonicalized = table.canonicalize_with_free_vars(in_env);
1688            let solution = table.db.trait_solve(
1689                table.trait_env.krate,
1690                table.trait_env.block,
1691                canonicalized.value.clone(),
1692            );
1693
1694            match solution {
1695                Some(Solution::Unique(canonical_subst)) => {
1696                    canonicalized.apply_solution(
1697                        table,
1698                        Canonical {
1699                            binders: canonical_subst.binders,
1700                            value: canonical_subst.value.subst,
1701                        },
1702                    );
1703                }
1704                Some(Solution::Ambig(Guidance::Definite(substs))) => {
1705                    canonicalized.apply_solution(table, substs);
1706                }
1707                Some(_) => (),
1708                None => return IsValidCandidate::No,
1709            }
1710        }
1711
1712        for goal in goals {
1713            if table.try_obligation(goal).is_none() {
1714                return IsValidCandidate::No;
1715            }
1716        }
1717
1718        IsValidCandidate::Yes
1719    })
1720}

1746/// This creates Substs for a trait with the given Self type and type variables
1747/// for all other parameters, to query Chalk with it.
1748#[tracing::instrument(skip_all)]
1749fn generic_implements_goal(
1750    db: &dyn HirDatabase,
1751    env: &TraitEnvironment,
1752    trait_: TraitId,
1753    self_ty: &Canonical<Ty>,
1754) -> Canonical<InEnvironment<super::DomainGoal>> {
1755    let binders = self_ty.binders.interned();
1756    let trait_ref = TyBuilder::trait_ref(db, trait_)
1757        .push(self_ty.value.clone())
1758        .fill_with_bound_vars(DebruijnIndex::INNERMOST, binders.len())
1759        .build();
1760
1761    let kinds =
1762        binders.iter().cloned().chain(trait_ref.substitution.iter(Interner).skip(1).map(|it| {
1763            let vk = match it.data(Interner) {
1764                GenericArgData::Ty(_) => VariableKind::Ty(chalk_ir::TyVariableKind::General),
1765                GenericArgData::Lifetime(_) => VariableKind::Lifetime,
1766                GenericArgData::Const(c) => VariableKind::Const(c.data(Interner).ty.clone()),
1767            };
1768            WithKind::new(vk, UniverseIndex::ROOT)
1769        }));
1770    let binders = CanonicalVarKinds::from_iter(Interner, kinds);
1771
1772    let obligation = trait_ref.cast(Interner);
1773    let value = InEnvironment::new(&env.env, obligation);
1774    Canonical { binders, value }
1775}

1795            Canonical { value: unsized_ty, binders },

1797        ));
