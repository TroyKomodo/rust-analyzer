120/// A token to ensuring that each drop scope is popped at most once, thanks to the compiler that checks moves.
121struct DropScopeToken;
122impl DropScopeToken {
123    fn pop_and_drop(
124        self,
125        ctx: &mut MirLowerCtx<'_>,
126        current: BasicBlockId,
127        span: MirSpan,
128    ) -> BasicBlockId {
129        std::mem::forget(self);
130        ctx.pop_drop_scope_internal(current, span)
131    }
132
133    /// It is useful when we want a drop scope is syntactically closed, but we don't want to execute any drop
134    /// code. Either when the control flow is diverging (so drop code doesn't reached) or when drop is handled
135    /// for us (for example a block that ended with a return statement. Return will drop everything, so the block shouldn't
136    /// do anything)
137    fn pop_assume_dropped(self, ctx: &mut MirLowerCtx<'_>) {
138        std::mem::forget(self);
139        ctx.pop_drop_scope_assume_dropped_internal();
140    }
141}

158impl MirLowerError {
159    pub fn pretty_print(
160        &self,
161        f: &mut String,
162        db: &dyn HirDatabase,
163        span_formatter: impl Fn(FileId, TextRange) -> String,
164        display_target: DisplayTarget,
165    ) -> std::result::Result<(), std::fmt::Error> {
166        match self {
167            MirLowerError::ConstEvalError(name, e) => {
168                writeln!(f, "In evaluating constant {name}")?;
169                match &**e {
170                    ConstEvalError::MirLowerError(e) => {
171                        e.pretty_print(f, db, span_formatter, display_target)?
172                    }
173                    ConstEvalError::MirEvalError(e) => {
174                        e.pretty_print(f, db, span_formatter, display_target)?
175                    }
176                }
177            }
178            MirLowerError::MissingFunctionDefinition(owner, it) => {
179                let body = db.body(*owner);
180                writeln!(
181                    f,
182                    "Missing function definition for {}",
183                    body.pretty_print_expr(db, *owner, *it, display_target.edition)
184                )?;
185            }
186            MirLowerError::HasErrors => writeln!(f, "Type inference result contains errors")?,
187            MirLowerError::TypeMismatch(e) => writeln!(
188                f,
189                "Type mismatch: Expected {}, found {}",
190                e.expected.display(db, display_target),
191                e.actual.display(db, display_target),
192            )?,
193            MirLowerError::GenericArgNotProvided(id, subst) => {
194                let parent = id.parent;
195                let param = &db.generic_params(parent)[id.local_id];
196                writeln!(
197                    f,
198                    "Generic arg not provided for {}",
199                    param.name().unwrap_or(&Name::missing()).display(db, display_target.edition)
200                )?;
201                writeln!(f, "Provided args: [")?;
202                for g in subst.iter(Interner) {
203                    write!(f, "    {},", g.display(db, display_target))?;
204                }
205                writeln!(f, "]")?;
206            }
207            MirLowerError::LayoutError(_)
208            | MirLowerError::UnsizedTemporary(_)
209            | MirLowerError::IncompleteExpr
210            | MirLowerError::IncompletePattern
211            | MirLowerError::InaccessibleLocal
212            | MirLowerError::TraitFunctionDefinition(_, _)
213            | MirLowerError::UnresolvedName(_)
214            | MirLowerError::RecordLiteralWithoutPath
215            | MirLowerError::UnresolvedMethod(_)
216            | MirLowerError::UnresolvedField
217            | MirLowerError::TypeError(_)
218            | MirLowerError::NotSupported(_)
219            | MirLowerError::ContinueWithoutLoop
220            | MirLowerError::BreakWithoutLoop
221            | MirLowerError::Loop
222            | MirLowerError::ImplementationError(_)
223            | MirLowerError::LangItemNotFound(_)
224            | MirLowerError::MutatingRvalue
225            | MirLowerError::UnresolvedLabel
226            | MirLowerError::UnresolvedUpvar(_) => writeln!(f, "{self:?}")?,
227        }
228        Ok(())
229    }
230}

309    fn temp(&mut self, ty: Ty, current: BasicBlockId, span: MirSpan) -> Result<LocalId> {
310        if matches!(ty.kind(Interner), TyKind::Slice(_) | TyKind::Dyn(_)) {
311            return Err(MirLowerError::UnsizedTemporary(ty));
312        }
313        let l = self.result.locals.alloc(Local { ty });
314        self.push_storage_live_for_local(l, current, span)?;
315        Ok(l)
316    }

318    fn lower_expr_to_some_operand(
319        &mut self,
320        expr_id: ExprId,
321        current: BasicBlockId,
322    ) -> Result<Option<(Operand, BasicBlockId)>> {
323        if !self.has_adjustments(expr_id)
324            && let Expr::Literal(l) = &self.body[expr_id]
325        {
326            let ty = self.expr_ty_without_adjust(expr_id);
327            return Ok(Some((self.lower_literal_to_operand(ty, l)?, current)));
328        }
329        let Some((p, current)) = self.lower_expr_as_place(current, expr_id, true)? else {
330            return Ok(None);
331        };
332        Ok(Some((Operand { kind: OperandKind::Copy(p), span: Some(expr_id.into()) }, current)))
333    }

335    fn lower_expr_to_place_with_adjust(
336        &mut self,
337        expr_id: ExprId,
338        place: Place,
339        current: BasicBlockId,
340        adjustments: &[Adjustment],
341    ) -> Result<Option<BasicBlockId>> {
342        match adjustments.split_last() {
343            Some((last, rest)) => match &last.kind {
344                Adjust::NeverToAny => {
345                    let temp =
346                        self.temp(TyKind::Never.intern(Interner), current, MirSpan::Unknown)?;
347                    self.lower_expr_to_place_with_adjust(expr_id, temp.into(), current, rest)
348                }
349                Adjust::Deref(_) => {
350                    let Some((p, current)) =
351                        self.lower_expr_as_place_with_adjust(current, expr_id, true, adjustments)?
352                    else {
353                        return Ok(None);
354                    };
355                    self.push_assignment(
356                        current,
357                        place,
358                        Operand { kind: OperandKind::Copy(p), span: None }.into(),
359                        expr_id.into(),
360                    );
361                    Ok(Some(current))
362                }
363                Adjust::Borrow(AutoBorrow::Ref(_, m) | AutoBorrow::RawPtr(m)) => {
364                    let Some((p, current)) =
365                        self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
366                    else {
367                        return Ok(None);
368                    };
369                    let bk = BorrowKind::from_chalk(*m);
370                    self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
371                    Ok(Some(current))
372                }
373                Adjust::Pointer(cast) => {
374                    let Some((p, current)) =
375                        self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
376                    else {
377                        return Ok(None);
378                    };
379                    self.push_assignment(
380                        current,
381                        place,
382                        Rvalue::Cast(
383                            CastKind::PointerCoercion(*cast),
384                            Operand { kind: OperandKind::Copy(p), span: None },
385                            last.target.clone(),
386                        ),
387                        expr_id.into(),
388                    );
389                    Ok(Some(current))
390                }
391            },
392            None => self.lower_expr_to_place_without_adjust(expr_id, place, current),
393        }
394    }

396    fn lower_expr_to_place(
397        &mut self,
398        expr_id: ExprId,
399        place: Place,
400        prev_block: BasicBlockId,
401    ) -> Result<Option<BasicBlockId>> {
402        if let Some(adjustments) = self.infer.expr_adjustments.get(&expr_id) {
403            return self.lower_expr_to_place_with_adjust(expr_id, place, prev_block, adjustments);
404        }
405        self.lower_expr_to_place_without_adjust(expr_id, place, prev_block)
406    }

408    fn lower_expr_to_place_without_adjust(
409        &mut self,
410        expr_id: ExprId,
411        place: Place,
412        mut current: BasicBlockId,
413    ) -> Result<Option<BasicBlockId>> {
414        match &self.body[expr_id] {
415            Expr::OffsetOf(_) => {
416                not_supported!("builtin#offset_of")
417            }
418            Expr::InlineAsm(_) => {
419                not_supported!("builtin#asm")
420            }
421            Expr::Missing => {
422                if let DefWithBodyId::FunctionId(f) = self.owner {
423                    let assoc = f.lookup(self.db);
424                    if let ItemContainerId::TraitId(t) = assoc.container {
425                        let name = &self.db.function_signature(f).name;
426                        return Err(MirLowerError::TraitFunctionDefinition(t, name.clone()));
427                    }
428                }
429                Err(MirLowerError::IncompleteExpr)
430            }
431            Expr::Path(p) => {
432                let pr =
433                    if let Some((assoc, subst)) = self.infer.assoc_resolutions_for_expr(expr_id) {
434                        match assoc {
435                            hir_def::AssocItemId::ConstId(c) => {
436                                self.lower_const(
437                                    c.into(),
438                                    current,
439                                    place,
440                                    subst,
441                                    expr_id.into(),
442                                    self.expr_ty_without_adjust(expr_id),
443                                )?;
444                                return Ok(Some(current));
445                            }
446                            hir_def::AssocItemId::FunctionId(_) => {
447                                // FnDefs are zero sized, no action is needed.
448                                return Ok(Some(current));
449                            }
450                            hir_def::AssocItemId::TypeAliasId(_) => {
451                                // FIXME: If it is unreachable, use proper error instead of `not_supported`.
452                                not_supported!("associated functions and types")
453                            }
454                        }
455                    } else if let Some(variant) = self.infer.variant_resolution_for_expr(expr_id) {
456                        match variant {
457                            VariantId::EnumVariantId(e) => ValueNs::EnumVariantId(e),
458                            VariantId::StructId(s) => ValueNs::StructId(s),
459                            VariantId::UnionId(_) => implementation_error!("Union variant as path"),
460                        }
461                    } else {
462                        let resolver_guard =
463                            self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
464                        let hygiene = self.body.expr_path_hygiene(expr_id);
465                        let result = self
466                            .resolver
467                            .resolve_path_in_value_ns_fully(self.db, p, hygiene)
468                            .ok_or_else(|| {
469                                MirLowerError::unresolved_path(
470                                    self.db,
471                                    p,
472                                    DisplayTarget::from_crate(self.db, self.krate()),
473                                    self.body,
474                                )
475                            })?;
476                        self.resolver.reset_to_guard(resolver_guard);
477                        result
478                    };
479                match pr {
480                    ValueNs::LocalBinding(_) | ValueNs::StaticId(_) => {
481                        let Some((temp, current)) =
482                            self.lower_expr_as_place_without_adjust(current, expr_id, false)?
483                        else {
484                            return Ok(None);
485                        };
486                        self.push_assignment(
487                            current,
488                            place,
489                            Operand { kind: OperandKind::Copy(temp), span: None }.into(),
490                            expr_id.into(),
491                        );
492                        Ok(Some(current))
493                    }
494                    ValueNs::ConstId(const_id) => {
495                        self.lower_const(
496                            const_id.into(),
497                            current,
498                            place,
499                            Substitution::empty(Interner),
500                            expr_id.into(),
501                            self.expr_ty_without_adjust(expr_id),
502                        )?;
503                        Ok(Some(current))
504                    }
505                    ValueNs::EnumVariantId(variant_id) => {
506                        let variant_fields = variant_id.fields(self.db);
507                        if variant_fields.shape == FieldsShape::Unit {
508                            let ty = self.infer.type_of_expr[expr_id].clone();
509                            current = self.lower_enum_variant(
510                                variant_id,
511                                current,
512                                place,
513                                ty,
514                                Box::new([]),
515                                expr_id.into(),
516                            )?;
517                        }
518                        // Otherwise its a tuple like enum, treated like a zero sized function, so no action is needed
519                        Ok(Some(current))
520                    }
521                    ValueNs::GenericParam(p) => {
522                        let Some(def) = self.owner.as_generic_def_id(self.db) else {
523                            not_supported!("owner without generic def id");
524                        };
525                        let generics = generics(self.db, def);
526                        let ty = self.expr_ty_without_adjust(expr_id);
527                        self.push_assignment(
528                            current,
529                            place,
530                            Rvalue::from(Operand {
531                                kind: OperandKind::Constant(
532                                    ConstData {
533                                        ty,
534                                        value: chalk_ir::ConstValue::BoundVar(BoundVar::new(
535                                            DebruijnIndex::INNERMOST,
536                                            generics.type_or_const_param_idx(p.into()).ok_or(
537                                                MirLowerError::TypeError(
538                                                    "fail to lower const generic param",
539                                                ),
540                                            )?,
541                                        )),
542                                    }
543                                    .intern(Interner),
544                                ),
545                                span: None,
546                            }),
547                            expr_id.into(),
548                        );
549                        Ok(Some(current))
550                    }
551                    ValueNs::FunctionId(_) | ValueNs::StructId(_) | ValueNs::ImplSelf(_) => {
552                        // It's probably a unit struct or a zero sized function, so no action is needed.
553                        Ok(Some(current))
554                    }
555                }
556            }
557            Expr::If { condition, then_branch, else_branch } => {
558                let Some((discr, current)) =
559                    self.lower_expr_to_some_operand(*condition, current)?
560                else {
561                    return Ok(None);
562                };
563                let start_of_then = self.new_basic_block();
564                let end_of_then = self.lower_expr_to_place(*then_branch, place, start_of_then)?;
565                let start_of_else = self.new_basic_block();
566                let end_of_else = if let Some(else_branch) = else_branch {
567                    self.lower_expr_to_place(*else_branch, place, start_of_else)?
568                } else {
569                    Some(start_of_else)
570                };
571                self.set_terminator(
572                    current,
573                    TerminatorKind::SwitchInt {
574                        discr,
575                        targets: SwitchTargets::static_if(1, start_of_then, start_of_else),
576                    },
577                    expr_id.into(),
578                );
579                Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()))
580            }
581            Expr::Let { pat, expr } => {
582                let Some((cond_place, current)) = self.lower_expr_as_place(current, *expr, true)?
583                else {
584                    return Ok(None);
585                };
586                self.push_fake_read(current, cond_place, expr_id.into());
587                let resolver_guard =
588                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
589                let (then_target, else_target) =
590                    self.pattern_match(current, None, cond_place, *pat)?;
591                self.resolver.reset_to_guard(resolver_guard);
592                self.write_bytes_to_place(
593                    then_target,
594                    place,
595                    Box::new([1]),
596                    TyBuilder::bool(),
597                    MirSpan::Unknown,
598                )?;
599                if let Some(else_target) = else_target {
600                    self.write_bytes_to_place(
601                        else_target,
602                        place,
603                        Box::new([0]),
604                        TyBuilder::bool(),
605                        MirSpan::Unknown,
606                    )?;
607                }
608                Ok(self.merge_blocks(Some(then_target), else_target, expr_id.into()))
609            }
610            Expr::Unsafe { id: _, statements, tail } => {
611                self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
612            }
613            Expr::Block { id: _, statements, tail, label } => {
614                if let Some(label) = label {
615                    self.lower_loop(current, place, Some(*label), expr_id.into(), |this, begin| {
616                        if let Some(current) = this.lower_block_to_place(
617                            statements,
618                            begin,
619                            *tail,
620                            place,
621                            expr_id.into(),
622                        )? {
623                            let end = this.current_loop_end()?;
624                            this.set_goto(current, end, expr_id.into());
625                        }
626                        Ok(())
627                    })
628                } else {
629                    self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
630                }
631            }
632            Expr::Loop { body, label } => {
633                self.lower_loop(current, place, *label, expr_id.into(), |this, begin| {
634                    let scope = this.push_drop_scope();
635                    if let Some((_, mut current)) = this.lower_expr_as_place(begin, *body, true)? {
636                        current = scope.pop_and_drop(this, current, body.into());
637                        this.set_goto(current, begin, expr_id.into());
638                    } else {
639                        scope.pop_assume_dropped(this);
640                    }
641                    Ok(())
642                })
643            }
644            Expr::Call { callee, args, .. } => {
645                if let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id) {
646                    let ty = chalk_ir::TyKind::FnDef(
647                        CallableDefId::FunctionId(func_id).to_chalk(self.db),
648                        generic_args,
649                    )
650                    .intern(Interner);
651                    let func = Operand::from_bytes(Box::default(), ty);
652                    return self.lower_call_and_args(
653                        func,
654                        iter::once(*callee).chain(args.iter().copied()),
655                        place,
656                        current,
657                        self.is_uninhabited(expr_id),
658                        expr_id.into(),
659                    );
660                }
661                let callee_ty = self.expr_ty_after_adjustments(*callee);
662                match &callee_ty.kind(Interner) {
663                    chalk_ir::TyKind::FnDef(..) => {
664                        let func = Operand::from_bytes(Box::default(), callee_ty.clone());
665                        self.lower_call_and_args(
666                            func,
667                            args.iter().copied(),
668                            place,
669                            current,
670                            self.is_uninhabited(expr_id),
671                            expr_id.into(),
672                        )
673                    }
674                    chalk_ir::TyKind::Function(_) => {
675                        let Some((func, current)) =
676                            self.lower_expr_to_some_operand(*callee, current)?
677                        else {
678                            return Ok(None);
679                        };
680                        self.lower_call_and_args(
681                            func,
682                            args.iter().copied(),
683                            place,
684                            current,
685                            self.is_uninhabited(expr_id),
686                            expr_id.into(),
687                        )
688                    }
689                    TyKind::Closure(_, _) => {
690                        not_supported!(
691                            "method resolution not emitted for closure (Are Fn traits available?)"
692                        );
693                    }
694                    TyKind::Error => {
695                        Err(MirLowerError::MissingFunctionDefinition(self.owner, expr_id))
696                    }
697                    _ => Err(MirLowerError::TypeError("function call on bad type")),
698                }
699            }
700            Expr::MethodCall { receiver, args, method_name, .. } => {
701                let (func_id, generic_args) =
702                    self.infer.method_resolution(expr_id).ok_or_else(|| {
703                        MirLowerError::UnresolvedMethod(
704                            method_name.display(self.db, self.edition()).to_string(),
705                        )
706                    })?;
707                let func = Operand::from_fn(self.db, func_id, generic_args);
708                self.lower_call_and_args(
709                    func,
710                    iter::once(*receiver).chain(args.iter().copied()),
711                    place,
712                    current,
713                    self.is_uninhabited(expr_id),
714                    expr_id.into(),
715                )
716            }
717            Expr::Match { expr, arms } => {
718                let Some((cond_place, mut current)) =
719                    self.lower_expr_as_place(current, *expr, true)?
720                else {
721                    return Ok(None);
722                };
723                self.push_fake_read(current, cond_place, expr_id.into());
724                let mut end = None;
725                let resolver_guard =
726                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
727                for MatchArm { pat, guard, expr } in arms.iter() {
728                    let (then, mut otherwise) =
729                        self.pattern_match(current, None, cond_place, *pat)?;
730                    let then = if let &Some(guard) = guard {
731                        let next = self.new_basic_block();
732                        let o = otherwise.get_or_insert_with(|| self.new_basic_block());
733                        if let Some((discr, c)) = self.lower_expr_to_some_operand(guard, then)? {
734                            self.set_terminator(
735                                c,
736                                TerminatorKind::SwitchInt {
737                                    discr,
738                                    targets: SwitchTargets::static_if(1, next, *o),
739                                },
740                                expr_id.into(),
741                            );
742                        }
743                        next
744                    } else {
745                        then
746                    };
747                    if let Some(block) = self.lower_expr_to_place(*expr, place, then)? {
748                        let r = end.get_or_insert_with(|| self.new_basic_block());
749                        self.set_goto(block, *r, expr_id.into());
750                    }
751                    match otherwise {
752                        Some(o) => current = o,
753                        None => {
754                            // The current pattern was irrefutable, so there is no need to generate code
755                            // for the rest of patterns
756                            break;
757                        }
758                    }
759                }
760                self.resolver.reset_to_guard(resolver_guard);
761                if self.is_unterminated(current) {
762                    self.set_terminator(current, TerminatorKind::Unreachable, expr_id.into());
763                }
764                Ok(end)
765            }
766            Expr::Continue { label } => {
767                let loop_data = match label {
768                    Some(l) => {
769                        self.labeled_loop_blocks.get(l).ok_or(MirLowerError::UnresolvedLabel)?
770                    }
771                    None => self
772                        .current_loop_blocks
773                        .as_ref()
774                        .ok_or(MirLowerError::ContinueWithoutLoop)?,
775                };
776                let begin = loop_data.begin;
777                current =
778                    self.drop_until_scope(loop_data.drop_scope_index, current, expr_id.into());
779                self.set_goto(current, begin, expr_id.into());
780                Ok(None)
781            }
782            &Expr::Break { expr, label } => {
783                if let Some(expr) = expr {
784                    let loop_data = match label {
785                        Some(l) => self
786                            .labeled_loop_blocks
787                            .get(&l)
788                            .ok_or(MirLowerError::UnresolvedLabel)?,
789                        None => self
790                            .current_loop_blocks
791                            .as_ref()
792                            .ok_or(MirLowerError::BreakWithoutLoop)?,
793                    };
794                    let Some(c) = self.lower_expr_to_place(expr, loop_data.place, current)? else {
795                        return Ok(None);
796                    };
797                    current = c;
798                }
799                let (end, drop_scope) = match label {
800                    Some(l) => {
801                        let loop_blocks = self
802                            .labeled_loop_blocks
803                            .get(&l)
804                            .ok_or(MirLowerError::UnresolvedLabel)?;
805                        (
806                            loop_blocks.end.expect("We always generate end for labeled loops"),
807                            loop_blocks.drop_scope_index,
808                        )
809                    }
810                    None => (
811                        self.current_loop_end()?,
812                        self.current_loop_blocks.as_ref().unwrap().drop_scope_index,
813                    ),
814                };
815                current = self.drop_until_scope(drop_scope, current, expr_id.into());
816                self.set_goto(current, end, expr_id.into());
817                Ok(None)
818            }
819            Expr::Return { expr } => {
820                if let Some(expr) = expr {
821                    if let Some(c) =
822                        self.lower_expr_to_place(*expr, return_slot().into(), current)?
823                    {
824                        current = c;
825                    } else {
826                        return Ok(None);
827                    }
828                }
829                current = self.drop_until_scope(0, current, expr_id.into());
830                self.set_terminator(current, TerminatorKind::Return, expr_id.into());
831                Ok(None)
832            }
833            Expr::Become { .. } => not_supported!("tail-calls"),
834            Expr::Yield { .. } => not_supported!("yield"),
835            Expr::RecordLit { fields, path, spread } => {
836                let spread_place = match spread {

839                            return Ok(None);

865                                variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;

867                            else {
868                                return Ok(None);
869                            };
870                            current = c;
871                            operands[u32::from(field_id.into_raw()) as usize] = Some(op);
872                        }
873                        let rvalue = Rvalue::Aggregate(
874                            AggregateKind::Adt(variant_id, subst),
875                            match spread_place {
876                                Some(sp) => operands
877                                    .into_iter()
878                                    .enumerate()
879                                    .map(|(i, it)| match it {
880                                        Some(it) => it,
881                                        None => {
882                                            let p = sp.project(
883                                                ProjectionElem::Field(Either::Left(FieldId {
884                                                    parent: variant_id,
885                                                    local_id: LocalFieldId::from_raw(RawIdx::from(
886                                                        i as u32,
887                                                    )),
888                                                })),
889                                                &mut self.result.projection_store,
890                                            );
891                                            Operand { kind: OperandKind::Copy(p), span: None }
892                                        }
893                                    })
894                                    .collect(),
895                                None => operands.into_iter().collect::<Option<_>>().ok_or(
896                                    MirLowerError::TypeError("missing field in record literal"),
897                                )?,
898                            },
899                        );
900                        self.push_assignment(current, place, rvalue, expr_id.into());
901                        Ok(Some(current))
902                    }
903                    VariantId::UnionId(union_id) => {
904                        let [RecordLitField { name, expr }] = fields.as_ref() else {
905                            not_supported!("Union record literal with more than one field");
906                        };
907                        let local_id =
908                            variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;
909                        let place = place.project(
910                            PlaceElem::Field(Either::Left(FieldId {
911                                parent: union_id.into(),
912                                local_id,
913                            })),
914                            &mut self.result.projection_store,
915                        );
916                        self.lower_expr_to_place(*expr, place, current)
917                    }
918                }
919            }
920            Expr::Await { .. } => not_supported!("await"),
921            Expr::Yeet { .. } => not_supported!("yeet"),
922            Expr::Async { .. } => not_supported!("async block"),
923            &Expr::Const(_) => {
924                // let subst = self.placeholder_subst();
925                // self.lower_const(
926                //     id.into(),
927                //     current,
928                //     place,
929                //     subst,
930                //     expr_id.into(),
931                //     self.expr_ty_without_adjust(expr_id),
932                // )?;
933                // Ok(Some(current))
934                not_supported!("const block")
935            }
936            Expr::Cast { expr, type_ref: _ } => {
937                let Some((it, current)) = self.lower_expr_to_some_operand(*expr, current)? else {
938                    return Ok(None);
939                };
940                // Since we don't have THIR, this is the "zipped" version of [rustc's HIR lowering](https://github.com/rust-lang/rust/blob/e71f9529121ca8f687e4b725e3c9adc3f1ebab4d/compiler/rustc_mir_build/src/thir/cx/expr.rs#L165-L178)

943                    Rvalue::Use(it)
944                } else {
945                    let source_ty = self.infer[*expr].clone();
946                    let target_ty = self.infer[expr_id].clone();
947                    let cast_kind = if source_ty.as_reference().is_some() {
948                        CastKind::PointerCoercion(PointerCast::ArrayToPointer)
949                    } else {
950                        let mut table = InferenceTable::new(self.db, self.env.clone());
951                        cast_kind(&mut table, &source_ty, &target_ty)?
952                    };
953
954                    Rvalue::Cast(cast_kind, it, target_ty)
955                };
956                self.push_assignment(current, place, rvalue, expr_id.into());
957                Ok(Some(current))
958            }
959            Expr::Ref { expr, rawness: _, mutability } => {
960                let Some((p, current)) = self.lower_expr_as_place(current, *expr, true)? else {
961                    return Ok(None);
962                };
963                let bk = BorrowKind::from_hir(*mutability);
964                self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
965                Ok(Some(current))
966            }
967            Expr::Box { expr } => {
968                let ty = self.expr_ty_after_adjustments(*expr);
969                self.push_assignment(
970                    current,
971                    place,
972                    Rvalue::ShallowInitBoxWithAlloc(ty),
973                    expr_id.into(),
974                );
975                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
976                else {
977                    return Ok(None);
978                };
979                let p = place.project(ProjectionElem::Deref, &mut self.result.projection_store);
980                self.push_assignment(current, p, operand.into(), expr_id.into());
981                Ok(Some(current))
982            }
983            Expr::Field { .. }
984            | Expr::Index { .. }
985            | Expr::UnaryOp { op: hir_def::hir::UnaryOp::Deref, .. } => {
986                let Some((p, current)) =
987                    self.lower_expr_as_place_without_adjust(current, expr_id, true)?
988                else {
989                    return Ok(None);
990                };
991                self.push_assignment(
992                    current,
993                    place,
994                    Operand { kind: OperandKind::Copy(p), span: None }.into(),
995                    expr_id.into(),
996                );
997                Ok(Some(current))
998            }
999            Expr::UnaryOp {
1000                expr,
1001                op: op @ (hir_def::hir::UnaryOp::Not | hir_def::hir::UnaryOp::Neg),
1002            } => {
1003                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
1004                else {
1005                    return Ok(None);
1006                };
1007                let operation = match op {
1008                    hir_def::hir::UnaryOp::Not => UnOp::Not,
1009                    hir_def::hir::UnaryOp::Neg => UnOp::Neg,
1010                    _ => unreachable!(),
1011                };
1012                self.push_assignment(
1013                    current,
1014                    place,
1015                    Rvalue::UnaryOp(operation, operand),
1016                    expr_id.into(),
1017                );
1018                Ok(Some(current))
1019            }
1020            Expr::BinaryOp { lhs, rhs, op } => {
1021                let op: BinaryOp = op.ok_or(MirLowerError::IncompleteExpr)?;
1022                let is_builtin = 'b: {
1023                    // Without adjust here is a hack. We assume that we know every possible adjustment
1024                    // for binary operator, and use without adjust to simplify our conditions.
1025                    let lhs_ty = self.expr_ty_without_adjust(*lhs);
1026                    let rhs_ty = self.expr_ty_without_adjust(*rhs);
1027                    if matches!(op, BinaryOp::CmpOp(syntax::ast::CmpOp::Eq { .. }))
1028                        && lhs_ty.as_raw_ptr().is_some()
1029                        && rhs_ty.as_raw_ptr().is_some()
1030                    {
1031                        break 'b true;
1032                    }
1033                    let builtin_inequal_impls = matches!(
1034                        op,
1035                        BinaryOp::ArithOp(ArithOp::Shl | ArithOp::Shr)
1036                            | BinaryOp::Assignment { op: Some(ArithOp::Shl | ArithOp::Shr) }
1037                    );
1038                    lhs_ty.is_scalar()
1039                        && rhs_ty.is_scalar()
1040                        && (lhs_ty == rhs_ty || builtin_inequal_impls)
1041                };
1042                if !is_builtin
1043                    && let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id)
1044                {
1045                    let func = Operand::from_fn(self.db, func_id, generic_args);
1046                    return self.lower_call_and_args(
1047                        func,
1048                        [*lhs, *rhs].into_iter(),
1049                        place,
1050                        current,
1051                        self.is_uninhabited(expr_id),
1052                        expr_id.into(),
1053                    );
1054                }
1055                if let hir_def::hir::BinaryOp::Assignment { op: Some(op) } = op {
1056                    // last adjustment is `&mut` which we don't want it.
1057                    let adjusts = self
1058                        .infer
1059                        .expr_adjustments
1060                        .get(lhs)
1061                        .and_then(|it| it.split_last())
1062                        .map(|it| it.1)
1063                        .ok_or(MirLowerError::TypeError("adjustment of binary op was missing"))?;
1064                    let Some((lhs_place, current)) =
1065                        self.lower_expr_as_place_with_adjust(current, *lhs, false, adjusts)?
1066                    else {
1067                        return Ok(None);
1068                    };
1069                    let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1070                    else {
1071                        return Ok(None);
1072                    };
1073                    let r_value = Rvalue::CheckedBinaryOp(
1074                        op.into(),
1075                        Operand { kind: OperandKind::Copy(lhs_place), span: None },
1076                        rhs_op,
1077                    );
1078                    self.push_assignment(current, lhs_place, r_value, expr_id.into());
1079                    return Ok(Some(current));
1080                }
1081                let Some((lhs_op, current)) = self.lower_expr_to_some_operand(*lhs, current)?
1082                else {
1083                    return Ok(None);
1084                };
1085                if let hir_def::hir::BinaryOp::LogicOp(op) = op {
1086                    let value_to_short = match op {
1087                        syntax::ast::LogicOp::And => 0,
1088                        syntax::ast::LogicOp::Or => 1,
1089                    };
1090                    let start_of_then = self.new_basic_block();
1091                    self.push_assignment(
1092                        start_of_then,
1093                        place,
1094                        lhs_op.clone().into(),
1095                        expr_id.into(),
1096                    );
1097                    let end_of_then = Some(start_of_then);
1098                    let start_of_else = self.new_basic_block();
1099                    let end_of_else = self.lower_expr_to_place(*rhs, place, start_of_else)?;
1100                    self.set_terminator(
1101                        current,
1102                        TerminatorKind::SwitchInt {
1103                            discr: lhs_op,
1104                            targets: SwitchTargets::static_if(
1105                                value_to_short,
1106                                start_of_then,
1107                                start_of_else,
1108                            ),
1109                        },
1110                        expr_id.into(),
1111                    );
1112                    return Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()));
1113                }
1114                let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1115                else {
1116                    return Ok(None);
1117                };
1118                self.push_assignment(
1119                    current,
1120                    place,
1121                    Rvalue::CheckedBinaryOp(
1122                        match op {
1123                            hir_def::hir::BinaryOp::LogicOp(op) => match op {
1124                                hir_def::hir::LogicOp::And => BinOp::BitAnd, // FIXME: make these short circuit
1125                                hir_def::hir::LogicOp::Or => BinOp::BitOr,
1126                            },
1127                            hir_def::hir::BinaryOp::ArithOp(op) => BinOp::from(op),
1128                            hir_def::hir::BinaryOp::CmpOp(op) => BinOp::from(op),
1129                            hir_def::hir::BinaryOp::Assignment { .. } => unreachable!(), // handled above
1130                        },
1131                        lhs_op,
1132                        rhs_op,
1133                    ),
1134                    expr_id.into(),
1135                );
1136                Ok(Some(current))
1137            }
1138            &Expr::Assignment { target, value } => {

1140                else {

1149            }

1176                    place,
1177                    Rvalue::Aggregate(

1181                            .iter()

1190                                };

1198            }

1224                                    }

1227                                    }

1230                                    }

1233                                })

1236                    };

1238                        CaptureKind::ByRef(bk) => {

1241                                capture.ty.clone().substitute(Interner, &placeholder_subst);
1242                            // FIXME: Handle more than one span.

1246                                current,
1247                                tmp,
1248                                Rvalue::Ref(*bk, p),
1249                                capture_spans[0],
1250                            );
1251                            operands.push(Operand { kind: OperandKind::Move(tmp), span: None });
1252                        }
1253                        CaptureKind::ByValue => {
1254                            operands.push(Operand { kind: OperandKind::Move(p), span: None })
1255                        }
1256                    }
1257                }
1258                self.push_assignment(
1259                    current,
1260                    place,
1261                    Rvalue::Aggregate(AggregateKind::Closure(ty), operands.into()),
1262                    expr_id.into(),
1263                );
1264                Ok(Some(current))
1265            }
1266            Expr::Tuple { exprs } => {
1267                let Some(values) = exprs
1268                    .iter()
1269                    .map(|it| {
1270                        let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)? else {
1271                            return Ok(None);
1272                        };
1273                        current = c;
1274                        Ok(Some(o))
1275                    })
1276                    .collect::<Result<Option<_>>>()?
1277                else {
1278                    return Ok(None);
1279                };
1280                let r = Rvalue::Aggregate(
1281                    AggregateKind::Tuple(self.expr_ty_without_adjust(expr_id)),
1282                    values,
1283                );
1284                self.push_assignment(current, place, r, expr_id.into());
1285                Ok(Some(current))
1286            }
1287            Expr::Array(l) => match l {
1288                Array::ElementList { elements, .. } => {
1289                    let elem_ty = match &self.expr_ty_without_adjust(expr_id).kind(Interner) {
1290                        TyKind::Array(ty, _) => ty.clone(),
1291                        _ => {
1292                            return Err(MirLowerError::TypeError(
1293                                "Array expression with non array type",
1294                            ));
1295                        }
1296                    };
1297                    let Some(values) = elements
1298                        .iter()
1299                        .map(|it| {
1300                            let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)?
1301                            else {
1302                                return Ok(None);
1303                            };
1304                            current = c;
1305                            Ok(Some(o))
1306                        })
1307                        .collect::<Result<Option<_>>>()?
1308                    else {
1309                        return Ok(None);
1310                    };
1311                    let r = Rvalue::Aggregate(AggregateKind::Array(elem_ty), values);
1312                    self.push_assignment(current, place, r, expr_id.into());
1313                    Ok(Some(current))
1314                }
1315                Array::Repeat { initializer, .. } => {
1316                    let Some((init, current)) =
1317                        self.lower_expr_to_some_operand(*initializer, current)?
1318                    else {
1319                        return Ok(None);
1320                    };
1321                    let len = match &self.expr_ty_without_adjust(expr_id).kind(Interner) {
1322                        TyKind::Array(_, len) => len.clone(),
1323                        _ => {
1324                            return Err(MirLowerError::TypeError(
1325                                "Array repeat expression with non array type",
1326                            ));
1327                        }
1328                    };
1329                    let r = Rvalue::Repeat(init, len);
1330                    self.push_assignment(current, place, r, expr_id.into());
1331                    Ok(Some(current))
1332                }
1333            },
1334            Expr::Literal(l) => {
1335                let ty = self.expr_ty_without_adjust(expr_id);
1336                let op = self.lower_literal_to_operand(ty, l)?;
1337                self.push_assignment(current, place, op.into(), expr_id.into());
1338                Ok(Some(current))
1339            }
1340            Expr::Underscore => Ok(Some(current)),
1341        }
1342    }

1413    fn lower_literal_to_operand(&mut self, ty: Ty, l: &Literal) -> Result<Operand> {
1414        let size =
1415            || self.db.layout_of_ty(ty.clone(), self.env.clone()).map(|it| it.size.bytes_usize());
1416        const USIZE_SIZE: usize = size_of::<usize>();
1417        let bytes: Box<[_]> = match l {
1418            hir_def::hir::Literal::String(b) => {
1419                let b = b.as_str();
1420                let mut data = [0; { 2 * USIZE_SIZE }];
1421                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1422                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1423                let mm = MemoryMap::simple(b.as_bytes().into());
1424                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1425            }
1426            hir_def::hir::Literal::CString(b) => {
1427                let bytes = b.iter().copied().chain(iter::once(0)).collect::<Box<_>>();
1428
1429                let mut data = [0; { 2 * USIZE_SIZE }];
1430                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1431                data[USIZE_SIZE..].copy_from_slice(&bytes.len().to_le_bytes());
1432                let mm = MemoryMap::simple(bytes);
1433                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1434            }
1435            hir_def::hir::Literal::ByteString(b) => {
1436                let mut data = [0; { 2 * USIZE_SIZE }];
1437                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1438                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1439                let mm = MemoryMap::simple(b.clone());
1440                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1441            }
1442            hir_def::hir::Literal::Char(c) => Box::new(u32::from(*c).to_le_bytes()),
1443            hir_def::hir::Literal::Bool(b) => Box::new([*b as u8]),
1444            hir_def::hir::Literal::Int(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1445            hir_def::hir::Literal::Uint(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1446            hir_def::hir::Literal::Float(f, _) => match size()? {
1447                16 => Box::new(f.to_f128().to_bits().to_le_bytes()),
1448                8 => Box::new(f.to_f64().to_le_bytes()),
1449                4 => Box::new(f.to_f32().to_le_bytes()),
1450                2 => Box::new(u16::try_from(f.to_f16().to_bits()).unwrap().to_le_bytes()),
1451                _ => {
1452                    return Err(MirLowerError::TypeError(
1453                        "float with size other than 2, 4, 8 or 16 bytes",
1454                    ));
1455                }
1456            },
1457        };
1458        Ok(Operand::from_concrete_const(bytes, MemoryMap::default(), ty))
1459    }

1639    fn lower_loop(
1640        &mut self,
1641        prev_block: BasicBlockId,
1642        place: Place,
1643        label: Option<LabelId>,
1644        span: MirSpan,
1645        f: impl FnOnce(&mut MirLowerCtx<'_>, BasicBlockId) -> Result<()>,
1646    ) -> Result<Option<BasicBlockId>> {
1647        let begin = self.new_basic_block();
1648        let prev = self.current_loop_blocks.replace(LoopBlocks {
1649            begin,
1650            end: None,
1651            place,
1652            drop_scope_index: self.drop_scopes.len(),
1653        });
1654        let prev_label = if let Some(label) = label {
1655            // We should generate the end now, to make sure that it wouldn't change later. It is
1656            // bad as we may emit end (unnecessary unreachable block) for unterminating loop, but
1657            // it should not affect correctness.
1658            self.current_loop_end()?;
1659            self.labeled_loop_blocks
1660                .insert(label, self.current_loop_blocks.as_ref().unwrap().clone())
1661        } else {
1662            None
1663        };
1664        self.set_goto(prev_block, begin, span);
1665        f(self, begin)?;
1666        let my = mem::replace(&mut self.current_loop_blocks, prev).ok_or(
1667            MirLowerError::ImplementationError("current_loop_blocks is corrupt".to_owned()),
1668        )?;
1669        if let Some(prev) = prev_label {
1670            self.labeled_loop_blocks.insert(label.unwrap(), prev);
1671        }
1672        Ok(my.end)
1673    }

1753    fn lower_block_to_place(
1754        &mut self,
1755        statements: &[hir_def::hir::Statement],
1756        mut current: BasicBlockId,
1757        tail: Option<ExprId>,
1758        place: Place,
1759        span: MirSpan,
1760    ) -> Result<Option<Idx<BasicBlock>>> {
1761        let scope = self.push_drop_scope();
1762        for statement in statements.iter() {
1763            match statement {
1764                hir_def::hir::Statement::Let { pat, initializer, else_branch, type_ref: _ } => {
1765                    if let Some(expr_id) = initializer {
1766                        let else_block;
1767                        let Some((init_place, c)) =
1768                            self.lower_expr_as_place(current, *expr_id, true)?
1769                        else {
1770                            scope.pop_assume_dropped(self);
1771                            return Ok(None);
1772                        };
1773                        current = c;
1774                        self.push_fake_read(current, init_place, span);
1775                        // Using the initializer for the resolver scope is good enough for us, as it cannot create new declarations
1776                        // and has all declarations of the `let`.
1777                        let resolver_guard =
1778                            self.resolver.update_to_inner_scope(self.db, self.owner, *expr_id);
1779                        (current, else_block) =
1780                            self.pattern_match(current, None, init_place, *pat)?;
1781                        self.resolver.reset_to_guard(resolver_guard);
1782                        match (else_block, else_branch) {
1783                            (None, _) => (),
1784                            (Some(else_block), None) => {
1785                                self.set_terminator(else_block, TerminatorKind::Unreachable, span);
1786                            }
1787                            (Some(else_block), Some(else_branch)) => {
1788                                if let Some((_, b)) =
1789                                    self.lower_expr_as_place(else_block, *else_branch, true)?
1790                                {
1791                                    self.set_terminator(b, TerminatorKind::Unreachable, span);
1792                                }
1793                            }
1794                        }
1795                    } else {
1796                        let mut err = None;
1797                        self.body.walk_bindings_in_pat(*pat, |b| {
1798                            if let Err(e) = self.push_storage_live(b, current) {
1799                                err = Some(e);
1800                            }
1801                        });
1802                        if let Some(e) = err {
1803                            return Err(e);
1804                        }
1805                    }
1806                }
1807                &hir_def::hir::Statement::Expr { expr, has_semi: _ } => {
1808                    let scope2 = self.push_drop_scope();
1809                    let Some((p, c)) = self.lower_expr_as_place(current, expr, true)? else {
1810                        scope2.pop_assume_dropped(self);
1811                        scope.pop_assume_dropped(self);
1812                        return Ok(None);
1813                    };
1814                    self.push_fake_read(c, p, expr.into());
1815                    current = scope2.pop_and_drop(self, c, expr.into());
1816                }
1817                hir_def::hir::Statement::Item(_) => (),
1818            }
1819        }
1820        if let Some(tail) = tail {
1821            let Some(c) = self.lower_expr_to_place(tail, place, current)? else {
1822                scope.pop_assume_dropped(self);
1823                return Ok(None);
1824            };
1825            current = c;
1826        }
1827        current = scope.pop_and_drop(self, current, span);
1828        Ok(Some(current))
1829    }

1839            Some((self_binding, ty)) => {
1840                let local_id = self.result.locals.alloc(Local { ty });
1841                self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1842                self.result.binding_locals.insert(self_binding, local_id);
1843                self.result.param_locals.push(local_id);
1844                Some(self_binding)
1845            }
1846            None => None,
1847        };
1848        self.result.param_locals.extend(params.clone().map(|(it, ty)| {
1849            let local_id = self.result.locals.alloc(Local { ty });
1850            self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1851            if let Pat::Bind { id, subpat: None } = self.body[it]
1852                && matches!(
1853                    self.body[id].mode,
1854                    BindingAnnotation::Unannotated | BindingAnnotation::Mutable
1855                )
1856            {
1857                self.result.binding_locals.insert(id, local_id);
1858            }
1859            local_id
1860        }));
1861        // and then rest of bindings
1862        for (id, _) in self.body.bindings() {
1863            if !pick_binding(id) {
1864                continue;
1865            }
1866            if !self.result.binding_locals.contains_idx(id) {
1867                self.result
1868                    .binding_locals
1869                    .insert(id, self.result.locals.alloc(Local { ty: self.infer[id].clone() }));
1870            }
1871        }
1872        let mut current = self.result.start_block;
1873        if let Some(self_binding) = self_binding {
1874            let local = self.result.param_locals.clone()[base_param_count];
1875            if local != self.binding_local(self_binding)? {
1876                let r = self.match_self_param(self_binding, current, local)?;
1877                if let Some(b) = r.1 {
1878                    self.set_terminator(b, TerminatorKind::Unreachable, MirSpan::SelfParam);
1879                }
1880                current = r.0;
1881            }
1882        }
1883        let local_params = self
1884            .result
1885            .param_locals
1886            .clone()
1887            .into_iter()
1888            .skip(base_param_count + self_binding.is_some() as usize);
1889        for ((param, _), local) in params.zip(local_params) {
1890            if let Pat::Bind { id, .. } = self.body[param]
1891                && local == self.binding_local(id)?
1892            {
1893                continue;
1894            }
1895            let r = self.pattern_match(current, None, local.into(), param)?;
1896            if let Some(b) = r.1 {
1897                self.set_terminator(b, TerminatorKind::Unreachable, param.into());
1898            }
1899            current = r.0;
1900        }
1901        Ok(current)
1902    }

1984        &mut self,

1986        span: MirSpan,
1987    ) -> Result<BasicBlockId> {
1988        current = self.pop_drop_scope_internal(current, span);
1989        if !self.drop_scopes.is_empty() {
1990            implementation_error!("Mismatched count between drop scope push and pops");
1991        }
1992        Ok(current)

1995    fn emit_drop_and_storage_dead_for_scope(
1996        &mut self,
1997        scope: &DropScope,
1998        current: &mut Idx<BasicBlock>,
1999        span: MirSpan,
2000    ) {
2001        for &l in scope.locals.iter().rev() {
2002            if !self.result.locals[l].ty.clone().is_copy(self.db, self.owner) {
2003                let prev = std::mem::replace(current, self.new_basic_block());
2004                self.set_terminator(
2005                    prev,
2006                    TerminatorKind::Drop { place: l.into(), target: *current, unwind: None },
2007                    span,
2008                );
2009            }
2010            self.push_statement(*current, StatementKind::StorageDead(l).with_span(span));
2011        }
2012    }

2015fn cast_kind(table: &mut InferenceTable<'_>, source_ty: &Ty, target_ty: &Ty) -> Result<CastKind> {
2016    let from = CastTy::from_ty(table, source_ty);
2017    let cast = CastTy::from_ty(table, target_ty);
2018    Ok(match (from, cast) {
2019        (Some(CastTy::Ptr(..) | CastTy::FnPtr), Some(CastTy::Int(_))) => {
2020            CastKind::PointerExposeAddress
2021        }
2022        (Some(CastTy::Int(_)), Some(CastTy::Ptr(..))) => CastKind::PointerFromExposedAddress,
2023        (Some(CastTy::Int(_)), Some(CastTy::Int(_))) => CastKind::IntToInt,
2024        (Some(CastTy::FnPtr), Some(CastTy::Ptr(..))) => CastKind::FnPtrToPtr,
2025        (Some(CastTy::Float), Some(CastTy::Int(_))) => CastKind::FloatToInt,
2026        (Some(CastTy::Int(_)), Some(CastTy::Float)) => CastKind::IntToFloat,
2027        (Some(CastTy::Float), Some(CastTy::Float)) => CastKind::FloatToFloat,

2030    })

2033pub fn mir_body_for_closure_query(
2034    db: &dyn HirDatabase,
2035    closure: InternedClosureId,
2036) -> Result<Arc<MirBody>> {
2037    let InternedClosure(owner, expr) = db.lookup_intern_closure(closure);
2038    let body = db.body(owner);
2039    let infer = db.infer(owner);
2040    let Expr::Closure { args, body: root, .. } = &body[expr] else {
2041        implementation_error!("closure expression is not closure");
2042    };
2043    let TyKind::Closure(_, substs) = &infer[expr].kind(Interner) else {
2044        implementation_error!("closure expression is not closure");
2045    };
2046    let (captures, kind) = infer.closure_info(&closure.into());
2047    let mut ctx = MirLowerCtx::new(db, owner, &body, &infer);
2048    // 0 is return local
2049    ctx.result.locals.alloc(Local { ty: infer[*root].clone() });
2050    let closure_local = ctx.result.locals.alloc(Local {
2051        ty: match kind {
2052            FnTrait::FnOnce | FnTrait::AsyncFnOnce => infer[expr].clone(),
2053            FnTrait::FnMut | FnTrait::AsyncFnMut => {
2054                TyKind::Ref(Mutability::Mut, error_lifetime(), infer[expr].clone()).intern(Interner)
2055            }
2056            FnTrait::Fn | FnTrait::AsyncFn => {
2057                TyKind::Ref(Mutability::Not, error_lifetime(), infer[expr].clone()).intern(Interner)
2058            }
2059        },
2060    });
2061    ctx.result.param_locals.push(closure_local);
2062    let Some(sig) = ClosureSubst(substs).sig_ty().callable_sig(db) else {
2063        implementation_error!("closure has not callable sig");
2064    };
2065    let resolver_guard = ctx.resolver.update_to_inner_scope(db, owner, expr);
2066    let current = ctx.lower_params_and_bindings(
2067        args.iter().zip(sig.params().iter()).map(|(it, y)| (*it, y.clone())),
2068        None,
2069        |_| true,
2070    )?;
2071    ctx.resolver.reset_to_guard(resolver_guard);
2072    if let Some(current) = ctx.lower_expr_to_place(*root, return_slot().into(), current)? {
2073        let current = ctx.pop_drop_scope_assert_finished(current, root.into())?;
2074        ctx.set_terminator(current, TerminatorKind::Return, (*root).into());
2075    }
2076    let mut upvar_map: FxHashMap<LocalId, Vec<(&CapturedItem, usize)>> = FxHashMap::default();
2077    for (i, capture) in captures.iter().enumerate() {
2078        let local = ctx.binding_local(capture.place.local)?;
2079        upvar_map.entry(local).or_default().push((capture, i));
2080    }
2081    let mut err = None;
2082    let closure_local = ctx.result.locals.iter().nth(1).unwrap().0;
2083    let closure_projection = match kind {
2084        FnTrait::FnOnce | FnTrait::AsyncFnOnce => vec![],
2085        FnTrait::FnMut | FnTrait::Fn | FnTrait::AsyncFnMut | FnTrait::AsyncFn => {
2086            vec![ProjectionElem::Deref]
2087        }
2088    };
2089    ctx.result.walk_places(|p, store| {
2090        if let Some(it) = upvar_map.get(&p.local) {
2091            let r = it.iter().find(|it| {
2092                if p.projection.lookup(store).len() < it.0.place.projections.len() {
2093                    return false;
2094                }
2095                for (it, y) in p.projection.lookup(store).iter().zip(it.0.place.projections.iter())
2096                {
2097                    match (it, y) {
2098                        (ProjectionElem::Deref, ProjectionElem::Deref) => (),
2099                        (ProjectionElem::Field(it), ProjectionElem::Field(y)) if it == y => (),
2100                        (ProjectionElem::ClosureField(it), ProjectionElem::ClosureField(y))
2101                            if it == y => {}
2102                        _ => return false,
2103                    }
2104                }
2105                true
2106            });
2107            match r {
2108                Some(it) => {
2109                    p.local = closure_local;
2110                    let mut next_projs = closure_projection.clone();
2111                    next_projs.push(PlaceElem::ClosureField(it.1));
2112                    let prev_projs = p.projection;
2113                    if it.0.kind != CaptureKind::ByValue {
2114                        next_projs.push(ProjectionElem::Deref);
2115                    }
2116                    next_projs.extend(
2117                        prev_projs.lookup(store).iter().skip(it.0.place.projections.len()).cloned(),
2118                    );
2119                    p.projection = store.intern(next_projs.into());
2120                }
2121                None => err = Some(*p),
2122            }
2123        }
2124    });
2125    ctx.result.binding_locals = ctx
2126        .result
2127        .binding_locals
2128        .into_iter()
2129        .filter(|it| ctx.body.binding_owner(it.0) == Some(expr))
2130        .collect();
2131    if let Some(err) = err {
2132        return Err(MirLowerError::UnresolvedUpvar(err));
2133    }
2134    ctx.result.shrink_to_fit();
2135    Ok(Arc::new(ctx.result))
2136}

2178pub fn lower_to_mir(
2179    db: &dyn HirDatabase,
2180    owner: DefWithBodyId,
2181    body: &Body,
2182    infer: &InferenceResult,
2183    // FIXME: root_expr should always be the body.body_expr, but since `X` in `[(); X]` doesn't have its own specific body yet, we
2184    // need to take this input explicitly.
2185    root_expr: ExprId,
2186) -> Result<MirBody> {
2187    if infer.type_mismatches().next().is_some() || infer.is_erroneous() {
2188        return Err(MirLowerError::HasErrors);
2189    }
2190    let mut ctx = MirLowerCtx::new(db, owner, body, infer);
2191    // 0 is return local
2192    ctx.result.locals.alloc(Local { ty: ctx.expr_ty_after_adjustments(root_expr) });
2193    let binding_picker = |b: BindingId| {
2194        let owner = ctx.body.binding_owner(b);
2195        if root_expr == body.body_expr { owner.is_none() } else { owner == Some(root_expr) }
2196    };
2197    // 1 to param_len is for params
2198    // FIXME: replace with let chain once it becomes stable
2199    let current = 'b: {
2200        if body.body_expr == root_expr {
2201            // otherwise it's an inline const, and has no parameter
2202            if let DefWithBodyId::FunctionId(fid) = owner {
2203                let substs = TyBuilder::placeholder_subst(db, fid);
2204                let callable_sig =
2205                    db.callable_item_signature(fid.into()).substitute(Interner, &substs);
2206                let mut params = callable_sig.params().iter();
2207                let self_param = body.self_param.and_then(|id| Some((id, params.next()?.clone())));
2208                break 'b ctx.lower_params_and_bindings(
2209                    body.params.iter().zip(params).map(|(it, y)| (*it, y.clone())),
2210                    self_param,
2211                    binding_picker,
2212                )?;
2213            }
2214        }
2215        ctx.lower_params_and_bindings([].into_iter(), None, binding_picker)?
2216    };
2217    if let Some(current) = ctx.lower_expr_to_place(root_expr, return_slot().into(), current)? {
2218        let current = ctx.pop_drop_scope_assert_finished(current, root_expr.into())?;
2219        ctx.set_terminator(current, TerminatorKind::Return, root_expr.into());
2220    }
2221    Ok(ctx.result)
