9macro_rules! not_supported {

11        return Err(MirLowerError::NotSupported(format!($it)))

15impl MirLowerCtx<'_> {
16    fn lower_expr_to_some_place_without_adjust(
17        &mut self,
18        expr_id: ExprId,
19        prev_block: BasicBlockId,
20    ) -> Result<Option<(Place, BasicBlockId)>> {
21        let ty = self.expr_ty_without_adjust(expr_id);
22        let place = self.temp(ty, prev_block, expr_id.into())?;
23        let Some(current) =
24            self.lower_expr_to_place_without_adjust(expr_id, place.into(), prev_block)?
25        else {
26            return Ok(None);
27        };
28        Ok(Some((place.into(), current)))
29    }
30
31    fn lower_expr_to_some_place_with_adjust(
32        &mut self,
33        expr_id: ExprId,
34        prev_block: BasicBlockId,
35        adjustments: &[Adjustment],
36    ) -> Result<Option<(Place, BasicBlockId)>> {
37        let ty = adjustments
38            .last()
39            .map(|it| it.target.clone())
40            .unwrap_or_else(|| self.expr_ty_without_adjust(expr_id));
41        let place = self.temp(ty, prev_block, expr_id.into())?;
42        let Some(current) =
43            self.lower_expr_to_place_with_adjust(expr_id, place.into(), prev_block, adjustments)?
44        else {
45            return Ok(None);
46        };
47        Ok(Some((place.into(), current)))
48    }

50    pub(super) fn lower_expr_as_place_with_adjust(
51        &mut self,
52        current: BasicBlockId,
53        expr_id: ExprId,
54        upgrade_rvalue: bool,
55        adjustments: &[Adjustment],
56    ) -> Result<Option<(Place, BasicBlockId)>> {
57        let try_rvalue = |this: &mut MirLowerCtx<'_>| {
58            if !upgrade_rvalue {
59                return Err(MirLowerError::MutatingRvalue);
60            }
61            this.lower_expr_to_some_place_with_adjust(expr_id, current, adjustments)
62        };
63        if let Some((last, rest)) = adjustments.split_last() {
64            match last.kind {
65                Adjust::Deref(None) => {
66                    let Some(mut it) = self.lower_expr_as_place_with_adjust(
67                        current,
68                        expr_id,
69                        upgrade_rvalue,
70                        rest,
71                    )?
72                    else {
73                        return Ok(None);
74                    };
75                    it.0 = it.0.project(ProjectionElem::Deref, &mut self.result.projection_store);
76                    Ok(Some(it))
77                }
78                Adjust::Deref(Some(od)) => {
79                    let Some((r, current)) = self.lower_expr_as_place_with_adjust(
80                        current,
81                        expr_id,
82                        upgrade_rvalue,
83                        rest,
84                    )?
85                    else {
86                        return Ok(None);
87                    };
88                    self.lower_overloaded_deref(
89                        current,
90                        r,
91                        rest.last()
92                            .map(|it| it.target.clone())
93                            .unwrap_or_else(|| self.expr_ty_without_adjust(expr_id)),
94                        last.target.clone(),
95                        expr_id.into(),
96                        match od.0 {
97                            Some(Mutability::Mut) => true,
98                            Some(Mutability::Not) => false,
99                            None => {
100                                not_supported!("implicit overloaded deref with unknown mutability")
101                            }
102                        },
103                    )
104                }
105                Adjust::NeverToAny | Adjust::Borrow(_) | Adjust::Pointer(_) => try_rvalue(self),
106            }
107        } else {
108            self.lower_expr_as_place_without_adjust(current, expr_id, upgrade_rvalue)
109        }
110    }

112    pub(super) fn lower_expr_as_place(
113        &mut self,
114        current: BasicBlockId,
115        expr_id: ExprId,
116        upgrade_rvalue: bool,
117    ) -> Result<Option<(Place, BasicBlockId)>> {
118        match self.infer.expr_adjustments.get(&expr_id) {
119            Some(a) => self.lower_expr_as_place_with_adjust(current, expr_id, upgrade_rvalue, a),
120            None => self.lower_expr_as_place_without_adjust(current, expr_id, upgrade_rvalue),
121        }
122    }

124    pub(super) fn lower_expr_as_place_without_adjust(
125        &mut self,
126        current: BasicBlockId,
127        expr_id: ExprId,
128        upgrade_rvalue: bool,
129    ) -> Result<Option<(Place, BasicBlockId)>> {
130        let try_rvalue = |this: &mut MirLowerCtx<'_>| {
131            if !upgrade_rvalue {
132                return Err(MirLowerError::MutatingRvalue);
133            }
134            this.lower_expr_to_some_place_without_adjust(expr_id, current)
135        };
136        match &self.body[expr_id] {
137            Expr::Path(p) => {
138                let resolver_guard =
139                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
140                let hygiene = self.body.expr_path_hygiene(expr_id);
141                let resolved = self.resolver.resolve_path_in_value_ns_fully(self.db, p, hygiene);
142                self.resolver.reset_to_guard(resolver_guard);
143                let Some(pr) = resolved else {
144                    return try_rvalue(self);
145                };
146                match pr {
147                    ValueNs::LocalBinding(pat_id) => {
148                        Ok(Some((self.binding_local(pat_id)?.into(), current)))
149                    }
150                    ValueNs::StaticId(s) => {
151                        let ty = self.expr_ty_without_adjust(expr_id);
152                        let ref_ty =
153                            TyKind::Ref(Mutability::Not, static_lifetime(), ty).intern(Interner);
154                        let temp: Place = self.temp(ref_ty, current, expr_id.into())?.into();
155                        self.push_assignment(
156                            current,
157                            temp,
158                            Operand { kind: OperandKind::Static(s), span: None }.into(),
159                            expr_id.into(),
160                        );
161                        Ok(Some((
162                            temp.project(ProjectionElem::Deref, &mut self.result.projection_store),
163                            current,
164                        )))
165                    }
166                    _ => try_rvalue(self),
167                }
168            }
169            Expr::UnaryOp { expr, op: hir_def::hir::UnaryOp::Deref } => {
170                let is_builtin = match self.expr_ty_without_adjust(*expr).kind(Interner) {
171                    TyKind::Ref(..) | TyKind::Raw(..) => true,
172                    TyKind::Adt(id, _) => {
173                        if let Some(lang_item) = self.db.lang_attr(id.0.into()) {
174                            lang_item == LangItem::OwnedBox
175                        } else {
176                            false
177                        }
178                    }
179                    _ => false,
180                };
181                if !is_builtin {
182                    let Some((p, current)) = self.lower_expr_as_place(current, *expr, true)? else {
183                        return Ok(None);
184                    };
185                    return self.lower_overloaded_deref(
186                        current,
187                        p,
188                        self.expr_ty_after_adjustments(*expr),
189                        self.expr_ty_without_adjust(expr_id),
190                        expr_id.into(),
191                        'b: {
192                            if let Some((f, _)) = self.infer.method_resolution(expr_id)
193                                && let Some(deref_trait) =
194                                    self.resolve_lang_item(LangItem::DerefMut)?.as_trait()
195                                && let Some(deref_fn) = deref_trait
196                                    .trait_items(self.db)
197                                    .method_by_name(&Name::new_symbol_root(sym::deref_mut))
198                            {
199                                break 'b deref_fn == f;
200                            }
201                            false
202                        },
203                    );
204                }
205                let Some((mut r, current)) = self.lower_expr_as_place(current, *expr, true)? else {
206                    return Ok(None);
207                };
208                r = r.project(ProjectionElem::Deref, &mut self.result.projection_store);
209                Ok(Some((r, current)))
210            }
211            Expr::UnaryOp { .. } => try_rvalue(self),
212            Expr::Field { expr, .. } => {
213                let Some((mut r, current)) = self.lower_expr_as_place(current, *expr, true)? else {
214                    return Ok(None);
215                };
216                self.push_field_projection(&mut r, expr_id)?;
217                Ok(Some((r, current)))
218            }
219            Expr::Index { base, index } => {
220                let base_ty = self.expr_ty_after_adjustments(*base);
221                let index_ty = self.expr_ty_after_adjustments(*index);
222                if index_ty != TyBuilder::usize()
223                    || !matches!(
224                        base_ty.strip_reference().kind(Interner),
225                        TyKind::Array(..) | TyKind::Slice(..)
226                    )
227                {
228                    let Some(index_fn) = self.infer.method_resolution(expr_id) else {
229                        return Err(MirLowerError::UnresolvedMethod(
230                            "[overloaded index]".to_owned(),
231                        ));
232                    };
233                    let Some((base_place, current)) =
234                        self.lower_expr_as_place(current, *base, true)?
235                    else {
236                        return Ok(None);
237                    };
238                    let Some((index_operand, current)) =
239                        self.lower_expr_to_some_operand(*index, current)?
240                    else {
241                        return Ok(None);
242                    };
243                    return self.lower_overloaded_index(
244                        current,
245                        base_place,
246                        base_ty,
247                        self.expr_ty_without_adjust(expr_id),
248                        index_operand,
249                        expr_id.into(),
250                        index_fn,
251                    );
252                }
253                let adjusts = self
254                    .infer
255                    .expr_adjustments
256                    .get(base)
257                    .and_then(|it| it.split_last())
258                    .map(|it| it.1)
259                    .unwrap_or(&[]);
260                let Some((mut p_base, current)) =
261                    self.lower_expr_as_place_with_adjust(current, *base, true, adjusts)?
262                else {
263                    return Ok(None);
264                };
265                let l_index =
266                    self.temp(self.expr_ty_after_adjustments(*index), current, expr_id.into())?;
267                let Some(current) = self.lower_expr_to_place(*index, l_index.into(), current)?
268                else {
269                    return Ok(None);
270                };
271                p_base = p_base
272                    .project(ProjectionElem::Index(l_index), &mut self.result.projection_store);
273                Ok(Some((p_base, current)))
274            }
275            _ => try_rvalue(self),
276        }
277    }

279    fn lower_overloaded_index(
280        &mut self,
281        current: BasicBlockId,
282        place: Place,
283        base_ty: Ty,
284        result_ty: Ty,
285        index_operand: Operand,
286        span: MirSpan,
287        index_fn: (FunctionId, Substitution),
288    ) -> Result<Option<(Place, BasicBlockId)>> {
289        let mutability = match base_ty.as_reference() {
290            Some((_, _, mutability)) => mutability,
291            None => Mutability::Not,
292        };
293        let result_ref = TyKind::Ref(mutability, error_lifetime(), result_ty).intern(Interner);
294        let mut result: Place = self.temp(result_ref, current, span)?.into();
295        let index_fn_op = Operand::const_zst(
296            TyKind::FnDef(CallableDefId::FunctionId(index_fn.0).to_chalk(self.db), index_fn.1)
297                .intern(Interner),
298        );
299        let Some(current) = self.lower_call(
300            index_fn_op,
301            Box::new([Operand { kind: OperandKind::Copy(place), span: None }, index_operand]),
302            result,
303            current,
304            false,
305            span,
306        )?
307        else {
308            return Ok(None);
309        };
310        result = result.project(ProjectionElem::Deref, &mut self.result.projection_store);
311        Ok(Some((result, current)))
312    }

314    fn lower_overloaded_deref(
315        &mut self,
316        current: BasicBlockId,
317        place: Place,
318        source_ty: Ty,
319        target_ty: Ty,
320        span: MirSpan,
321        mutability: bool,
322    ) -> Result<Option<(Place, BasicBlockId)>> {
323        let (chalk_mut, trait_lang_item, trait_method_name, borrow_kind) = if !mutability {
324            (
325                Mutability::Not,
326                LangItem::Deref,
327                Name::new_symbol_root(sym::deref),
328                BorrowKind::Shared,
329            )
330        } else {
331            (
332                Mutability::Mut,
333                LangItem::DerefMut,
334                Name::new_symbol_root(sym::deref_mut),
335                BorrowKind::Mut { kind: MutBorrowKind::Default },
336            )
337        };
338        let ty_ref = TyKind::Ref(chalk_mut, error_lifetime(), source_ty.clone()).intern(Interner);
339        let target_ty_ref = TyKind::Ref(chalk_mut, error_lifetime(), target_ty).intern(Interner);
340        let ref_place: Place = self.temp(ty_ref, current, span)?.into();
341        self.push_assignment(current, ref_place, Rvalue::Ref(borrow_kind, place), span);
342        let deref_trait = self
343            .resolve_lang_item(trait_lang_item)?
344            .as_trait()
345            .ok_or(MirLowerError::LangItemNotFound(trait_lang_item))?;
346        let deref_fn = deref_trait
347            .trait_items(self.db)
348            .method_by_name(&trait_method_name)
349            .ok_or(MirLowerError::LangItemNotFound(trait_lang_item))?;
350        let deref_fn_op = Operand::const_zst(
351            TyKind::FnDef(
352                CallableDefId::FunctionId(deref_fn).to_chalk(self.db),
353                Substitution::from1(Interner, source_ty),
354            )
355            .intern(Interner),
356        );
357        let mut result: Place = self.temp(target_ty_ref, current, span)?.into();
358        let Some(current) = self.lower_call(
359            deref_fn_op,
360            Box::new([Operand { kind: OperandKind::Copy(ref_place), span: None }]),
361            result,
362            current,
363            false,
364            span,
365        )?
366        else {
367            return Ok(None);
368        };
369        result = result.project(ProjectionElem::Deref, &mut self.result.projection_store);
370        Ok(Some((result, current)))
371    }
