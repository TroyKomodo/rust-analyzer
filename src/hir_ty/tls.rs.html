1//! Implementation of Chalk debug helper functions using TLS.

15pub(crate) struct DebugContext<'a>(&'a dyn HirDatabase);
16
17impl DebugContext<'_> {
18    pub(crate) fn debug_struct_id(
19        &self,
20        id: chalk_db::AdtId,
21        f: &mut fmt::Formatter<'_>,
22    ) -> Result<(), fmt::Error> {
23        let name = match id.0 {
24            AdtId::StructId(it) => self.0.struct_signature(it).name.clone(),
25            AdtId::UnionId(it) => self.0.union_signature(it).name.clone(),
26            AdtId::EnumId(it) => self.0.enum_signature(it).name.clone(),
27        };
28        name.display(self.0, Edition::LATEST).fmt(f)?;
29        Ok(())
30    }
31
32    pub(crate) fn debug_trait_id(
33        &self,
34        id: chalk_db::TraitId,
35        f: &mut fmt::Formatter<'_>,
36    ) -> Result<(), fmt::Error> {
37        let trait_: hir_def::TraitId = from_chalk_trait_id(id);
38        let trait_data = self.0.trait_signature(trait_);
39        trait_data.name.display(self.0, Edition::LATEST).fmt(f)?;
40        Ok(())
41    }
42
43    pub(crate) fn debug_assoc_type_id(
44        &self,
45        id: chalk_db::AssocTypeId,
46        fmt: &mut fmt::Formatter<'_>,
47    ) -> Result<(), fmt::Error> {
48        let type_alias: TypeAliasId = from_assoc_type_id(id);
49        let type_alias_data = self.0.type_alias_signature(type_alias);
50        let trait_ = match type_alias.lookup(self.0).container {
51            ItemContainerId::TraitId(t) => t,
52            _ => panic!("associated type not in trait"),
53        };
54        let trait_data = self.0.trait_signature(trait_);
55        write!(
56            fmt,
57            "{}::{}",
58            trait_data.name.display(self.0, Edition::LATEST),
59            type_alias_data.name.display(self.0, Edition::LATEST)
60        )?;
61        Ok(())
62    }
63
64    pub(crate) fn debug_projection_ty(
65        &self,
66        projection_ty: &chalk_ir::ProjectionTy<Interner>,
67        fmt: &mut fmt::Formatter<'_>,
68    ) -> Result<(), fmt::Error> {
69        let type_alias = from_assoc_type_id(projection_ty.associated_ty_id);
70        let type_alias_data = self.0.type_alias_signature(type_alias);
71        let trait_ = match type_alias.lookup(self.0).container {
72            ItemContainerId::TraitId(t) => t,
73            _ => panic!("associated type not in trait"),
74        };
75        let trait_name = &self.0.trait_signature(trait_).name;
76        let trait_ref = projection_ty.trait_ref(self.0);
77        let trait_params = trait_ref.substitution.as_slice(Interner);
78        let self_ty = trait_ref.self_type_parameter(Interner);
79        write!(fmt, "<{self_ty:?} as {}", trait_name.display(self.0, Edition::LATEST))?;
80        if trait_params.len() > 1 {
81            write!(
82                fmt,
83                "<{}>",
84                trait_params[1..].iter().format_with(", ", |x, f| f(&format_args!("{x:?}"))),
85            )?;
86        }
87        write!(fmt, ">::{}", type_alias_data.name.display(self.0, Edition::LATEST))?;
88
89        let proj_params = &projection_ty.substitution.as_slice(Interner)[trait_params.len()..];
90        if !proj_params.is_empty() {
91            write!(
92                fmt,
93                "<{}>",
94                proj_params.iter().format_with(", ", |x, f| f(&format_args!("{x:?}"))),
95            )?;
96        }
97
98        Ok(())
99    }
100
101    pub(crate) fn debug_fn_def_id(
102        &self,
103        fn_def_id: chalk_ir::FnDefId<Interner>,
104        fmt: &mut fmt::Formatter<'_>,
105    ) -> Result<(), fmt::Error> {
106        let def: CallableDefId = from_chalk(self.0, fn_def_id);
107        let name = match def {
108            CallableDefId::FunctionId(ff) => self.0.function_signature(ff).name.clone(),
109            CallableDefId::StructId(s) => self.0.struct_signature(s).name.clone(),
110            CallableDefId::EnumVariantId(e) => {
111                let loc = e.lookup(self.0);
112                loc.parent.enum_variants(self.0).variants[loc.index as usize].1.clone()
113            }
114        };
115        match def {
116            CallableDefId::FunctionId(_) => {
117                write!(fmt, "{{fn {}}}", name.display(self.0, Edition::LATEST))
118            }
119            CallableDefId::StructId(_) | CallableDefId::EnumVariantId(_) => {
120                write!(fmt, "{{ctor {}}}", name.display(self.0, Edition::LATEST))
121            }
122        }
123    }
124}

126mod unsafe_tls {
127    use super::DebugContext;
128    use crate::db::HirDatabase;
129    use scoped_tls::scoped_thread_local;
130
131    scoped_thread_local!(static PROGRAM: DebugContext<'_>);
132
133    pub(crate) fn with_current_program<R>(
134        op: impl for<'a> FnOnce(Option<&'a DebugContext<'a>>) -> R,
135    ) -> R {
136        if PROGRAM.is_set() { PROGRAM.with(|prog| op(Some(prog))) } else { op(None) }
137    }
138
139    pub(crate) fn set_current_program<OP, R>(p: &dyn HirDatabase, op: OP) -> R
140    where
141        OP: FnOnce() -> R,
142    {
143        let ctx = DebugContext(p);
144        // we're transmuting the lifetime in the DebugContext to static. This is
145        // fine because we only keep the reference for the lifetime of this
146        // function, *and* the only way to access the context is through
147        // `with_current_program`, which hides the lifetime through the `for`
148        // type.
149        let static_p: &DebugContext<'static> =
150            unsafe { std::mem::transmute::<&DebugContext<'_>, &DebugContext<'static>>(&ctx) };
151        PROGRAM.set(static_p, op)
152    }
153}
