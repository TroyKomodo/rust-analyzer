34fn create_chalk_solver() -> chalk_recursive::RecursiveSolver<Interner> {
35    let overflow_depth =
36        var("CHALK_OVERFLOW_DEPTH").ok().and_then(|s| s.parse().ok()).unwrap_or(500);
37    let max_size = var("CHALK_SOLVER_MAX_SIZE").ok().and_then(|s| s.parse().ok()).unwrap_or(150);
38    chalk_recursive::RecursiveSolver::new(overflow_depth, max_size, Some(Cache::new()))
39}

41/// A set of clauses that we assume to be true. E.g. if we are inside this function:
42/// ```rust
43/// fn foo<T: Default>(t: T) {}
44/// ```
45/// we assume that `T: Default`.
46#[derive(Debug, Clone, PartialEq, Eq, Hash)]
47pub struct TraitEnvironment {
48    pub krate: Crate,
49    pub block: Option<BlockId>,
50    // FIXME make this a BTreeMap
51    traits_from_clauses: Box<[(Ty, TraitId)]>,
52    pub env: chalk_ir::Environment<Interner>,
53}

55impl TraitEnvironment {
56    pub fn empty(krate: Crate) -> Arc<Self> {
57        Arc::new(TraitEnvironment {
58            krate,
59            block: None,
60            traits_from_clauses: Box::default(),
61            env: chalk_ir::Environment::new(Interner),
62        })
63    }
64
65    pub fn new(
66        krate: Crate,
67        block: Option<BlockId>,
68        traits_from_clauses: Box<[(Ty, TraitId)]>,
69        env: chalk_ir::Environment<Interner>,
70    ) -> Arc<Self> {
71        Arc::new(TraitEnvironment { krate, block, traits_from_clauses, env })
72    }
73
74    // pub fn with_block(self: &mut Arc<Self>, block: BlockId) {
75    pub fn with_block(this: &mut Arc<Self>, block: BlockId) {
76        Arc::make_mut(this).block = Some(block);
77    }
78
79    pub fn traits_in_scope_from_clauses(&self, ty: Ty) -> impl Iterator<Item = TraitId> + '_ {
80        self.traits_from_clauses
81            .iter()
82            .filter_map(move |(self_ty, trait_id)| (*self_ty == ty).then_some(*trait_id))
83    }
84}

86pub(crate) fn normalize_projection_query(
87    db: &dyn HirDatabase,
88    projection: ProjectionTy,
89    env: Arc<TraitEnvironment>,
90) -> Ty {
91    if projection.substitution.iter(Interner).any(|arg| {
92        arg.ty(Interner)
93            .is_some_and(|ty| ty.data(Interner).flags.intersects(TypeFlags::HAS_TY_INFER))
94    }) {
95        never!(
96            "Invoking `normalize_projection_query` with a projection type containing inference var"
97        );
98        return TyKind::Error.intern(Interner);
99    }
100
101    let mut table = InferenceTable::new(db, env);
102    let ty = table.normalize_projection_ty(projection);
103    table.resolve_completely(ty)
104}

106/// Solve a trait goal using Chalk.
107pub(crate) fn trait_solve_query(
108    db: &dyn HirDatabase,
109    krate: Crate,
110    block: Option<BlockId>,
111    goal: Canonical<InEnvironment<Goal>>,
112) -> Option<Solution> {
113    let _p = tracing::info_span!("trait_solve_query", detail = ?match &goal.value.goal.data(Interner) {
114        GoalData::DomainGoal(DomainGoal::Holds(WhereClause::Implemented(it))) => db
115            .trait_signature(it.hir_trait_id())
116            .name
117            .display(db, Edition::LATEST)
118            .to_string(),
119        GoalData::DomainGoal(DomainGoal::Holds(WhereClause::AliasEq(_))) => "alias_eq".to_owned(),
120        _ => "??".to_owned(),
121    })
122    .entered();
123
124    if let GoalData::DomainGoal(DomainGoal::Holds(WhereClause::AliasEq(AliasEq {
125        alias: AliasTy::Projection(projection_ty),
126        ..
127    }))) = &goal.value.goal.data(Interner)
128        && let TyKind::BoundVar(_) = projection_ty.self_type_parameter(db).kind(Interner)
129    {
130        // Hack: don't ask Chalk to normalize with an unknown self type, it'll say that's impossible
131        return Some(Solution::Ambig(Guidance::Unknown));
132    }
133
134    // Chalk see `UnevaluatedConst` as a unique concrete value, but we see it as an alias for another const. So
135    // we should get rid of it when talking to chalk.
136    let goal = goal
137        .try_fold_with(&mut UnevaluatedConstEvaluatorFolder { db }, DebruijnIndex::INNERMOST)
138        .unwrap();
139
140    // We currently don't deal with universes (I think / hope they're not yet
141    // relevant for our use cases?)
142    let u_canonical = chalk_ir::UCanonical { canonical: goal, universes: 1 };
143    solve(db, krate, block, &u_canonical)
144}

146fn solve(
147    db: &dyn HirDatabase,
148    krate: Crate,
149    block: Option<BlockId>,
150    goal: &chalk_ir::UCanonical<chalk_ir::InEnvironment<chalk_ir::Goal<Interner>>>,
151) -> Option<chalk_solve::Solution<Interner>> {
152    let _p = tracing::info_span!("solve", ?krate, ?block).entered();
153    let context = ChalkContext { db, krate, block };
154    tracing::debug!("solve goal: {:?}", goal);
155    let mut solver = create_chalk_solver();
156
157    let fuel = std::cell::Cell::new(CHALK_SOLVER_FUEL);
158
159    let should_continue = || {
160        db.unwind_if_revision_cancelled();
161        let remaining = fuel.get();
162        fuel.set(remaining - 1);
163        if remaining == 0 {
164            tracing::debug!("fuel exhausted");
165        }
166        remaining > 0
167    };
168
169    let mut solve = || {
170        let _ctx = if is_chalk_debug() || is_chalk_print() {
171            Some(panic_context::enter(format!("solving {goal:?}")))
172        } else {
173            None
174        };
175        let solution = if is_chalk_print() {
176            let logging_db =
177                LoggingRustIrDatabaseLoggingOnDrop(LoggingRustIrDatabase::new(context));
178            solver.solve_limited(&logging_db.0, goal, &should_continue)
179        } else {
180            solver.solve_limited(&context, goal, &should_continue)
181        };
182
183        tracing::debug!("solve({:?}) => {:?}", goal, solution);
184
185        solution
186    };
187
188    // don't set the TLS for Chalk unless Chalk debugging is active, to make
189    // extra sure we only use it for debugging
190    if is_chalk_debug() { crate::tls::set_current_program(db, solve) } else { solve() }
191}

193struct LoggingRustIrDatabaseLoggingOnDrop<'a>(LoggingRustIrDatabase<Interner, ChalkContext<'a>>);

195impl Drop for LoggingRustIrDatabaseLoggingOnDrop<'_> {
196    fn drop(&mut self) {
197        tracing::info!("chalk program:\n{}", self.0);

209#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
210pub enum FnTrait {
211    // Warning: Order is important. If something implements `x` it should also implement
212    // `y` if `y <= x`.
213    FnOnce,
214    FnMut,
215    Fn,
216
217    AsyncFnOnce,
218    AsyncFnMut,
219    AsyncFn,
220}
221
222impl fmt::Display for FnTrait {
223    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
224        match self {
225            FnTrait::FnOnce => write!(f, "FnOnce"),
226            FnTrait::FnMut => write!(f, "FnMut"),
227            FnTrait::Fn => write!(f, "Fn"),
228            FnTrait::AsyncFnOnce => write!(f, "AsyncFnOnce"),
229            FnTrait::AsyncFnMut => write!(f, "AsyncFnMut"),
230            FnTrait::AsyncFn => write!(f, "AsyncFn"),
231        }
232    }
233}
234
235impl FnTrait {
236    pub const fn function_name(&self) -> &'static str {
237        match self {
238            FnTrait::FnOnce => "call_once",
239            FnTrait::FnMut => "call_mut",
240            FnTrait::Fn => "call",
241            FnTrait::AsyncFnOnce => "async_call_once",
242            FnTrait::AsyncFnMut => "async_call_mut",
243            FnTrait::AsyncFn => "async_call",
244        }
245    }
246
247    const fn lang_item(self) -> LangItem {
248        match self {
249            FnTrait::FnOnce => LangItem::FnOnce,
250            FnTrait::FnMut => LangItem::FnMut,
251            FnTrait::Fn => LangItem::Fn,
252            FnTrait::AsyncFnOnce => LangItem::AsyncFnOnce,
253            FnTrait::AsyncFnMut => LangItem::AsyncFnMut,
254            FnTrait::AsyncFn => LangItem::AsyncFn,
255        }
256    }
257
258    pub const fn from_lang_item(lang_item: LangItem) -> Option<Self> {
259        match lang_item {
260            LangItem::FnOnce => Some(FnTrait::FnOnce),
261            LangItem::FnMut => Some(FnTrait::FnMut),
262            LangItem::Fn => Some(FnTrait::Fn),
263            LangItem::AsyncFnOnce => Some(FnTrait::AsyncFnOnce),
264            LangItem::AsyncFnMut => Some(FnTrait::AsyncFnMut),
265            LangItem::AsyncFn => Some(FnTrait::AsyncFn),
266            _ => None,
267        }
268    }
269
270    pub const fn to_chalk_ir(self) -> rust_ir::ClosureKind {
271        // Chalk doesn't support async fn traits.
272        match self {
273            FnTrait::AsyncFnOnce | FnTrait::FnOnce => rust_ir::ClosureKind::FnOnce,
274            FnTrait::AsyncFnMut | FnTrait::FnMut => rust_ir::ClosureKind::FnMut,
275            FnTrait::AsyncFn | FnTrait::Fn => rust_ir::ClosureKind::Fn,
276        }
277    }
278
279    pub fn method_name(self) -> Name {
280        match self {
281            FnTrait::FnOnce => Name::new_symbol_root(sym::call_once),
282            FnTrait::FnMut => Name::new_symbol_root(sym::call_mut),
283            FnTrait::Fn => Name::new_symbol_root(sym::call),
284            FnTrait::AsyncFnOnce => Name::new_symbol_root(sym::async_call_once),
285            FnTrait::AsyncFnMut => Name::new_symbol_root(sym::async_call_mut),
286            FnTrait::AsyncFn => Name::new_symbol_root(sym::async_call),
287        }
288    }
289
290    pub fn get_id(self, db: &dyn HirDatabase, krate: Crate) -> Option<TraitId> {
291        self.lang_item().resolve_trait(db, krate)
292    }
293
294    #[inline]
295    pub(crate) fn is_async(self) -> bool {
296        matches!(self, FnTrait::AsyncFn | FnTrait::AsyncFnMut | FnTrait::AsyncFnOnce)
297    }
