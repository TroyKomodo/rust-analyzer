24/// `NavigationTarget` represents an element in the editor's UI which you can
25/// click on to navigate to a particular piece of code.
26///
27/// Typically, a `NavigationTarget` corresponds to some element in the source
28/// code, like a function or a struct, but this is not strictly required.
29#[derive(Clone, PartialEq, Eq, Hash)]
30pub struct NavigationTarget {
31    pub file_id: FileId,
32    /// Range which encompasses the whole element.
33    ///
34    /// Should include body, doc comments, attributes, etc.
35    ///
36    /// Clients should use this range to answer "is the cursor inside the
37    /// element?" question.
38    pub full_range: TextRange,
39    /// A "most interesting" range within the `full_range`.
40    ///
41    /// Typically, `full_range` is the whole syntax node, including doc
42    /// comments, and `focus_range` is the range of the identifier.
43    ///
44    /// Clients should place the cursor on this range when navigating to this target.
45    ///
46    /// This range must be contained within [`Self::full_range`].
47    pub focus_range: Option<TextRange>,
48    // FIXME: Symbol
49    pub name: SmolStr,
50    pub kind: Option<SymbolKind>,
51    // FIXME: Symbol
52    pub container_name: Option<SmolStr>,
53    pub description: Option<String>,
54    pub docs: Option<Documentation>,
55    /// In addition to a `name` field, a `NavigationTarget` may also be aliased
56    /// In such cases we want a `NavigationTarget` to be accessible by its alias
57    // FIXME: Symbol
58    pub alias: Option<SmolStr>,
59}

61impl fmt::Debug for NavigationTarget {
62    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
63        let mut f = f.debug_struct("NavigationTarget");
64        macro_rules! opt {
65            ($($name:ident)*) => {$(
66                if let Some(it) = &self.$name {
67                    f.field(stringify!($name), it);
68                }
69            )*}
70        }
71        f.field("file_id", &self.file_id).field("full_range", &self.full_range);
72        opt!(focus_range);
73        f.field("name", &self.name);
74        opt!(kind container_name description docs);
75        f.finish()
76    }

101    pub(crate) fn from_module_to_decl(
102        db: &RootDatabase,
103        module: hir::Module,
104    ) -> UpmappingResult<NavigationTarget> {
105        let edition = module.krate().edition(db);
106        let name =
107            module.name(db).map(|it| it.display_no_db(edition).to_smolstr()).unwrap_or_default();
108        match module.declaration_source(db) {
109            Some(InFile { value, file_id }) => {
110                orig_range_with_focus(db, file_id, value.syntax(), value.name()).map(
111                    |(FileRange { file_id, range: full_range }, focus_range)| {
112                        let mut res = NavigationTarget::from_syntax(
113                            file_id,
114                            name.clone(),
115                            focus_range,
116                            full_range,
117                            SymbolKind::Module,
118                        );
119                        res.docs = module.docs(db);
120                        res.description = Some(
121                            module.display(db, module.krate().to_display_target(db)).to_string(),
122                        );
123                        res
124                    },
125                )
126            }
127            _ => module.to_nav(db),
128        }
129    }

185impl TryToNav for FileSymbol {
186    fn try_to_nav(&self, db: &RootDatabase) -> Option<UpmappingResult<NavigationTarget>> {
187        let edition =
188            self.def.module(db).map(|it| it.krate().edition(db)).unwrap_or(Edition::CURRENT);
189        let display_target = self.def.krate(db).to_display_target(db);
190        Some(
191            orig_range_with_focus_r(
192                db,
193                self.loc.hir_file_id,
194                self.loc.ptr.text_range(),
195                Some(self.loc.name_ptr.text_range()),
196            )
197            .map(|(FileRange { file_id, range: full_range }, focus_range)| {
198                NavigationTarget {
199                    file_id,
200                    name: self.is_alias.then(|| self.def.name(db)).flatten().map_or_else(
201                        || self.name.as_str().into(),
202                        |it| it.display_no_db(edition).to_smolstr(),
203                    ),
204                    alias: self.is_alias.then(|| self.name.as_str().into()),
205                    kind: Some(self.def.into()),
206                    full_range,
207                    focus_range,
208                    container_name: self.container_name.clone(),
209                    description: match self.def {
210                        hir::ModuleDef::Module(it) => {
211                            Some(it.display(db, display_target).to_string())
212                        }
213                        hir::ModuleDef::Function(it) => {
214                            Some(it.display(db, display_target).to_string())
215                        }
216                        hir::ModuleDef::Adt(it) => Some(it.display(db, display_target).to_string()),
217                        hir::ModuleDef::Variant(it) => {

224                            Some(it.display(db, display_target).to_string())

227                            Some(it.display(db, display_target).to_string())
228                        }

259            Definition::Adt(it) => it.try_to_nav(db),

261            Definition::Const(it) => it.try_to_nav(db),
262            Definition::Static(it) => it.try_to_nav(db),
263            Definition::Trait(it) => it.try_to_nav(db),
264            Definition::TraitAlias(it) => it.try_to_nav(db),

400impl ToNav for hir::Module {
401    fn to_nav(&self, db: &RootDatabase) -> UpmappingResult<NavigationTarget> {
402        let InFile { file_id, value } = self.definition_source(db);
403        let edition = self.krate(db).edition(db);
404
405        let name =
406            self.name(db).map(|it| it.display_no_db(edition).to_smolstr()).unwrap_or_default();
407        let (syntax, focus) = match &value {
408            ModuleSource::SourceFile(node) => (node.syntax(), None),
409            ModuleSource::Module(node) => (node.syntax(), node.name()),
410            ModuleSource::BlockExpr(node) => (node.syntax(), None),
411        };
412
413        orig_range_with_focus(db, file_id, syntax, focus).map(
414            |(FileRange { file_id, range: full_range }, focus_range)| {
415                NavigationTarget::from_syntax(
416                    file_id,
417                    name.clone(),
418                    focus_range,
419                    full_range,
420                    SymbolKind::Module,
421                )
422            },
423        )
424    }
425}

457impl TryToNav for hir::ExternCrateDecl {
458    fn try_to_nav(&self, db: &RootDatabase) -> Option<UpmappingResult<NavigationTarget>> {
459        let src = self.source(db)?;
460        let InFile { file_id, value } = src;
461        let focus = value
462            .rename()
463            .map_or_else(|| value.name_ref().map(Either::Left), |it| it.name().map(Either::Right));
464        let krate = self.module(db).krate();
465        let edition = krate.edition(db);
466
467        Some(orig_range_with_focus(db, file_id, value.syntax(), focus).map(
468            |(FileRange { file_id, range: full_range }, focus_range)| {
469                let mut res = NavigationTarget::from_syntax(
470                    file_id,
471                    self.alias_or_name(db)
472                        .unwrap_or_else(|| self.name(db))
473                        .display_no_db(edition)
474                        .to_smolstr(),
475                    focus_range,
476                    full_range,
477                    SymbolKind::Module,
478                );
479
480                res.docs = self.docs(db);
481                res.description = Some(self.display(db, krate.to_display_target(db)).to_string());
482                res.container_name = container_name(db, *self, edition);
483                res
484            },
485        ))
486    }

489impl TryToNav for hir::Field {
490    fn try_to_nav(&self, db: &RootDatabase) -> Option<UpmappingResult<NavigationTarget>> {
491        let src = self.source(db)?;
492        let krate = self.parent_def(db).module(db).krate();
493
494        let field_source = match &src.value {
495            FieldSource::Named(it) => {
496                NavigationTarget::from_named(db, src.with_value(it), SymbolKind::Field).map(
497                    |mut res| {
498                        res.docs = self.docs(db);
499                        res.description =
500                            Some(self.display(db, krate.to_display_target(db)).to_string());
501                        res
502                    },
503                )
504            }
505            FieldSource::Pos(it) => orig_range(db, src.file_id, it.syntax()).map(
506                |(FileRange { file_id, range: full_range }, focus_range)| {
507                    NavigationTarget::from_syntax(
508                        file_id,
509                        format_smolstr!("{}", self.index()),
510                        focus_range,
511                        full_range,
512                        SymbolKind::Field,
513                    )
514                },
515            ),
516        };
517        Some(field_source)
518    }

572impl ToNav for LocalSource {
573    fn to_nav(&self, db: &RootDatabase) -> UpmappingResult<NavigationTarget> {
574        let InFile { file_id, value } = &self.source;
575        let file_id = *file_id;
576        let local = self.local;
577        let (node, name) = match &value {
578            Either::Left(bind_pat) => (bind_pat.syntax(), bind_pat.name()),
579            Either::Right(it) => (it.syntax(), it.name()),
580        };
581        let edition = self.local.parent(db).module(db).krate().edition(db);
582
583        orig_range_with_focus(db, file_id, node, name).map(
584            |(FileRange { file_id, range: full_range }, focus_range)| {
585                let name = local.name(db).display_no_db(edition).to_smolstr();
586                let kind = if local.is_self(db) {
587                    SymbolKind::SelfParam
588                } else if local.is_param(db) {
589                    SymbolKind::ValueParam
590                } else {
591                    SymbolKind::Local
592                };
593                NavigationTarget {
594                    file_id,
595                    name,
596                    alias: None,
597                    kind: Some(kind),
598                    full_range,
599                    focus_range,
600                    container_name: None,
601                    description: None,
602                    docs: None,
603                }
604            },
605        )
606    }
607}

637impl TryToNav for hir::TypeParam {
638    fn try_to_nav(&self, db: &RootDatabase) -> Option<UpmappingResult<NavigationTarget>> {
639        let InFile { file_id, value } = self.merge().source(db)?;
640        let edition = self.module(db).krate().edition(db);
641        let name = self.name(db).display_no_db(edition).to_smolstr();
642
643        let value = match value {
644            Either::Left(ast::TypeOrConstParam::Type(x)) => Either::Left(x),
645            Either::Left(ast::TypeOrConstParam::Const(_)) => {
646                never!();
647                return None;
648            }
649            Either::Right(x) => Either::Right(x),
650        };
651
652        let syntax = match &value {
653            Either::Left(type_param) => type_param.syntax(),
654            Either::Right(trait_) => trait_.syntax(),
655        };
656        let focus = value.as_ref().either(|it| it.name(), |it| it.name());
657
658        Some(orig_range_with_focus(db, file_id, syntax, focus).map(
659            |(FileRange { file_id, range: full_range }, focus_range)| NavigationTarget {
660                file_id,
661                name: name.clone(),
662                alias: None,
663                kind: Some(SymbolKind::TypeParam),
664                full_range,
665                focus_range,
666                container_name: None,
667                description: None,
668                docs: None,
669            },
670        ))
671    }
672}

757#[derive(Debug)]
758pub struct UpmappingResult<T> {
759    /// The macro call site.
760    pub call_site: T,
761    /// The macro definition site, if relevant.
762    pub def_site: Option<T>,
763}
764
765impl<T> UpmappingResult<T> {
766    pub fn call_site(self) -> T {
767        self.call_site
768    }
769
770    pub fn collect<FI: FromIterator<T>>(self) -> FI {
771        FI::from_iter(self)
772    }
773}
774
775impl<T> IntoIterator for UpmappingResult<T> {
776    type Item = T;
777
778    type IntoIter = <ArrayVec<T, 2> as IntoIterator>::IntoIter;
779
780    fn into_iter(self) -> Self::IntoIter {
781        self.def_site
782            .into_iter()
783            .chain(Some(self.call_site))
784            .collect::<ArrayVec<_, 2>>()
785            .into_iter()
786    }
787}
788
789impl<T> UpmappingResult<T> {
790    pub(crate) fn map<U>(self, f: impl Fn(T) -> U) -> UpmappingResult<U> {
791        UpmappingResult { call_site: f(self.call_site), def_site: self.def_site.map(f) }
792    }
793}

795/// Returns the original range of the syntax node, and the range of the name mapped out of macro expansions
796/// May return two results if the mapped node originates from a macro definition in which case the
797/// second result is the creating macro call.
798fn orig_range_with_focus(
799    db: &RootDatabase,
800    hir_file: HirFileId,
801    value: &SyntaxNode,
802    name: Option<impl AstNode>,
803) -> UpmappingResult<(FileRange, Option<TextRange>)> {
804    orig_range_with_focus_r(
805        db,
806        hir_file,
807        value.text_range(),
808        name.map(|it| it.syntax().text_range()),
809    )
810}
811
812pub(crate) fn orig_range_with_focus_r(
813    db: &RootDatabase,
814    hir_file: HirFileId,
815    value: TextRange,
816    focus_range: Option<TextRange>,
817) -> UpmappingResult<(FileRange, Option<TextRange>)> {
818    let Some(name) = focus_range else { return orig_range_r(db, hir_file, value) };
819
820    let call_kind = || db.lookup_intern_macro_call(hir_file.macro_file().unwrap()).kind;
821
822    let def_range =
823        || db.lookup_intern_macro_call(hir_file.macro_file().unwrap()).def.definition_range(db);
824
825    // FIXME: Also make use of the syntax context to determine which site we are at?
826    let value_range = InFile::new(hir_file, value).original_node_file_range_opt(db);
827    let ((call_site_range, call_site_focus), def_site) =
828        match InFile::new(hir_file, name).original_node_file_range_opt(db) {
829            // call site name
830            Some((focus_range, ctxt)) if ctxt.is_root() => {
831                // Try to upmap the node as well, if it ends up in the def site, go back to the call site
832                (
833                    (
834                        match value_range {
835                            // name is in the node in the macro input so we can return it
836                            Some((range, ctxt))
837                                if ctxt.is_root()
838                                    && range.file_id == focus_range.file_id
839                                    && range.range.contains_range(focus_range.range) =>
840                            {
841                                range
842                            }
843                            // name lies outside the node, so instead point to the macro call which
844                            // *should* contain the name
845                            _ => {
846                                let kind = call_kind();
847                                let range = kind.clone().original_call_range_with_input(db);
848                                //If the focus range is in the attribute/derive body, we
849                                // need to point the call site to the entire body, if not, fall back
850                                // to the name range of the attribute/derive call
851                                // FIXME: Do this differently, this is very inflexible the caller
852                                // should choose this behavior
853                                if range.file_id == focus_range.file_id
854                                    && range.range.contains_range(focus_range.range)
855                                {
856                                    range
857                                } else {
858                                    kind.original_call_range(db)
859                                }
860                            }
861                        },
862                        Some(focus_range),
863                    ),
864                    // no def site relevant
865                    None,
866                )
867            }
868
869            // def site name
870            // FIXME: This can be improved
871            Some((focus_range, _ctxt)) => {
872                match value_range {
873                    // but overall node is in macro input
874                    Some((range, ctxt)) if ctxt.is_root() => (
875                        // node mapped up in call site, show the node
876                        (range, None),
877                        // def site, if the name is in the (possibly) upmapped def site range, show the
878                        // def site
879                        {
880                            let (def_site, _) = def_range().original_node_file_range(db);
881                            (def_site.file_id == focus_range.file_id
882                                && def_site.range.contains_range(focus_range.range))
883                            .then_some((def_site, Some(focus_range)))
884                        },
885                    ),
886                    // node is in macro def, just show the focus
887                    _ => (
888                        // show the macro call
889                        (call_kind().original_call_range(db), None),
890                        Some((focus_range, Some(focus_range))),
891                    ),
892                }
893            }
894            // lost name? can't happen for single tokens
895            None => return orig_range_r(db, hir_file, value),
896        };
897
898    UpmappingResult {
899        call_site: (
900            call_site_range.into_file_id(db),
901            call_site_focus.and_then(|hir::FileRange { file_id, range }| {
902                if call_site_range.file_id == file_id && call_site_range.range.contains_range(range)
903                {
904                    Some(range)
905                } else {
906                    None
907                }
908            }),
909        ),
910        def_site: def_site.map(|(def_site_range, def_site_focus)| {
911            (
912                def_site_range.into_file_id(db),
913                def_site_focus.and_then(|hir::FileRange { file_id, range }| {
914                    if def_site_range.file_id == file_id
915                        && def_site_range.range.contains_range(range)
916                    {
917                        Some(range)
918                    } else {
919                        None
920                    }
921                }),
922            )
923        }),
924    }
925}

929    hir_file: HirFileId,
930    value: &SyntaxNode,
