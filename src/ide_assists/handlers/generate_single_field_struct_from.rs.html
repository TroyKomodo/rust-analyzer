17// Assist: generate_single_field_struct_from
18//
19// Implement From for a single field structure, ignore trivial types.
20//
21// ```
22// # //- minicore: from, phantom_data
23// use core::marker::PhantomData;
24// struct $0Foo<T> {
25//     id: i32,
26//     _phantom_data: PhantomData<T>,
27// }
28// ```
29// ->
30// ```
31// use core::marker::PhantomData;
32// struct Foo<T> {
33//     id: i32,
34//     _phantom_data: PhantomData<T>,
35// }
36//
37// impl<T> From<i32> for Foo<T> {
38//     fn from(id: i32) -> Self {
39//         Self { id, _phantom_data: PhantomData }
40//     }
41// }
42// ```
43pub(crate) fn generate_single_field_struct_from(
44    acc: &mut Assists,
45    ctx: &AssistContext<'_>,
46) -> Option<()> {
47    let strukt_name = ctx.find_node_at_offset::<ast::Name>()?;
48    let adt = ast::Adt::cast(strukt_name.syntax().parent()?)?;
49    let ast::Adt::Struct(strukt) = adt else {
50        return None;
51    };
52
53    let sema = &ctx.sema;
54    let (names, types) = get_fields(&strukt)?;
55
56    let module = sema.scope(strukt.syntax())?.module();
57    let constructors = make_constructors(ctx, module, &types);
58
59    if constructors.iter().filter(|expr| expr.is_none()).count() != 1 {
60        return None;
61    }
62    let main_field_i = constructors.iter().position(Option::is_none)?;
63    if from_impl_exists(&strukt, main_field_i, &ctx.sema).is_some() {
64        return None;
65    }
66
67    let main_field_name =
68        names.as_ref().map_or(TokenText::borrowed("value"), |names| names[main_field_i].text());
69    let main_field_ty = types[main_field_i].clone();
70
71    acc.add(
72        AssistId::generate("generate_single_field_struct_from"),
73        "Generate single field `From`",
74        strukt.syntax().text_range(),
75        |builder| {
76            let indent = strukt.indent_level();
77            let ty_where_clause = strukt.where_clause();
78            let type_gen_params = strukt.generic_param_list();
79            let type_gen_args = type_gen_params.as_ref().map(|params| params.to_generic_args());
80            let trait_gen_args = Some(make::generic_arg_list([ast::GenericArg::TypeArg(
81                make::type_arg(main_field_ty.clone()),
82            )]));
83
84            let ty = make::ty(&strukt_name.text());
85
86            let constructor =
87                make_adt_constructor(names.as_deref(), constructors, &main_field_name);
88            let body = make::block_expr([], Some(constructor));
89
90            let fn_ = make::fn_(
91                None,
92                None,
93                make::name("from"),
94                None,
95                None,
96                make::param_list(
97                    None,
98                    [make::param(
99                        make::path_pat(make::path_from_text(&main_field_name)),
100                        main_field_ty,
101                    )],
102                ),
103                body,
104                Some(make::ret_type(make::ty("Self"))),
105                false,
106                false,
107                false,
108                false,
109            )
110            .clone_for_update();
111
112            fn_.indent(1.into());
113            let cfg_attrs = strukt
114                .attrs()
115                .filter(|attr| attr.as_simple_call().is_some_and(|(name, _arg)| name == "cfg"));
116
117            let impl_ = make::impl_trait(
118                cfg_attrs,
119                false,
120                None,
121                trait_gen_args,
122                type_gen_params,
123                type_gen_args,
124                false,
125                make::ty("From"),
126                ty.clone(),
127                None,
128                ty_where_clause.map(|wc| edit::AstNodeEdit::reset_indent(&wc)),
129                None,
130            )
131            .clone_for_update();
132
133            impl_.get_or_create_assoc_item_list().add_item(fn_.into());
134
135            impl_.reindent_to(indent);
136
137            builder.insert(strukt.syntax().text_range().end(), format!("\n\n{indent}{impl_}"));
138        },
139    )
140}

142fn make_adt_constructor(
143    names: Option<&[ast::Name]>,
144    constructors: Vec<Option<ast::Expr>>,
145    main_field_name: &TokenText<'_>,
146) -> ast::Expr {
147    if let Some(names) = names {
148        let fields = make::record_expr_field_list(names.iter().zip(constructors).map(
149            |(name, initializer)| {
150                make::record_expr_field(make::name_ref(&name.text()), initializer)
151            },
152        ));
153        make::record_expr(make::path_from_text("Self"), fields).into()
154    } else {
155        let arg_list = make::arg_list(constructors.into_iter().map(|expr| {
156            expr.unwrap_or_else(|| make::expr_path(make::path_from_text(main_field_name)))
157        }));
158        make::expr_call(make::expr_path(make::path_from_text("Self")), arg_list).into()
159    }
160}

162fn make_constructors(
163    ctx: &AssistContext<'_>,
164    module: hir::Module,
165    types: &[ast::Type],
166) -> Vec<Option<ast::Expr>> {
167    let (db, sema) = (ctx.db(), &ctx.sema);
168    types
169        .iter()
170        .map(|ty| {
171            let ty = sema.resolve_type(ty)?;
172            if ty.is_unit() {
173                return Some(make::expr_tuple([]).into());
174            }
175            let item_in_ns = ModuleDef::Adt(ty.as_adt()?).into();
176            let edition = module.krate().edition(db);
177
178            let ty_path = module.find_path(
179                db,
180                item_for_path_search(db, item_in_ns)?,
181                ctx.config.import_path_config(),
182            )?;
183
184            use_trivial_constructor(db, mod_path_to_ast(&ty_path, edition), &ty, edition)
185        })
186        .collect()
187}

189fn get_fields(strukt: &ast::Struct) -> Option<(Option<Vec<ast::Name>>, Vec<ast::Type>)> {
190    Some(match strukt.kind() {
191        ast::StructKind::Unit => return None,
192        ast::StructKind::Record(fields) => {
193            let names = fields.fields().map(|field| field.name()).collect::<Option<_>>()?;
194            let types = fields.fields().map(|field| field.ty()).collect::<Option<_>>()?;
195            (Some(names), types)
196        }
197        ast::StructKind::Tuple(fields) => {
198            (None, fields.fields().map(|field| field.ty()).collect::<Option<_>>()?)
199        }
200    })
201}

203fn from_impl_exists(
204    strukt: &ast::Struct,
205    main_field_i: usize,
206    sema: &Semantics<'_, RootDatabase>,
207) -> Option<()> {
208    let db = sema.db;
209    let strukt = sema.to_def(strukt)?;
210    let krate = strukt.krate(db);
211    let from_trait = FamousDefs(sema, krate).core_convert_From()?;
212    let ty = strukt.fields(db).get(main_field_i)?.ty(db);
213
214    strukt.ty(db).impls_trait(db, from_trait, &[ty]).then_some(())
215}
