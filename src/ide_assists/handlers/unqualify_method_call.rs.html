8// Assist: unqualify_method_call
9//
10// Transforms universal function call syntax into a method call.
11//
12// ```
13// fn main() {
14//     std::ops::Add::add$0(1, 2);
15// }
16// # mod std { pub mod ops { pub trait Add { fn add(self, _: Self) {} } impl Add for i32 {} } }
17// ```
18// ->
19// ```
20// use std::ops::Add;
21//
22// fn main() {
23//     1.add(2);
24// }
25// # mod std { pub mod ops { pub trait Add { fn add(self, _: Self) {} } impl Add for i32 {} } }
26// ```
27pub(crate) fn unqualify_method_call(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
28    let call = ctx.find_node_at_offset::<ast::CallExpr>()?;
29    let ast::Expr::PathExpr(path_expr) = call.expr()? else { return None };
30    let path = path_expr.path()?;
31
32    let cursor_in_range = path.syntax().text_range().contains_range(ctx.selection_trimmed());
33    if !cursor_in_range {
34        return None;
35    }
36
37    let args = call.arg_list()?;
38    let l_paren = args.l_paren_token()?;
39    let mut args_iter = args.args();
40    let first_arg = args_iter.next()?;
41    let second_arg = args_iter.next();
42
43    let qualifier = path.qualifier()?;
44    let method_name = path.segment()?.name_ref()?;
45
46    let res = ctx.sema.resolve_path(&path)?;
47    let hir::PathResolution::Def(hir::ModuleDef::Function(fun)) = res else { return None };
48    if !fun.has_self_param(ctx.sema.db) {
49        return None;
50    }
51
52    // `core::ops::Add::add(` -> ``
53    let delete_path =
54        TextRange::new(path.syntax().text_range().start(), l_paren.text_range().end());
55
56    // Parens around `expr` if needed
57    let parens = first_arg.precedence().needs_parentheses_in(ExprPrecedence::Postfix).then(|| {
58        let range = first_arg.syntax().text_range();
59        (range.start(), range.end())
60    });
61
62    // `, ` -> `.add(`
63    let replace_comma = TextRange::new(
64        first_arg.syntax().text_range().end(),
65        second_arg
66            .map(|a| a.syntax().text_range().start())
67            .unwrap_or_else(|| first_arg.syntax().text_range().end()),
68    );
69
70    acc.add(
71        AssistId::refactor_rewrite("unqualify_method_call"),
72        "Unqualify method call",
73        call.syntax().text_range(),
74        |edit| {
75            edit.delete(delete_path);
76            if let Some((open, close)) = parens {
77                edit.insert(open, "(");
78                edit.insert(close, ")");
79            }
80            edit.replace(replace_comma, format!(".{method_name}("));
81            add_import(qualifier, ctx, edit);
82        },
83    )
84}
85
86fn add_import(
87    qualifier: ast::Path,
88    ctx: &AssistContext<'_>,
89    edit: &mut ide_db::source_change::SourceChangeBuilder,
90) {
91    if let Some(path_segment) = qualifier.segment() {
92        // for `<i32 as std::ops::Add>`
93        let path_type = path_segment.qualifying_trait();
94        let import = match path_type {
95            Some(it) => {
96                if let Some(path) = it.path() {
97                    path
98                } else {
99                    return;
100                }
101            }
102            None => qualifier,
103        };
104
105        // in case for `<_>`
106        if import.coloncolon_token().is_none() {
107            return;
108        }
109
110        let scope = ide_db::imports::insert_use::ImportScope::find_insert_use_container(
111            import.syntax(),
112            &ctx.sema,
113        );
114
115        if let Some(scope) = scope {
116            let scope = edit.make_import_scope_mut(scope);
117            ide_db::imports::insert_use::insert_use(&scope, import, &ctx.config.insert_use);
118        }
119    }
120}
