29pub(crate) fn complete_postfix(
30    acc: &mut Completions,
31    ctx: &CompletionContext<'_>,
32    dot_access: &DotAccess<'_>,
33) {
34    if !ctx.config.enable_postfix_completions {
35        return;
36    }
37
38    let (dot_receiver, receiver_ty, receiver_is_ambiguous_float_literal) = match dot_access {
39        DotAccess { receiver_ty: Some(ty), receiver: Some(it), kind, .. } => (
40            it,
41            &ty.original,
42            match *kind {
43                DotAccessKind::Field { receiver_is_ambiguous_float_literal } => {
44                    receiver_is_ambiguous_float_literal
45                }
46                DotAccessKind::Method { .. } => false,
47            },
48        ),
49        _ => return,
50    };
51    let expr_ctx = &dot_access.ctx;
52
53    let receiver_text =
54        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
55
56    let cap = match ctx.config.snippet_cap {
57        Some(it) => it,
58        None => return,
59    };
60
61    let postfix_snippet = match build_postfix_snippet_builder(ctx, cap, dot_receiver) {
62        Some(it) => it,
63        None => return,
64    };
65
66    let cfg = ctx.config.import_path_config(ctx.is_nightly);
67
68    if let Some(drop_trait) = ctx.famous_defs().core_ops_Drop()
69        && receiver_ty.impls_trait(ctx.db, drop_trait, &[])
70        && let Some(drop_fn) = ctx.famous_defs().core_mem_drop()
71        && let Some(path) = ctx.module.find_path(ctx.db, ItemInNs::Values(drop_fn.into()), cfg)
72    {
73        cov_mark::hit!(postfix_drop_completion);
74        let mut item = postfix_snippet(
75            "drop",
76            "fn drop(&mut self)",
77            &format!("{path}($0{receiver_text})", path = path.display(ctx.db, ctx.edition)),
78        );
79        item.set_documentation(drop_fn.docs(ctx.db));
80        item.add_to(acc, ctx.db);
81    }
82
83    postfix_snippet("ref", "&expr", &format!("&{receiver_text}")).add_to(acc, ctx.db);
84    postfix_snippet("refm", "&mut expr", &format!("&mut {receiver_text}")).add_to(acc, ctx.db);
85    postfix_snippet("deref", "*expr", &format!("*{receiver_text}")).add_to(acc, ctx.db);
86
87    // The rest of the postfix completions create an expression that moves an argument,
88    // so it's better to consider references now to avoid breaking the compilation
89
90    let (dot_receiver_including_refs, prefix) = include_references(dot_receiver);
91    let mut receiver_text =
92        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
93    receiver_text.insert_str(0, &prefix);
94    let postfix_snippet =
95        match build_postfix_snippet_builder(ctx, cap, &dot_receiver_including_refs) {
96            Some(it) => it,
97            None => return,
98        };
99
100    if !ctx.config.snippets.is_empty() {
101        add_custom_postfix_completions(acc, ctx, &postfix_snippet, &receiver_text);
102    }
103
104    postfix_snippet("box", "Box::new(expr)", &format!("Box::new({receiver_text})"))
105        .add_to(acc, ctx.db);
106    postfix_snippet("dbg", "dbg!(expr)", &format!("dbg!({receiver_text})")).add_to(acc, ctx.db); // fixme
107    postfix_snippet("dbgr", "dbg!(&expr)", &format!("dbg!(&{receiver_text})")).add_to(acc, ctx.db);
108    postfix_snippet("call", "function(expr)", &format!("${{1}}({receiver_text})"))
109        .add_to(acc, ctx.db);
110
111    let try_enum = TryEnum::from_ty(&ctx.sema, &receiver_ty.strip_references());
112    let mut is_in_cond = false;
113    if let Some(parent) = dot_receiver_including_refs.syntax().parent()
114        && let Some(second_ancestor) = parent.parent()
115    {
116        let sec_ancestor_kind = second_ancestor.kind();
117        if let Some(expr) = <Either<ast::IfExpr, ast::WhileExpr>>::cast(second_ancestor) {
118            is_in_cond = match expr {
119                Either::Left(it) => it.condition().is_some_and(|cond| *cond.syntax() == parent),
120                Either::Right(it) => it.condition().is_some_and(|cond| *cond.syntax() == parent),
121            }
122        }
123        match &try_enum {
124            Some(try_enum) if is_in_cond => match try_enum {
125                TryEnum::Result => {
126                    postfix_snippet("let", "let Ok(_)", &format!("let Ok($0) = {receiver_text}"))
127                        .add_to(acc, ctx.db);
128                    postfix_snippet(
129                        "letm",
130                        "let Ok(mut _)",
131                        &format!("let Ok(mut $0) = {receiver_text}"),
132                    )
133                    .add_to(acc, ctx.db);
134                }
135                TryEnum::Option => {
136                    postfix_snippet(
137                        "let",
138                        "let Some(_)",
139                        &format!("let Some($0) = {receiver_text}"),
140                    )
141                    .add_to(acc, ctx.db);
142                    postfix_snippet(
143                        "letm",
144                        "let Some(mut _)",
145                        &format!("let Some(mut $0) = {receiver_text}"),
146                    )
147                    .add_to(acc, ctx.db);
148                }
149            },
150            _ if matches!(sec_ancestor_kind, STMT_LIST | EXPR_STMT) => {
151                postfix_snippet("let", "let", &format!("let $0 = {receiver_text};"))
152                    .add_to(acc, ctx.db);
153                postfix_snippet("letm", "let mut", &format!("let mut $0 = {receiver_text};"))
154                    .add_to(acc, ctx.db);
155            }
156            _ => (),
157        }
158    }
159
160    if !is_in_cond {
161        match try_enum {
162            Some(try_enum) => match try_enum {
163                TryEnum::Result => {
164                    postfix_snippet(
165                    "match",
166                    "match expr {}",
167                    &format!("match {receiver_text} {{\n    Ok(${{1:_}}) => {{$2}},\n    Err(${{3:_}}) => {{$0}},\n}}"),
168                )
169                .add_to(acc, ctx.db);
170                }
171                TryEnum::Option => {
172                    postfix_snippet(
173                    "match",
174                    "match expr {}",
175                    &format!(
176                        "match {receiver_text} {{\n    Some(${{1:_}}) => {{$2}},\n    None => {{$0}},\n}}"
177                    ),
178                )
179                .add_to(acc, ctx.db);
180                }
181            },
182            None => {
183                postfix_snippet(
184                    "match",
185                    "match expr {}",
186                    &format!("match {receiver_text} {{\n    ${{1:_}} => {{$0}},\n}}"),
187                )
188                .add_to(acc, ctx.db);
189            }
190        }
191        if let Some(try_enum) = &try_enum {
192            match try_enum {
193                TryEnum::Result => {
194                    postfix_snippet(
195                        "ifl",
196                        "if let Ok {}",
197                        &format!("if let Ok($1) = {receiver_text} {{\n    $0\n}}"),
198                    )
199                    .add_to(acc, ctx.db);
200
201                    postfix_snippet(
202                        "lete",
203                        "let Ok else {}",
204                        &format!("let Ok($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
205                    )
206                    .add_to(acc, ctx.db);
207
208                    postfix_snippet(
209                        "while",
210                        "while let Ok {}",
211                        &format!("while let Ok($1) = {receiver_text} {{\n    $0\n}}"),
212                    )
213                    .add_to(acc, ctx.db);
214                }
215                TryEnum::Option => {
216                    postfix_snippet(
217                        "ifl",
218                        "if let Some {}",
219                        &format!("if let Some($1) = {receiver_text} {{\n    $0\n}}"),
220                    )
221                    .add_to(acc, ctx.db);
222
223                    postfix_snippet(
224                        "lete",
225                        "let Some else {}",
226                        &format!("let Some($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
227                    )
228                    .add_to(acc, ctx.db);
229
230                    postfix_snippet(
231                        "while",
232                        "while let Some {}",
233                        &format!("while let Some($1) = {receiver_text} {{\n    $0\n}}"),
234                    )
235                    .add_to(acc, ctx.db);
236                }
237            }
238        } else if receiver_ty.is_bool() || receiver_ty.is_unknown() {
239            postfix_snippet("if", "if expr {}", &format!("if {receiver_text} {{\n    $0\n}}"))
240                .add_to(acc, ctx.db);
241            postfix_snippet(
242                "while",
243                "while expr {}",
244                &format!("while {receiver_text} {{\n    $0\n}}"),
245            )
246            .add_to(acc, ctx.db);
247            postfix_snippet("not", "!expr", &format!("!{receiver_text}")).add_to(acc, ctx.db);
248        } else if let Some(trait_) = ctx.famous_defs().core_iter_IntoIterator()
249            && receiver_ty.impls_trait(ctx.db, trait_, &[])
250        {
251            postfix_snippet(
252                "for",
253                "for ele in expr {}",
254                &format!("for ele in {receiver_text} {{\n    $0\n}}"),
255            )
256            .add_to(acc, ctx.db);
257        }
258    }
259
260    let mut block_should_be_wrapped = true;
261    if dot_receiver.syntax().kind() == BLOCK_EXPR {
262        block_should_be_wrapped = false;
263        if let Some(parent) = dot_receiver.syntax().parent()
264            && matches!(parent.kind(), IF_EXPR | WHILE_EXPR | LOOP_EXPR | FOR_EXPR)
265        {
266            block_should_be_wrapped = true;
267        }
268    };
269    {
270        let (open_brace, close_brace) =
271            if block_should_be_wrapped { ("{ ", " }") } else { ("", "") };
272        let (open_paren, close_paren) = if is_in_cond { ("(", ")") } else { ("", "") };
273        let unsafe_completion_string =
274            format!("{open_paren}unsafe {open_brace}{receiver_text}{close_brace}{close_paren}");
275        postfix_snippet("unsafe", "unsafe {}", &unsafe_completion_string).add_to(acc, ctx.db);
276
277        let const_completion_string =
278            format!("{open_paren}const {open_brace}{receiver_text}{close_brace}{close_paren}");
279        postfix_snippet("const", "const {}", &const_completion_string).add_to(acc, ctx.db);
280    }
281
282    if let ast::Expr::Literal(literal) = dot_receiver_including_refs.clone()
283        && let Some(literal_text) = ast::String::cast(literal.token())
284    {
285        add_format_like_completions(acc, ctx, &dot_receiver_including_refs, cap, &literal_text);
286    }
287
288    postfix_snippet(
289        "return",
290        "return expr",
291        &format!(
292            "return {receiver_text}{semi}",
293            semi = if expr_ctx.in_block_expr { ";" } else { "" }
294        ),
295    )
296    .add_to(acc, ctx.db);
297
298    if let BreakableKind::Block | BreakableKind::Loop = expr_ctx.in_breakable {
299        postfix_snippet(
300            "break",
301            "break expr",
302            &format!(
303                "break {receiver_text}{semi}",
304                semi = if expr_ctx.in_block_expr { ";" } else { "" }
305            ),
306        )
307        .add_to(acc, ctx.db);
308    }
309}

311fn get_receiver_text(
312    sema: &Semantics<'_, RootDatabase>,
313    receiver: &ast::Expr,
314    receiver_is_ambiguous_float_literal: bool,
315) -> String {
316    // Do not just call `receiver.to_string()`, as that will mess up whitespaces inside macros.
317    let Some(mut range) = sema.original_range_opt(receiver.syntax()) else {
318        return receiver.to_string();
319    };
320    if receiver_is_ambiguous_float_literal {
321        range.range = TextRange::at(range.range.start(), range.range.len() - TextSize::of('.'))
322    }
323    let file_text = sema.db.file_text(range.file_id.file_id(sema.db));
324    let mut text = file_text.text(sema.db)[range.range].to_owned();
325
326    // The receiver texts should be interpreted as-is, as they are expected to be
327    // normal Rust expressions.
328    escape_snippet_bits(&mut text);
329    text
330}

332/// Escapes `\` and `$` so that they don't get interpreted as snippet-specific constructs.
333///
334/// Note that we don't need to escape the other characters that can be escaped,
335/// because they wouldn't be treated as snippet-specific constructs without '$'.
336fn escape_snippet_bits(text: &mut String) {
337    stdx::replace(text, '\\', "\\\\");
338    stdx::replace(text, '$', "\\$");
339}

341fn include_references(initial_element: &ast::Expr) -> (ast::Expr, String) {
342    let mut resulting_element = initial_element.clone();
343
344    while let Some(field_expr) = resulting_element.syntax().parent().and_then(ast::FieldExpr::cast)
345    {
346        resulting_element = ast::Expr::from(field_expr);
347    }
348
349    let mut prefix = String::new();
350
351    let mut found_ref_or_deref = false;
352
353    while let Some(parent_deref_element) =
354        resulting_element.syntax().parent().and_then(ast::PrefixExpr::cast)
355    {
356        if parent_deref_element.op_kind() != Some(ast::UnaryOp::Deref) {
357            break;
358        }
359
360        found_ref_or_deref = true;
361        resulting_element = ast::Expr::from(parent_deref_element);
362
363        prefix.insert(0, '*');
364    }
365
366    while let Some(parent_ref_element) =
367        resulting_element.syntax().parent().and_then(ast::RefExpr::cast)
368    {
369        found_ref_or_deref = true;
370        let exclusive = parent_ref_element.mut_token().is_some();
371        resulting_element = ast::Expr::from(parent_ref_element);
372
373        prefix.insert_str(0, if exclusive { "&mut " } else { "&" });
374    }
375
376    if !found_ref_or_deref {
377        // If we do not find any ref/deref expressions, restore
378        // all the progress of tree climbing
379        prefix.clear();
380        resulting_element = initial_element.clone();
381    }
382
383    (resulting_element, prefix)
384}

386fn build_postfix_snippet_builder<'ctx>(
387    ctx: &'ctx CompletionContext<'_>,
388    cap: SnippetCap,
389    receiver: &'ctx ast::Expr,
390) -> Option<impl Fn(&str, &str, &str) -> Builder + 'ctx> {
391    let receiver_range = ctx.sema.original_range_opt(receiver.syntax())?.range;
392    if ctx.source_range().end() < receiver_range.start() {
393        // This shouldn't happen, yet it does. I assume this might be due to an incorrect token
394        // mapping.
395        never!();
396        return None;
397    }
398    let delete_range = TextRange::new(receiver_range.start(), ctx.source_range().end());
399
400    // Wrapping impl Fn in an option ruins lifetime inference for the parameters in a way that
401    // can't be annotated for the closure, hence fix it by constructing it without the Option first
402    fn build<'ctx>(
403        ctx: &'ctx CompletionContext<'_>,
404        cap: SnippetCap,
405        delete_range: TextRange,
406    ) -> impl Fn(&str, &str, &str) -> Builder + 'ctx {
407        move |label, detail, snippet| {
408            let edit = TextEdit::replace(delete_range, snippet.to_owned());
409            let mut item = CompletionItem::new(
410                CompletionItemKind::Snippet,
411                ctx.source_range(),
412                label,
413                ctx.edition,
414            );
415            item.detail(detail).snippet_edit(cap, edit);
416            let postfix_match = if ctx.original_token.text() == label {
417                cov_mark::hit!(postfix_exact_match_is_high_priority);
418                Some(CompletionRelevancePostfixMatch::Exact)
419            } else {
420                cov_mark::hit!(postfix_inexact_match_is_low_priority);
421                Some(CompletionRelevancePostfixMatch::NonExact)
422            };
423            let relevance = CompletionRelevance { postfix_match, ..Default::default() };
424            item.set_relevance(relevance);
425            item
426        }
427    }
428    Some(build(ctx, cap, delete_range))
429}

431fn add_custom_postfix_completions(
432    acc: &mut Completions,
433    ctx: &CompletionContext<'_>,
434    postfix_snippet: impl Fn(&str, &str, &str) -> Builder,
435    receiver_text: &str,
436) -> Option<()> {
437    ImportScope::find_insert_use_container(&ctx.token.parent()?, &ctx.sema)?;
438    ctx.config.postfix_snippets().filter(|(_, snip)| snip.scope == SnippetScope::Expr).for_each(
439        |(trigger, snippet)| {
440            let imports = match snippet.imports(ctx) {
441                Some(imports) => imports,
442                None => return,
443            };
444            let body = snippet.postfix_snippet(receiver_text);
445            let mut builder =
446                postfix_snippet(trigger, snippet.description.as_deref().unwrap_or_default(), &body);
447            builder.documentation(Documentation::new(format!("```rust\n{body}\n```")));
448            for import in imports.into_iter() {
449                builder.add_import(import);
450            }
451            builder.add_to(acc, ctx.db);
452        },
453    );
454    None
455}

762        check_edit(kind, r#"fn main() { foo.$0 }"#, &format!("fn main() {{ {kind} {{ foo }} }}"));
763        check_edit(
764            kind,
765            r#"fn main() { { foo }.$0 }"#,
766            &format!("fn main() {{ {kind} {{ foo }} }}"),
767        );
768        check_edit(
769            kind,
770            r#"fn main() { if x { foo }.$0 }"#,
771            &format!("fn main() {{ {kind} {{ if x {{ foo }} }} }}"),
772        );
773        check_edit(
774            kind,
775            r#"fn main() { loop { foo }.$0 }"#,
776            &format!("fn main() {{ {kind} {{ loop {{ foo }} }} }}"),
777        );
778        check_edit(
779            kind,
780            r#"fn main() { if true {}.$0 }"#,
781            &format!("fn main() {{ {kind} {{ if true {{}} }} }}"),
782        );
783        check_edit(
784            kind,
785            r#"fn main() { while true {}.$0 }"#,
786            &format!("fn main() {{ {kind} {{ while true {{}} }} }}"),
787        );
788        check_edit(
789            kind,
790            r#"fn main() { for i in 0..10 {}.$0 }"#,
791            &format!("fn main() {{ {kind} {{ for i in 0..10 {{}} }} }}"),
792        );
793        check_edit(
794            kind,
795            r#"fn main() { let x = if true {1} else {2}.$0 }"#,
