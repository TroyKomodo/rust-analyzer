10// Diagnostic: missing-unsafe
11//
12// This diagnostic is triggered if an operation marked as `unsafe` is used outside of an `unsafe` function or block.
13pub(crate) fn missing_unsafe(ctx: &DiagnosticsContext<'_>, d: &hir::MissingUnsafe) -> Diagnostic {
14    let code = match d.lint {
15        UnsafeLint::HardError => DiagnosticCode::RustcHardError("E0133"),
16        UnsafeLint::UnsafeOpInUnsafeFn => DiagnosticCode::RustcLint("unsafe_op_in_unsafe_fn"),
17        UnsafeLint::DeprecatedSafe2024 => DiagnosticCode::RustcLint("deprecated_safe_2024"),
18    };
19    let operation = display_unsafety_reason(d.reason);
20    Diagnostic::new_with_syntax_node_ptr(
21        ctx,
22        code,
23        format!("{operation} is unsafe and requires an unsafe function or block"),
24        d.node.map(|it| it.into()),
25    )
26    .stable()
27    .with_fixes(fixes(ctx, d))
28}
29
30fn display_unsafety_reason(reason: UnsafetyReason) -> &'static str {
31    match reason {
32        UnsafetyReason::UnionField => "access to union field",
33        UnsafetyReason::UnsafeFnCall => "call to unsafe function",
34        UnsafetyReason::InlineAsm => "use of inline assembly",
35        UnsafetyReason::RawPtrDeref => "dereference of raw pointer",
36        UnsafetyReason::MutableStatic => "use of mutable static",
37        UnsafetyReason::ExternStatic => "use of extern static",
38    }
39}

41fn fixes(ctx: &DiagnosticsContext<'_>, d: &hir::MissingUnsafe) -> Option<Vec<Assist>> {
42    // The fixit will not work correctly for macro expansions, so we don't offer it in that case.
43    if d.node.file_id.is_macro() {
44        return None;
45    }
46
47    let root = ctx.sema.db.parse_or_expand(d.node.file_id);
48    let node = d.node.value.to_node(&root);
49    let expr = node.syntax().ancestors().find_map(ast::Expr::cast)?;
50
51    let node_to_add_unsafe_block = pick_best_node_to_add_unsafe_block(&expr)?;
52
53    let replacement = format!("unsafe {{ {} }}", node_to_add_unsafe_block.text());
54    let edit = TextEdit::replace(node_to_add_unsafe_block.text_range(), replacement);
55    let source_change = SourceChange::from_text_edit(
56        d.node.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
57        edit,
58    );
59    Some(vec![fix("add_unsafe", "Add unsafe block", source_change, expr.syntax().text_range())])
60}

62// Pick the first ancestor expression of the unsafe `expr` that is not a
63// receiver of a method call, a field access, the left-hand side of an
64// assignment, or a reference. As all of those cases would incur a forced move
65// if wrapped which might not be wanted. That is:
66// - `unsafe_expr.foo` -> `unsafe { unsafe_expr.foo }`
67// - `unsafe_expr.foo.bar` -> `unsafe { unsafe_expr.foo.bar }`
68// - `unsafe_expr.foo()` -> `unsafe { unsafe_expr.foo() }`
69// - `unsafe_expr.foo.bar()` -> `unsafe { unsafe_expr.foo.bar() }`
70// - `unsafe_expr += 1` -> `unsafe { unsafe_expr += 1 }`
71// - `&unsafe_expr` -> `unsafe { &unsafe_expr }`
72// - `&&unsafe_expr` -> `unsafe { &&unsafe_expr }`
73fn pick_best_node_to_add_unsafe_block(unsafe_expr: &ast::Expr) -> Option<SyntaxNode> {
74    // The `unsafe_expr` might be:
75    // - `ast::CallExpr`: call an unsafe function
76    // - `ast::MethodCallExpr`: call an unsafe method
77    // - `ast::PrefixExpr`: dereference a raw pointer
78    // - `ast::PathExpr`: access a static mut variable
79    for (node, parent) in
80        unsafe_expr.syntax().ancestors().zip(unsafe_expr.syntax().ancestors().skip(1))
81    {
82        match_ast! {
83            match parent {
84                // If the `parent` is a `MethodCallExpr`, that means the `node`
85                // is the receiver of the method call, because only the receiver
86                // can be a direct child of a method call. The method name
87                // itself is not an expression but a `NameRef`, and an argument
88                // is a direct child of an `ArgList`.
89                ast::MethodCallExpr(_) => continue,
90                ast::FieldExpr(_) => continue,
91                ast::RefExpr(_) => continue,
92                ast::BinExpr(it) => {
93                    // Check if the `node` is the left-hand side of an
94                    // assignment, if so, we don't want to wrap it in an unsafe
95                    // block, e.g. `unsafe_expr += 1`
96                    let is_left_hand_side_of_assignment = {
97                        if let Some(ast::BinaryOp::Assignment { .. }) = it.op_kind() {
98                            it.lhs().map(|lhs| lhs.syntax().text_range().contains_range(node.text_range())).unwrap_or(false)
99                        } else {
100                            false
101                        }
102                    };
103                    if !is_left_hand_side_of_assignment {
104                        return Some(node);
105                    }
106                },
107                _ => { return Some(node); }
108
109            }
110        }
111    }
112    None
113}
