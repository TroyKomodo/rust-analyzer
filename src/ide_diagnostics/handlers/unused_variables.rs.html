13// Diagnostic: unused-variables
14//
15// This diagnostic is triggered when a local variable is not used.
16pub(crate) fn unused_variables(
17    ctx: &DiagnosticsContext<'_>,
18    d: &hir::UnusedVariable,
19) -> Option<Diagnostic> {
20    let ast = d.local.primary_source(ctx.sema.db).syntax_ptr();
21    if ast.file_id.macro_file().is_some() {
22        // FIXME: Our infra can't handle allow from within macro expansions rn
23        return None;
24    }
25    let diagnostic_range = ctx.sema.diagnostics_display_range(ast);
26    // The range for the Actual Name. We don't want to replace the entire declaration. Using the diagnostic range causes issues within in Array Destructuring.
27    let name_range = d
28        .local
29        .primary_source(ctx.sema.db)
30        .name()
31        .map(|v| v.syntax().original_file_range_rooted(ctx.sema.db))
32        .filter(|it| {
33            Some(it.file_id) == ast.file_id.file_id()
34                && diagnostic_range.range.contains_range(it.range)
35        });
36    let var_name = d.local.name(ctx.sema.db);
37    Some(
38        Diagnostic::new_with_syntax_node_ptr(
39            ctx,
40            DiagnosticCode::RustcLint("unused_variables"),
41            "unused variable",
42            ast,
43        )
44        .with_fixes(name_range.and_then(|it| {
45            fixes(
46                ctx.sema.db,
47                var_name,
48                it.range,
49                diagnostic_range,
50                ast.file_id.is_macro(),
51                ctx.edition,
52            )
53        })),
54    )
55}
56
57fn fixes(
58    db: &RootDatabase,
59    var_name: Name,
60    name_range: TextRange,
61    diagnostic_range: FileRange,
62    is_in_marco: bool,
63    edition: Edition,
64) -> Option<Vec<Assist>> {
65    if is_in_marco {
66        return None;
67    }
68
69    Some(vec![Assist {
70        id: AssistId::quick_fix("unscore_unused_variable_name"),
71        label: Label::new(format!(
72            "Rename unused {} to _{}",
73            var_name.display(db, edition),
74            var_name.display(db, edition)
75        )),
76        group: None,
77        target: diagnostic_range.range,
78        source_change: Some(SourceChange::from_text_edit(
79            diagnostic_range.file_id,
80            TextEdit::replace(name_range, format!("_{}", var_name.display(db, edition))),
81        )),
82        command: None,
83    }])
84}

91    fn unused_variables_simple() {
92        check_diagnostics(
93            r#"
94//- minicore: fn
95struct Foo { f1: i32, f2: i64 }
96
97fn f(kkk: i32) {}
98   //^^^ 💡 warn: unused variable
99fn main() {
100    let a = 2;
101      //^ 💡 warn: unused variable
102    let b = 5;
103    // note: `unused variable` implies `unused mut`, so we should not emit both at the same time.
104    let mut c = f(b);
105      //^^^^^ 💡 warn: unused variable
106    let (d, e) = (3, 5);
107       //^ 💡 warn: unused variable
108    let _ = e;
109    let f1 = 2;
110    let f2 = 5;
111    let f = Foo { f1, f2 };
112    match f {
113        Foo { f1, f2 } => {
114            //^^ 💡 warn: unused variable
115            _ = f2;
116        }
117    }
118    let g = false;
119    if g {}
120    let h: fn() -> i32 = || 2;
121    let i = h();
122      //^ 💡 warn: unused variable
123}
124"#,
125        );
126    }
