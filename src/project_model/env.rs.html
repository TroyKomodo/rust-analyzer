8/// Recreates the compile-time environment variables that Cargo sets.
9///
10/// Should be synced with
11/// <https://doc.rust-lang.org/cargo/reference/environment-variables.html#environment-variables-cargo-sets-for-crates>
12///
13/// FIXME: ask Cargo to provide this data instead of re-deriving.
14pub(crate) fn inject_cargo_package_env(env: &mut Env, package: &PackageData) {
15    // FIXME: Missing variables:
16    // CARGO_BIN_NAME, CARGO_BIN_EXE_<name>
17
18    let manifest_dir = package.manifest.parent();
19    env.set("CARGO_MANIFEST_DIR", manifest_dir.as_str());
20    env.set("CARGO_MANIFEST_PATH", package.manifest.as_str());
21
22    env.set("CARGO_PKG_VERSION", package.version.to_string());
23    env.set("CARGO_PKG_VERSION_MAJOR", package.version.major.to_string());
24    env.set("CARGO_PKG_VERSION_MINOR", package.version.minor.to_string());
25    env.set("CARGO_PKG_VERSION_PATCH", package.version.patch.to_string());
26    env.set("CARGO_PKG_VERSION_PRE", package.version.pre.to_string());
27
28    env.set("CARGO_PKG_AUTHORS", package.authors.join(":"));
29
30    env.set("CARGO_PKG_NAME", package.name.clone());
31    env.set("CARGO_PKG_DESCRIPTION", package.description.as_deref().unwrap_or_default());
32    env.set("CARGO_PKG_HOMEPAGE", package.homepage.as_deref().unwrap_or_default());
33    env.set("CARGO_PKG_REPOSITORY", package.repository.as_deref().unwrap_or_default());
34    env.set("CARGO_PKG_LICENSE", package.license.as_deref().unwrap_or_default());
35    env.set(
36        "CARGO_PKG_LICENSE_FILE",
37        package.license_file.as_ref().map(ToString::to_string).unwrap_or_default(),
38    );
39    env.set(
40        "CARGO_PKG_README",
41        package.readme.as_ref().map(ToString::to_string).unwrap_or_default(),
42    );
43
44    env.set(
45        "CARGO_PKG_RUST_VERSION",
46        package.rust_version.as_ref().map(ToString::to_string).unwrap_or_default(),
47    );
48}

63pub(crate) fn cargo_config_env(manifest: &ManifestPath, config: &Option<CargoConfigFile>) -> Env {
64    let mut env = Env::default();
65    let Some(serde_json::Value::Object(env_json)) = config.as_ref().and_then(|c| c.get("env"))
66    else {
67        return env;
68    };
69
70    // FIXME: The base here should be the parent of the `.cargo/config` file, not the manifest.
71    // But cargo does not provide this information.
72    let base = <_ as AsRef<Utf8Path>>::as_ref(manifest.parent());
73
74    for (key, entry) in env_json {
75        let serde_json::Value::Object(entry) = entry else {
76            continue;
77        };
78        let Some(value) = entry.get("value").and_then(|v| v.as_str()) else {
79            continue;
80        };
81
82        let value = if entry
83            .get("relative")
84            .and_then(|v| v.as_bool())
85            .is_some_and(std::convert::identity)
86        {
87            base.join(value).to_string()
88        } else {
89            value.to_owned()
90        };
91        env.insert(key, value);
92    }
93
94    env
95}

97#[test]
98fn parse_output_cargo_config_env_works() {
99    let raw = r#"
100{
101  "env": {
102    "CARGO_WORKSPACE_DIR": {
103      "relative": true,
104      "value": ""
105    },
106    "INVALID": {
107      "relative": "invalidbool",
108      "value": "../relative"
109    },
110    "RELATIVE": {
111      "relative": true,
112      "value": "../relative"
113    },
114    "TEST": {
115      "value": "test"
116    }
117  }
118}
119"#;
120    let config: CargoConfigFile = serde_json::from_str(raw).unwrap();
121    let cwd = paths::Utf8PathBuf::try_from(std::env::current_dir().unwrap()).unwrap();
122    let manifest = paths::AbsPathBuf::assert(cwd.join("Cargo.toml"));
123    let manifest = ManifestPath::try_from(manifest).unwrap();
124    let env = cargo_config_env(&manifest, &Some(config));
125    assert_eq!(env.get("CARGO_WORKSPACE_DIR").as_deref(), Some(cwd.join("").as_str()));
126    assert_eq!(env.get("RELATIVE").as_deref(), Some(cwd.join("../relative").as_str()));
127    assert_eq!(env.get("INVALID").as_deref(), Some("../relative"));
128    assert_eq!(env.get("TEST").as_deref(), Some("test"));
129}
