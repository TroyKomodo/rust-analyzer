1//! Runs `rustc --print target-spec-json` to get the target_data_layout.
2
3use anyhow::Context;
4use rustc_hash::FxHashMap;
5use toolchain::Tool;
6
7use crate::{Sysroot, toolchain_info::QueryConfig, utf8_stdout};
8
9/// Uses `rustc --print target-spec-json`.
10pub fn get(
11    config: QueryConfig<'_>,
12    target: Option<&str>,
13    extra_env: &FxHashMap<String, Option<String>>,
14) -> anyhow::Result<String> {
15    const RUSTC_ARGS: [&str; 2] = ["--print", "target-spec-json"];
16    let process = |output: String| {
17        (|| Some(output.split_once(r#""data-layout": ""#)?.1.split_once('"')?.0.to_owned()))()
18            .ok_or_else(|| {
19                anyhow::format_err!("could not parse target-spec-json from command output")
20            })
21    };
22    let (sysroot, current_dir) = match config {
23        QueryConfig::Cargo(sysroot, cargo_toml, _) => {
24            let mut cmd = sysroot.tool(Tool::Cargo, cargo_toml.parent(), extra_env);
25            cmd.env("RUSTC_BOOTSTRAP", "1");
26            cmd.args(["rustc", "-Z", "unstable-options"]).args(RUSTC_ARGS).args([
27                "--",
28                "-Z",
29                "unstable-options",
30            ]);
31            if let Some(target) = target {
32                cmd.args(["--target", target]);
33            }
34            match utf8_stdout(&mut cmd) {
35                Ok(output) => return process(output),
36                Err(e) => {
37                    tracing::warn!(%e, "failed to run `{cmd:?}`, falling back to invoking rustc directly");
38                    (sysroot, cargo_toml.parent().as_ref())
39                }
40            }
41        }
42        QueryConfig::Rustc(sysroot, current_dir) => (sysroot, current_dir),
43    };
44
45    let mut cmd = Sysroot::tool(sysroot, Tool::Rustc, current_dir, extra_env);
46    cmd.env("RUSTC_BOOTSTRAP", "1").args(["-Z", "unstable-options"]).args(RUSTC_ARGS);
47    if let Some(target) = target {
48        cmd.args(["--target", target]);
49    }
50    utf8_stdout(&mut cmd)
51        .with_context(|| format!("unable to fetch target-data-layout via `{cmd:?}`"))
52        .and_then(process)
53}

55#[cfg(test)]
56mod tests {
57    use paths::{AbsPathBuf, Utf8PathBuf};
58
59    use crate::{ManifestPath, Sysroot};
60
61    use super::*;
62
63    #[test]
64    fn cargo() {
65        let manifest_path = concat!(env!("CARGO_MANIFEST_DIR"), "/Cargo.toml");
66        let sysroot = Sysroot::empty();
67        let manifest_path =
68            ManifestPath::try_from(AbsPathBuf::assert(Utf8PathBuf::from(manifest_path))).unwrap();
69        let cfg = QueryConfig::Cargo(&sysroot, &manifest_path, &None);
70        assert!(get(cfg, None, &FxHashMap::default()).is_ok());
71    }
72
73    #[test]
74    fn rustc() {
75        let sysroot = Sysroot::empty();
76        let cfg = QueryConfig::Rustc(&sysroot, env!("CARGO_MANIFEST_DIR").as_ref());
77        assert!(get(cfg, None, &FxHashMap::default()).is_ok());
78    }
