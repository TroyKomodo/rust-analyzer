16/// Cargo output is structured as one JSON per line. This trait abstracts parsing one line of
17/// cargo output into a Rust data type
18pub(crate) trait CargoParser<T>: Send + 'static {
19    fn from_line(&self, line: &str, error: &mut String) -> Option<T>;
20    fn from_eof(&self) -> Option<T>;
21}
22
23struct CargoActor<T> {
24    parser: Box<dyn CargoParser<T>>,
25    sender: Sender<T>,
26    stdout: ChildStdout,
27    stderr: ChildStderr,
28}
29
30impl<T: Sized + Send + 'static> CargoActor<T> {
31    fn new(
32        parser: impl CargoParser<T>,
33        sender: Sender<T>,
34        stdout: ChildStdout,
35        stderr: ChildStderr,
36    ) -> Self {
37        let parser = Box::new(parser);
38        CargoActor { parser, sender, stdout, stderr }
39    }
40}
41
42impl<T: Sized + Send + 'static> CargoActor<T> {
43    fn run(self) -> io::Result<(bool, String)> {
44        // We manually read a line at a time, instead of using serde's
45        // stream deserializers, because the deserializer cannot recover
46        // from an error, resulting in it getting stuck, because we try to
47        // be resilient against failures.
48        //
49        // Because cargo only outputs one JSON object per line, we can
50        // simply skip a line if it doesn't parse, which just ignores any
51        // erroneous output.
52
53        let mut stdout_errors = String::new();
54        let mut stderr_errors = String::new();
55        let mut read_at_least_one_stdout_message = false;
56        let mut read_at_least_one_stderr_message = false;
57        let process_line = |line: &str, error: &mut String| {
58            // Try to deserialize a message from Cargo or Rustc.
59            if let Some(t) = self.parser.from_line(line, error) {
60                self.sender.send(t).unwrap();
61                true
62            } else {
63                false
64            }
65        };
66        let output = streaming_output(
67            self.stdout,
68            self.stderr,
69            &mut |line| {
70                if process_line(line, &mut stdout_errors) {
71                    read_at_least_one_stdout_message = true;
72                }
73            },
74            &mut |line| {
75                if process_line(line, &mut stderr_errors) {
76                    read_at_least_one_stderr_message = true;
77                }
78            },
79            &mut || {
80                if let Some(t) = self.parser.from_eof() {
81                    self.sender.send(t).unwrap();
82                }
83            },
84        );
85
86        let read_at_least_one_message =
87            read_at_least_one_stdout_message || read_at_least_one_stderr_message;
88        let mut error = stdout_errors;
89        error.push_str(&stderr_errors);
90        match output {
91            Ok(_) => Ok((read_at_least_one_message, error)),
92            Err(e) => Err(io::Error::new(e.kind(), format!("{e:?}: {error}"))),
93        }
94    }
95}

97struct JodGroupChild(Box<dyn StdChildWrapper>);
98
99impl Drop for JodGroupChild {
100    fn drop(&mut self) {
101        _ = self.0.kill();
102        _ = self.0.wait();
103    }
104}

106/// A handle to a cargo process used for fly-checking.
107pub(crate) struct CommandHandle<T> {
108    /// The handle to the actual cargo process. As we cannot cancel directly from with
109    /// a read syscall dropping and therefore terminating the process is our best option.
110    child: JodGroupChild,
111    thread: stdx::thread::JoinHandle<io::Result<(bool, String)>>,
112    program: OsString,
113    arguments: Vec<OsString>,
114    current_dir: Option<PathBuf>,
115    _phantom: PhantomData<T>,
116}
117
118impl<T> fmt::Debug for CommandHandle<T> {
119    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
120        f.debug_struct("CommandHandle")
121            .field("program", &self.program)
122            .field("arguments", &self.arguments)
123            .field("current_dir", &self.current_dir)
124            .finish()
125    }
126}
127
128impl<T: Sized + Send + 'static> CommandHandle<T> {
129    pub(crate) fn spawn(
130        mut command: Command,
131        parser: impl CargoParser<T>,
132        sender: Sender<T>,
133    ) -> std::io::Result<Self> {
134        command.stdout(Stdio::piped()).stderr(Stdio::piped()).stdin(Stdio::null());
135
136        let program = command.get_program().into();
137        let arguments = command.get_args().map(|arg| arg.into()).collect::<Vec<OsString>>();
138        let current_dir = command.get_current_dir().map(|arg| arg.to_path_buf());
139
140        let mut child = StdCommandWrap::from(command);
141        #[cfg(unix)]
142        child.wrap(process_wrap::std::ProcessSession);
143        #[cfg(windows)]
144        child.wrap(process_wrap::std::JobObject);
145        let mut child = child.spawn().map(JodGroupChild)?;
146
147        let stdout = child.0.stdout().take().unwrap();
148        let stderr = child.0.stderr().take().unwrap();
149
150        let actor = CargoActor::<T>::new(parser, sender, stdout, stderr);
151        let thread =
152            stdx::thread::Builder::new(stdx::thread::ThreadIntent::Worker, "CommandHandle")
153                .spawn(move || actor.run())
154                .expect("failed to spawn thread");
155        Ok(CommandHandle { program, arguments, current_dir, child, thread, _phantom: PhantomData })
156    }
157
158    pub(crate) fn cancel(mut self) {
159        let _ = self.child.0.kill();
160        let _ = self.child.0.wait();
161    }
162
163    pub(crate) fn join(mut self) -> io::Result<()> {
164        let exit_status = self.child.0.wait()?;
165        let (read_at_least_one_message, error) = self.thread.join()?;
166        if read_at_least_one_message || exit_status.success() {
167            Ok(())
168        } else {
169            Err(io::Error::other(format!(
170                "Cargo watcher failed, the command produced no valid metadata (exit code: {exit_status:?}):\n{error}"
171            )))
172        }
173    }
174}
