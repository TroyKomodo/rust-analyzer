1//! Project loading & configuration updates.
2//!
3//! This is quite tricky. The main problem is time and changes -- there's no
4//! fixed "project" rust-analyzer is working with, "current project" is itself
5//! mutable state. For example, when the user edits `Cargo.toml` by adding a new
6//! dependency, project model changes. What's more, switching project model is
7//! not instantaneous -- it takes time to run `cargo metadata` and (for proc
8//! macros) `cargo check`.
9//!
10//! The main guiding principle here is, as elsewhere in rust-analyzer,
11//! robustness. We try not to assume that the project model exists or is
12//! correct. Instead, we try to provide a best-effort service. Even if the
13//! project is currently loading and we don't have a full project model, we
14//! still want to respond to various  requests.

66    /// Is the server quiescent?
67    ///
68    /// This indicates that we've fully loaded the projects and
69    /// are ready to do semantic work.
70    pub(crate) fn is_quiescent(&self) -> bool {
71        self.vfs_done
72            && self.fetch_ws_receiver.is_none()
73            && !self.fetch_workspaces_queue.op_in_progress()
74            && !self.fetch_build_data_queue.op_in_progress()
75            && !self.fetch_proc_macros_queue.op_in_progress()
76            && !self.discover_workspace_queue.op_in_progress()
77            && self.vfs_progress_config_version >= self.vfs_config_version
78    }
79
80    /// Is the server ready to respond to analysis dependent LSP requests?
81    ///
82    /// Unlike `is_quiescent`, this returns false when we're indexing
83    /// the project, because we're holding the salsa lock and cannot
84    /// respond to LSP requests that depend on salsa data.
85    fn is_fully_ready(&self) -> bool {
86        self.is_quiescent() && !self.prime_caches_queue.op_in_progress()
87    }

89    pub(crate) fn update_configuration(&mut self, config: Config) {
90        let _p = tracing::info_span!("GlobalState::update_configuration").entered();
91        let old_config = mem::replace(&mut self.config, Arc::new(config));
92        if self.config.lru_parse_query_capacity() != old_config.lru_parse_query_capacity() {
93            self.analysis_host.update_lru_capacity(self.config.lru_parse_query_capacity());
94        }
95        if self.config.lru_query_capacities_config() != old_config.lru_query_capacities_config() {
96            self.analysis_host.update_lru_capacities(
97                &self.config.lru_query_capacities_config().cloned().unwrap_or_default(),
98            );
99        }
100
101        if self.config.linked_or_discovered_projects() != old_config.linked_or_discovered_projects()
102        {
103            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
104            self.fetch_workspaces_queue.request_op("discovered projects changed".to_owned(), req)
105        } else if self.config.flycheck(None) != old_config.flycheck(None) {
106            self.reload_flycheck();
107        }
108
109        if self.analysis_host.raw_database().expand_proc_attr_macros()
110            != self.config.expand_proc_attr_macros()
111        {
112            self.analysis_host.raw_database_mut().set_expand_proc_attr_macros_with_durability(
113                self.config.expand_proc_attr_macros(),
114                Durability::HIGH,
115            );
116        }
117
118        if self.config.cargo(None) != old_config.cargo(None) {
119            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
120            self.fetch_workspaces_queue.request_op("cargo config changed".to_owned(), req)
121        }
122
123        if self.config.cfg_set_test(None) != old_config.cfg_set_test(None) {
124            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
125            self.fetch_workspaces_queue.request_op("cfg_set_test config changed".to_owned(), req)
126        }
127    }

129    pub(crate) fn current_status(&self) -> lsp_ext::ServerStatusParams {
130        let mut status = lsp_ext::ServerStatusParams {
131            health: lsp_ext::Health::Ok,
132            quiescent: self.is_fully_ready(),
133            message: None,
134        };
135        let mut message = String::new();
136
137        if !self.config.cargo_autoreload_config(None)
138            && self.is_quiescent()
139            && self.fetch_workspaces_queue.op_requested()
140            && self.config.discover_workspace_config().is_none()
141        {
142            status.health |= lsp_ext::Health::Warning;
143            message.push_str("Auto-reloading is disabled and the workspace has changed, a manual workspace reload is required.\n\n");
144        }
145
146        if self.build_deps_changed {
147            status.health |= lsp_ext::Health::Warning;
148            message.push_str(
149                "Proc-macros and/or build scripts have changed and need to be rebuilt.\n\n",
150            );
151        }
152        if self.fetch_build_data_error().is_err() {
153            status.health |= lsp_ext::Health::Warning;
154            message.push_str("Failed to run build scripts of some packages.\n\n");
155            message.push_str("Please refer to the logs for more details on the errors.");
156        }
157        if let Some(err) = &self.config_errors {
158            status.health |= lsp_ext::Health::Warning;
159            format_to!(message, "{err}\n");
160        }
161        if let Some(err) = &self.last_flycheck_error {
162            status.health |= lsp_ext::Health::Warning;
163            message.push_str(err);
164            message.push('\n');
165        }
166
167        if self.config.linked_or_discovered_projects().is_empty()
168            && self.config.detached_files().is_empty()
169        {
170            status.health |= lsp_ext::Health::Warning;
171            message.push_str("Failed to discover workspace.\n");
172            message.push_str("Consider adding the `Cargo.toml` of the workspace to the [`linkedProjects`](https://rust-analyzer.github.io/book/configuration.html#linkedProjects) setting.\n\n");
173        }
174        if self.fetch_workspace_error().is_err() {
175            status.health |= lsp_ext::Health::Error;
176            message.push_str("Failed to load workspaces.");
177
178            if self.config.has_linked_projects() {
179                message.push_str(
180                    "`rust-analyzer.linkedProjects` have been specified, which may be incorrect. Specified project paths:\n",
181                );
182                message
183                    .push_str(&format!("    {}", self.config.linked_manifests().format("\n    ")));
184                if self.config.has_linked_project_jsons() {
185                    message.push_str("\nAdditionally, one or more project jsons are specified")
186                }
187            }
188            message.push_str("\n\n");
189        }
190
191        if !self.workspaces.is_empty() {
192            self.check_workspaces_msrv().for_each(|e| {
193                status.health |= lsp_ext::Health::Warning;
194                format_to!(message, "{e}");
195            });
196
197            let proc_macro_clients = self.proc_macro_clients.iter().chain(iter::repeat(&None));
198
199            for (ws, proc_macro_client) in self.workspaces.iter().zip(proc_macro_clients) {
200                if let ProjectWorkspaceKind::Cargo { error: Some(error), .. }
201                | ProjectWorkspaceKind::DetachedFile {
202                    cargo: Some((_, _, Some(error))), ..
203                } = &ws.kind
204                {
205                    status.health |= lsp_ext::Health::Warning;
206                    format_to!(
207                        message,
208                        "Failed to read Cargo metadata with dependencies for `{}`: {:#}\n\n",
209                        ws.manifest_or_root(),
210                        error
211                    );
212                }
213                if let Some(err) = ws.sysroot.error() {
214                    status.health |= lsp_ext::Health::Warning;
215                    format_to!(
216                        message,
217                        "Workspace `{}` has sysroot errors: ",
218                        ws.manifest_or_root()
219                    );
220                    message.push_str(err);
221                    message.push_str("\n\n");
222                }
223                if let ProjectWorkspaceKind::Cargo { rustc: Err(Some(err)), .. } = &ws.kind {
224                    status.health |= lsp_ext::Health::Warning;
225                    format_to!(
226                        message,
227                        "Failed loading rustc_private crates for workspace `{}`: ",
228                        ws.manifest_or_root()
229                    );
230                    message.push_str(err);
231                    message.push_str("\n\n");
232                };
233                match proc_macro_client {
234                    Some(Err(err)) => {
235                        status.health |= lsp_ext::Health::Warning;
236                        format_to!(
237                            message,
238                            "Failed spawning proc-macro server for workspace `{}`: {err}",
239                            ws.manifest_or_root()
240                        );
241                        message.push_str("\n\n");
242                    }
243                    Some(Ok(client)) => {
244                        if let Some(err) = client.exited() {
245                            status.health |= lsp_ext::Health::Warning;
246                            format_to!(
247                                message,
248                                "proc-macro server for workspace `{}` exited: {err}",
249                                ws.manifest_or_root()
250                            );
251                            message.push_str("\n\n");
252                        }
253                    }
254                    // sysroot was explicitly not set so we didn't discover a server
255                    None => {}
256                }
257            }
258        }
259
260        if !message.is_empty() {
261            status.message = Some(message.trim_end().to_owned());
262        }
263
264        status
265    }

267    pub(crate) fn fetch_workspaces(
268        &mut self,
269        cause: Cause,
270        path: Option<AbsPathBuf>,
271        force_crate_graph_reload: bool,
272    ) {
273        info!(%cause, "will fetch workspaces");
274
275        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, {
276            let linked_projects = self.config.linked_or_discovered_projects();
277            let detached_files: Vec<_> = self
278                .config
279                .detached_files()
280                .iter()
281                .cloned()
282                .map(ManifestPath::try_from)
283                .filter_map(Result::ok)
284                .collect();
285            let cargo_config = self.config.cargo(None);
286            let discover_command = self.config.discover_workspace_config().cloned();
287            let is_quiescent = !(self.discover_workspace_queue.op_in_progress()
288                || self.vfs_progress_config_version < self.vfs_config_version
289                || !self.vfs_done);
290
291            move |sender| {
292                let progress = {
293                    let sender = sender.clone();
294                    move |msg| {
295                        sender
296                            .send(Task::FetchWorkspace(ProjectWorkspaceProgress::Report(msg)))
297                            .unwrap()
298                    }
299                };
300
301                sender.send(Task::FetchWorkspace(ProjectWorkspaceProgress::Begin)).unwrap();
302
303                if let (Some(_command), Some(path)) = (&discover_command, &path) {
304                    let build = linked_projects.iter().find_map(|project| match project {
305                        LinkedProject::InlineProjectJson(it) => it.crate_by_buildfile(path),
306                        _ => None,
307                    });
308
309                    if let Some(build) = build
310                        && is_quiescent
311                    {
312                        let path = AbsPathBuf::try_from(build.build_file)
313                            .expect("Unable to convert to an AbsPath");
314                        let arg = DiscoverProjectParam::Buildfile(path);
315                        sender.send(Task::DiscoverLinkedProjects(arg)).unwrap();
316                    }
317                }
318
319                let mut workspaces = linked_projects
320                    .iter()
321                    .map(|project| match project {
322                        LinkedProject::ProjectManifest(manifest) => {
323                            debug!(path = %manifest, "loading project from manifest");
324
325                            project_model::ProjectWorkspace::load(
326                                manifest.clone(),
327                                &cargo_config,
328                                &progress,
329                            )
330                        }
331                        LinkedProject::InlineProjectJson(it) => {
332                            let workspace = project_model::ProjectWorkspace::load_inline(
333                                it.clone(),
334                                &cargo_config,
335                                &progress,
336                            );
337                            Ok(workspace)
338                        }
339                    })
340                    .collect::<Vec<_>>();
341
342                let mut i = 0;
343                while i < workspaces.len() {
344                    if let Ok(w) = &workspaces[i] {
345                        let dupes: Vec<_> = workspaces[i + 1..]
346                            .iter()
347                            .positions(|it| it.as_ref().is_ok_and(|ws| ws.eq_ignore_build_data(w)))
348                            .collect();
349                        dupes.into_iter().rev().for_each(|d| {
350                            _ = workspaces.remove(d + i + 1);
351                        });
352                    }
353                    i += 1;
354                }
355
356                if !detached_files.is_empty() {
357                    workspaces.extend(project_model::ProjectWorkspace::load_detached_files(
358                        detached_files,
359                        &cargo_config,
360                    ));
361                }
362
363                info!(?workspaces, "did fetch workspaces");
364                sender
365                    .send(Task::FetchWorkspace(ProjectWorkspaceProgress::End(
366                        workspaces,
367                        force_crate_graph_reload,
368                    )))
369                    .unwrap();
370            }
371        });
372    }

374    pub(crate) fn fetch_build_data(&mut self, cause: Cause) {
375        info!(%cause, "will fetch build data");
376        let workspaces = Arc::clone(&self.workspaces);
377        let config = self.config.cargo(None);
378        let root_path = self.config.root_path().clone();
379
380        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, move |sender| {
381            sender.send(Task::FetchBuildData(BuildDataProgress::Begin)).unwrap();
382
383            let progress = {
384                let sender = sender.clone();
385                move |msg| {
386                    sender.send(Task::FetchBuildData(BuildDataProgress::Report(msg))).unwrap()
387                }
388            };
389            let res = ProjectWorkspace::run_all_build_scripts(
390                &workspaces,
391                &config,
392                &progress,
393                &root_path,
394            );
395
396            sender.send(Task::FetchBuildData(BuildDataProgress::End((workspaces, res)))).unwrap();
397        });
398    }

400    pub(crate) fn fetch_proc_macros(
401        &mut self,
402        cause: Cause,
403        mut change: ChangeWithProcMacros,
404        paths: Vec<ProcMacroPaths>,
405    ) {
406        info!(%cause, "will load proc macros");
407        let ignored_proc_macros = self.config.ignored_proc_macros(None).clone();
408        let proc_macro_clients = self.proc_macro_clients.clone();
409
410        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, move |sender| {
411            sender.send(Task::LoadProcMacros(ProcMacroProgress::Begin)).unwrap();
412
413            let ignored_proc_macros = &ignored_proc_macros;
414            let progress = {
415                let sender = sender.clone();
416                &move |msg| {
417                    sender.send(Task::LoadProcMacros(ProcMacroProgress::Report(msg))).unwrap()
418                }
419            };
420
421            let mut builder = ProcMacrosBuilder::default();
422            let proc_macro_clients = proc_macro_clients.iter().chain(iter::repeat(&None));
423            for (client, paths) in proc_macro_clients.zip(paths) {
424                for (crate_id, res) in paths.iter() {
425                    let expansion_res = match client {
426                        Some(Ok(client)) => match res {
427                            Ok((crate_name, path)) => {
428                                progress(format!("loading proc-macros: {path}"));
429                                let ignored_proc_macros = ignored_proc_macros
430                                    .iter()
431                                    .find_map(|(name, macros)| {
432                                        eq_ignore_underscore(name, crate_name).then_some(&**macros)
433                                    })
434                                    .unwrap_or_default();
435
436                                load_proc_macro(client, path, ignored_proc_macros)
437                            }
438                            Err(e) => Err(e.clone()),
439                        },
440                        Some(Err(e)) => Err(ProcMacroLoadingError::ProcMacroSrvError(
441                            e.to_string().into_boxed_str(),
442                        )),
443                        None => Err(ProcMacroLoadingError::ProcMacroSrvError(
444                            "proc-macro-srv is not running".into(),
445                        )),
446                    };
447                    builder.insert(*crate_id, expansion_res)
448                }
449            }
450
451            change.set_proc_macros(builder);
452            sender.send(Task::LoadProcMacros(ProcMacroProgress::End(change))).unwrap();
453        });
454    }

456    pub(crate) fn switch_workspaces(&mut self, cause: Cause) {
457        let _p = tracing::info_span!("GlobalState::switch_workspaces").entered();
458        tracing::info!(%cause, "will switch workspaces");
459
460        let Some(FetchWorkspaceResponse { workspaces, force_crate_graph_reload }) =
461            self.fetch_workspaces_queue.last_op_result()
462        else {
463            return;
464        };
465        let switching_from_empty_workspace = self.workspaces.is_empty();
466
467        info!(%cause, ?force_crate_graph_reload, %switching_from_empty_workspace);
468        if self.fetch_workspace_error().is_err() && !switching_from_empty_workspace {
469            if *force_crate_graph_reload {
470                self.recreate_crate_graph(cause, false);
471            }
472            // It only makes sense to switch to a partially broken workspace
473            // if we don't have any workspace at all yet.
474            return;
475        }
476
477        let workspaces =
478            workspaces.iter().filter_map(|res| res.as_ref().ok().cloned()).collect::<Vec<_>>();
479
480        let same_workspaces = workspaces.len() == self.workspaces.len()
481            && workspaces
482                .iter()
483                .zip(self.workspaces.iter())
484                .all(|(l, r)| l.eq_ignore_build_data(r));
485
486        if same_workspaces {
487            if switching_from_empty_workspace {
488                // Switching from empty to empty is a no-op
489                return;
490            }
491            if let Some(FetchBuildDataResponse { workspaces, build_scripts }) =
492                self.fetch_build_data_queue.last_op_result()
493            {
494                if Arc::ptr_eq(workspaces, &self.workspaces) {
495                    info!("set build scripts to workspaces");
496
497                    let workspaces = workspaces
498                        .iter()
499                        .cloned()
500                        .zip(build_scripts)
501                        .map(|(mut ws, bs)| {
502                            ws.set_build_scripts(bs.as_ref().ok().cloned().unwrap_or_default());
503                            ws
504                        })
505                        .collect::<Vec<_>>();
506                    // Workspaces are the same, but we've updated build data.
507                    info!("same workspace, but new build data");
508                    self.workspaces = Arc::new(workspaces);
509                } else {
510                    info!("build scripts do not match the version of the active workspace");
511                    if *force_crate_graph_reload {
512                        self.recreate_crate_graph(cause, switching_from_empty_workspace);
513                    }
514
515                    // Current build scripts do not match the version of the active
516                    // workspace, so there's nothing for us to update.
517                    return;
518                }
519            } else {
520                if *force_crate_graph_reload {
521                    self.recreate_crate_graph(cause, switching_from_empty_workspace);
522                }
523
524                // No build scripts but unchanged workspaces, nothing to do here
525                return;
526            }
527        } else {
528            info!("abandon build scripts for workspaces");
529
530            // Here, we completely changed the workspace (Cargo.toml edit), so
531            // we don't care about build-script results, they are stale.
532            // FIXME: can we abort the build scripts here if they are already running?
533            self.workspaces = Arc::new(workspaces);
534            self.check_workspaces_msrv().for_each(|message| {
535                self.send_notification::<lsp_types::notification::ShowMessage>(
536                    lsp_types::ShowMessageParams { typ: lsp_types::MessageType::WARNING, message },
537                );
538            });
539
540            if self.config.run_build_scripts(None) {
541                self.build_deps_changed = false;
542                self.fetch_build_data_queue.request_op("workspace updated".to_owned(), ());
543
544                if !switching_from_empty_workspace {
545                    // `switch_workspaces()` will be called again when build scripts already run, which should
546                    // take a short time. If we update the workspace now we will invalidate proc macros and cfgs,
547                    // and then when build scripts complete we will invalidate them again.
548                    return;
549                }
550            }
551        }
552
553        if let FilesWatcher::Client = self.config.files().watcher {
554            let filter = self
555                .workspaces
556                .iter()
557                .flat_map(|ws| ws.to_roots())
558                .filter(|it| it.is_local)
559                .map(|it| it.include);
560
561            let mut watchers: Vec<FileSystemWatcher> =
562                if self.config.did_change_watched_files_relative_pattern_support() {
563                    // When relative patterns are supported by the client, prefer using them
564                    filter
565                        .flat_map(|include| {
566                            include.into_iter().flat_map(|base| {
567                                [
568                                    (base.clone(), "**/*.rs"),
569                                    (base.clone(), "**/Cargo.{lock,toml}"),
570                                    (base, "**/rust-analyzer.toml"),
571                                ]
572                            })
573                        })
574                        .map(|(base, pat)| lsp_types::FileSystemWatcher {
575                            glob_pattern: lsp_types::GlobPattern::Relative(
576                                lsp_types::RelativePattern {
577                                    base_uri: lsp_types::OneOf::Right(
578                                        lsp_types::Url::from_file_path(base).unwrap(),
579                                    ),
580                                    pattern: pat.to_owned(),
581                                },
582                            ),
583                            kind: None,
584                        })
585                        .collect()
586                } else {
587                    // When they're not, integrate the base to make them into absolute patterns
588                    filter
589                        .flat_map(|include| {
590                            include.into_iter().flat_map(|base| {
591                                [
592                                    format!("{base}/**/*.rs"),
593                                    format!("{base}/**/Cargo.{{toml,lock}}"),
594                                    format!("{base}/**/rust-analyzer.toml"),
595                                ]
596                            })
597                        })
598                        .map(|glob_pattern| lsp_types::FileSystemWatcher {
599                            glob_pattern: lsp_types::GlobPattern::String(glob_pattern),
600                            kind: None,
601                        })
602                        .collect()
603                };
604
605            // Also explicitly watch any build files configured in JSON project files.
606            for ws in self.workspaces.iter() {
607                if let ProjectWorkspaceKind::Json(project_json) = &ws.kind {
608                    for (_, krate) in project_json.crates() {
609                        let Some(build) = &krate.build else {
610                            continue;
611                        };
612                        watchers.push(lsp_types::FileSystemWatcher {
613                            glob_pattern: lsp_types::GlobPattern::String(
614                                build.build_file.to_string(),
615                            ),
616                            kind: None,
617                        });
618                    }
619                }
620            }
621
622            watchers.extend(
623                iter::once(Config::user_config_dir_path().as_deref())
624                    .chain(self.workspaces.iter().map(|ws| ws.manifest().map(ManifestPath::as_ref)))
625                    .flatten()
626                    .map(|glob_pattern| lsp_types::FileSystemWatcher {
627                        glob_pattern: lsp_types::GlobPattern::String(glob_pattern.to_string()),
628                        kind: None,
629                    }),
630            );
631
632            let registration_options =
633                lsp_types::DidChangeWatchedFilesRegistrationOptions { watchers };
634            let registration = lsp_types::Registration {
635                id: "workspace/didChangeWatchedFiles".to_owned(),
636                method: "workspace/didChangeWatchedFiles".to_owned(),
637                register_options: Some(serde_json::to_value(registration_options).unwrap()),
638            };
639            self.send_request::<lsp_types::request::RegisterCapability>(
640                lsp_types::RegistrationParams { registrations: vec![registration] },
641                |_, _| (),
642            );
643        }
644
645        let files_config = self.config.files();
646        let project_folders = ProjectFolders::new(
647            &self.workspaces,
648            &files_config.exclude,
649            Config::user_config_dir_path().as_deref(),
650        );
651
652        if (self.proc_macro_clients.len() < self.workspaces.len() || !same_workspaces)
653            && self.config.expand_proc_macros()
654        {
655            info!("Spawning proc-macro servers");
656
657            self.proc_macro_clients = Arc::from_iter(self.workspaces.iter().map(|ws| {
658                let path = match self.config.proc_macro_srv() {
659                    Some(path) => path,
660                    None => match ws.find_sysroot_proc_macro_srv()? {
661                        Ok(path) => path,
662                        Err(e) => return Some(Err(e)),
663                    },
664                };
665
666                let env: FxHashMap<_, _> = match &ws.kind {
667                    ProjectWorkspaceKind::Cargo { cargo, .. }
668                    | ProjectWorkspaceKind::DetachedFile { cargo: Some((cargo, ..)), .. } => cargo
669                        .env()
670                        .into_iter()
671                        .map(|(k, v)| (k.clone(), Some(v.clone())))
672                        .chain(
673                            self.config.extra_env(None).iter().map(|(k, v)| (k.clone(), v.clone())),
674                        )
675                        .chain(
676                            ws.sysroot
677                                .root()
678                                .filter(|_| {
679                                    !self.config.extra_env(None).contains_key("RUSTUP_TOOLCHAIN")
680                                        && std::env::var_os("RUSTUP_TOOLCHAIN").is_none()
681                                })
682                                .map(|it| ("RUSTUP_TOOLCHAIN".to_owned(), Some(it.to_string()))),
683                        )
684                        .collect(),
685
686                    _ => Default::default(),
687                };
688                info!("Using proc-macro server at {path}");
689
690                Some(ProcMacroClient::spawn(&path, &env).map_err(|err| {
691                    tracing::error!(
692                        "Failed to run proc-macro server from path {path}, error: {err:?}",
693                    );
694                    anyhow::format_err!(
695                        "Failed to run proc-macro server from path {path}, error: {err:?}",
696                    )
697                }))
698            }))
699        }
700
701        let watch = match files_config.watcher {
702            FilesWatcher::Client => vec![],
703            FilesWatcher::Server => project_folders.watch,
704        };
705        self.vfs_config_version += 1;
706        self.loader.handle.set_config(vfs::loader::Config {
707            load: project_folders.load,
708            watch,
709            version: self.vfs_config_version,
710        });
711        self.source_root_config = project_folders.source_root_config;
712        self.local_roots_parent_map = Arc::new(self.source_root_config.source_root_parent_map());
713
714        info!(?cause, "recreating the crate graph");
715        self.recreate_crate_graph(cause, switching_from_empty_workspace);
716
717        info!("did switch workspaces");
718    }
719
720    fn recreate_crate_graph(&mut self, cause: String, initial_build: bool) {
721        info!(?cause, "Building Crate Graph");
722        self.report_progress(
723            "Building CrateGraph",
724            crate::lsp::utils::Progress::Begin,
725            None,
726            None,
727            None,
728        );
729
730        // crate graph construction relies on these paths, record them so when one of them gets
731        // deleted or created we trigger a reconstruction of the crate graph
732        self.crate_graph_file_dependencies.clear();
733        self.detached_files = self
734            .workspaces
735            .iter()
736            .filter_map(|ws| match &ws.kind {
737                ProjectWorkspaceKind::DetachedFile { file, .. } => Some(file.clone()),
738                _ => None,
739            })
740            .collect();
741
742        let (crate_graph, proc_macro_paths) = {
743            // Create crate graph from all the workspaces
744            let vfs = &self.vfs.read().0;
745            let load = |path: &AbsPath| {
746                let vfs_path = vfs::VfsPath::from(path.to_path_buf());
747                self.crate_graph_file_dependencies.insert(vfs_path.clone());
748                vfs.file_id(&vfs_path).and_then(|(file_id, excluded)| {
749                    (excluded == vfs::FileExcluded::No).then_some(file_id)
750                })
751            };
752
753            ws_to_crate_graph(&self.workspaces, self.config.extra_env(None), load)
754        };
755        let mut change = ChangeWithProcMacros::default();
756        if initial_build || !self.config.expand_proc_macros() {
757            if self.config.expand_proc_macros() {
758                change.set_proc_macros(
759                    crate_graph
760                        .iter()
761                        .map(|id| (id, Err(ProcMacroLoadingError::NotYetBuilt)))
762                        .collect(),
763                );
764            } else {
765                change.set_proc_macros(
766                    crate_graph
767                        .iter()
768                        .map(|id| (id, Err(ProcMacroLoadingError::Disabled)))
769                        .collect(),
770                );
771            }
772
773            change.set_crate_graph(crate_graph);
774            self.analysis_host.apply_change(change);
775
776            self.finish_loading_crate_graph();
777        } else {
778            change.set_crate_graph(crate_graph);
779            self.fetch_proc_macros_queue.request_op(cause, (change, proc_macro_paths));
780        }
781
782        self.report_progress(
783            "Building CrateGraph",
784            crate::lsp::utils::Progress::End,
785            None,
786            None,
787            None,
788        );
789    }
790
791    pub(crate) fn finish_loading_crate_graph(&mut self) {
792        self.process_changes();
793        self.reload_flycheck();
794    }

796    pub(super) fn fetch_workspace_error(&self) -> Result<(), String> {
797        let mut buf = String::new();
798
799        let Some(FetchWorkspaceResponse { workspaces, .. }) =
800            self.fetch_workspaces_queue.last_op_result()
801        else {
802            return Ok(());
803        };
804
805        if workspaces.is_empty() && self.config.discover_workspace_config().is_none() {
806            stdx::format_to!(buf, "rust-analyzer failed to fetch workspace");
807        } else {
808            for ws in workspaces {
809                if let Err(err) = ws {
810                    stdx::format_to!(buf, "rust-analyzer failed to load workspace: {:#}\n", err);
811                }
812            }
813        }
814
815        if buf.is_empty() {
816            return Ok(());
817        }
818
819        Err(buf)
820    }

822    pub(super) fn fetch_build_data_error(&self) -> Result<(), String> {
823        let mut buf = String::new();
824
825        let Some(FetchBuildDataResponse { build_scripts, .. }) =
826            &self.fetch_build_data_queue.last_op_result()
827        else {
828            return Ok(());
829        };
830
831        for script in build_scripts {
832            match script {
833                Ok(data) => {
834                    if let Some(stderr) = data.error() {
835                        stdx::format_to!(buf, "{:#}\n", stderr)
836                    }
837                }
838                // io errors
839                Err(err) => stdx::format_to!(buf, "{:#}\n", err),
840            }
841        }
842
843        if buf.is_empty() { Ok(()) } else { Err(buf) }
844    }

846    fn reload_flycheck(&mut self) {
847        let _p = tracing::info_span!("GlobalState::reload_flycheck").entered();
848        let config = self.config.flycheck(None);
849        let sender = self.flycheck_sender.clone();
850        let invocation_strategy = match config {
851            FlycheckConfig::CargoCommand { .. } => {
852                crate::flycheck::InvocationStrategy::PerWorkspace
853            }
854            FlycheckConfig::CustomCommand { ref invocation_strategy, .. } => {
855                invocation_strategy.clone()
856            }
857        };
858
859        self.flycheck = match invocation_strategy {
860            crate::flycheck::InvocationStrategy::Once => {
861                vec![FlycheckHandle::spawn(
862                    0,
863                    sender,
864                    config,
865                    None,
866                    self.config.root_path().clone(),
867                    None,
868                )]
869            }
870            crate::flycheck::InvocationStrategy::PerWorkspace => {
871                self.workspaces
872                    .iter()
873                    .enumerate()
874                    .filter_map(|(id, ws)| {
875                        Some((
876                            id,
877                            match &ws.kind {
878                                ProjectWorkspaceKind::Cargo { cargo, .. }
879                                | ProjectWorkspaceKind::DetachedFile {
880                                    cargo: Some((cargo, _, _)),
881                                    ..
882                                } => (cargo.workspace_root(), Some(cargo.manifest_path())),
883                                ProjectWorkspaceKind::Json(project) => {
884                                    // Enable flychecks for json projects if a custom flycheck command was supplied
885                                    // in the workspace configuration.
886                                    match config {
887                                        FlycheckConfig::CustomCommand { .. } => {
888                                            (project.path(), None)
889                                        }
890                                        _ => return None,
891                                    }
892                                }
893                                ProjectWorkspaceKind::DetachedFile { .. } => return None,
894                            },
895                            ws.sysroot.root().map(ToOwned::to_owned),
896                        ))
897                    })
898                    .map(|(id, (root, manifest_path), sysroot_root)| {
899                        FlycheckHandle::spawn(
900                            id,
901                            sender.clone(),
902                            config.clone(),
903                            sysroot_root,
904                            root.to_path_buf(),
905                            manifest_path.map(|it| it.to_path_buf()),
906                        )
907                    })
908                    .collect()
909            }
910        }
911        .into();
912    }

916pub fn ws_to_crate_graph(
917    workspaces: &[ProjectWorkspace],
918    extra_env: &FxHashMap<String, Option<String>>,
919    mut load: impl FnMut(&AbsPath) -> Option<vfs::FileId>,
920) -> (CrateGraphBuilder, Vec<ProcMacroPaths>) {
921    let mut crate_graph = CrateGraphBuilder::default();
922    let mut proc_macro_paths = Vec::default();
923    for ws in workspaces {
924        let (other, mut crate_proc_macros) = ws.to_crate_graph(&mut load, extra_env);
925
926        crate_graph.extend(other, &mut crate_proc_macros);
927        proc_macro_paths.push(crate_proc_macros);
928    }
929
930    crate_graph.shrink_to_fit();
931    proc_macro_paths.shrink_to_fit();
932    (crate_graph, proc_macro_paths)
933}

935pub(crate) fn should_refresh_for_change(
936    path: &AbsPath,
937    change_kind: ChangeKind,
938    additional_paths: &[&str],
939) -> bool {
940    const IMPLICIT_TARGET_FILES: &[&str] = &["build.rs", "src/main.rs", "src/lib.rs"];
941    const IMPLICIT_TARGET_DIRS: &[&str] = &["src/bin", "examples", "tests", "benches"];
942
943    let file_name = match path.file_name() {
944        Some(it) => it,
945        None => return false,
946    };
947
948    if let "Cargo.toml" | "Cargo.lock" = file_name {
949        return true;
950    }
951
952    if additional_paths.contains(&file_name) {
953        return true;
954    }
955
956    if change_kind == ChangeKind::Modify {
957        return false;
958    }
959
960    // .cargo/config{.toml}
961    if path.extension().unwrap_or_default() != "rs" {
962        let is_cargo_config = matches!(file_name, "config.toml" | "config")
963            && path.parent().map(|parent| parent.as_str().ends_with(".cargo")).unwrap_or(false);
964        return is_cargo_config;
965    }
966
967    if IMPLICIT_TARGET_FILES.iter().any(|it| path.as_str().ends_with(it)) {
968        return true;
969    }
970    let parent = match path.parent() {
971        Some(it) => it,
972        None => return false,
973    };
974    if IMPLICIT_TARGET_DIRS.iter().any(|it| parent.as_str().ends_with(it)) {
975        return true;
976    }
977    if file_name == "main.rs" {
978        let grand_parent = match parent.parent() {
979            Some(it) => it,
980            None => return false,
981        };
982        if IMPLICIT_TARGET_DIRS.iter().any(|it| grand_parent.as_str().ends_with(it)) {
983            return true;
984        }
985    }
986    false
987}

989/// Similar to [`str::eq_ignore_ascii_case`] but instead of ignoring
990/// case, we say that `-` and `_` are equal.
991fn eq_ignore_underscore(s1: &str, s2: &str) -> bool {
992    if s1.len() != s2.len() {
993        return false;
994    }
995
996    s1.as_bytes().iter().zip(s2.as_bytes()).all(|(c1, c2)| {
997        let c1_underscore = c1 == &b'_' || c1 == &b'-';
998        let c2_underscore = c2 == &b'_' || c2 == &b'-';
999
1000        c1 == c2 || (c1_underscore && c2_underscore)
1001    })
